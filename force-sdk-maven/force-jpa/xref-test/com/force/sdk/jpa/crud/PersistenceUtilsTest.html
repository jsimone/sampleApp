51  public class PersistenceUtilsTest {
52  
53      // The EntityManager used to execute Force.com JPA queries
54      protected EntityManager em;
55      
56      // Schema handler which stores table and field names
57      protected MockForceStoreSchemaHandler mockSchemaHandler;
58      
59      @BeforeClass
60      public void classSetUp() throws IOException, ClassNotFoundException {
61          // Register all of our JPA entities.
62          // This would normally be done by PersistenceUtils in createSchema or loadSchema
63          mockSchemaHandler = new MockForceStoreSchemaHandler();
64          mockSchemaHandler.registerAllTables();
65          
66          Mockit.setUpMocks(MockForceTableMetaData.class, MockForceColumnMetaData.class,
67                              MockForceManagedConnection.class, mockSchemaHandler, MockForceSchemaWriter.class);
68  
69          // Setup the EntityManager just like normal
70          EntityManagerFactory emf = Persistence.createEntityManagerFactory("jpaQueryUnitTest");
71          em = emf.createEntityManager();
72      }
73      
74      @DataProvider
75      public Object[][] javaFieldProvider() {
76          return new Object[][]{
77                  // Tests API names for standard fields common to all entities (e.g. Id, Name)
78                  {"testCommonStandardField", User.class, "id", "id"},
79                  
80                  {"testStandardField", User.class, "emailEncodingKey", "emailEncodingKey"},
81                  {"testExistingCustomField", User.class, "existingCustomField", "existingCustomField__c"},
82                  {"testNewCustomField", UserCustomFields.class, "newCustomField", "newCustomField__c"},
83                  {"testStandardFieldOnSubclass", UserCustomFields.class, "emailEncodingKey", "emailEncodingKey"},
84                  {"testExistingCustomFieldOnSubclass", UserCustomFields.class, "existingCustomField", "existingCustomField__c"},
85                  
86                  {"testStandardRelationshipField", Account.class, "childOpportunities", "opportunities"},
87                  {"testCustomRelationshipField", BasicParentTestEntity.class, "childEntities",
88                      "BasicParentTestEntity_BasicChildTestEnti__r"},
89          };
90      }
91      
92      @Test(dataProvider = "javaFieldProvider")
93      public void testGetForceApiName(String testName, Class entityClass, String memberName, String expectedApiName) {
94          ObjectManager om = (ObjectManager) em.getDelegate();
95  
96          AbstractClassMetaData acmd = om.getMetaDataManager().getMetaDataForClass(entityClass, null);
97          AbstractMemberMetaData ammd = acmd.getMetaDataForMember(memberName);
98          
99          String apiName = PersistenceUtils.getForceApiName(ammd, om.getOMFContext());
100         assertEquals(apiName, expectedApiName, "Unexpected API name");
101     }
