56  /**
57   * 
58   * Unit test for schema deletion. Verify that if the purgeOnDeleteSchema persistence property is set that the metadata deploy
59   * options get set as well
60   *
61   * @author Jill Wetzler
62   */
63  public class SchemaDeletionTest {
64      
65      MockMetadataConnectionDelete mockMetadataConn;
66      
67      @BeforeClass
68      public void classSetUp() throws IOException, ClassNotFoundException {
69          // Register all of our JPA entities.
70          // This would normally be done by PersistenceUtils in createSchema or loadSchema
71          MockForceStoreSchemaHandler mockSchemaHandler = new MockForceStoreSchemaHandler();
72          mockSchemaHandler.registerAllTables();
73          
74          Mockit.setUpMocks(MockForceManagedConnection.class, mockSchemaHandler, MockAsyncResultsProcessor.class,
75                  MockTableImpl.class);
76      }
77      
78      @AfterClass(alwaysRun = true)
79      public void classTearDown() {
80          Mockit.tearDownMocks();
81      }
82      
83      @BeforeMethod
84      public void methodSetUp() {
85          // Each test method gets its own mocked MetadataConnection
86          mockMetadataConn = new MockMetadataConnectionDelete();
87          
88          Mockit.setUpMock(MetadataConnection.class, mockMetadataConn);
89      }
90      
91      /**
92       * 
93       * Mock class used to verify that the purgeOnDeleteSchema option is set
94       * when we expect it to be set.
95       *
96       * @author Jill Wetzler
97       */
98      @MockClass(realClass = MetadataConnection.class, instantiation = Instantiation.PerMockSetup)
99      public static class MockMetadataConnectionDelete {
100 
101         boolean purgeOnDeleteShouldBeSet;
102         
103         @Mock
104         public AsyncResult deploy(byte[] zipFile, DeployOptions deployOptions) throws ConnectionException {
105             Assert.assertEquals(deployOptions.getPurgeOnDelete(), purgeOnDeleteShouldBeSet,
106                     "The deploy options have purge on delete as " + deployOptions.getPurgeOnDelete()
107                     + " but we expected " + purgeOnDeleteShouldBeSet);
108             
109             AsyncResult result = new AsyncResult();
110             result.setDone(true);
111             return result;
112         }
113         
114         @Mock
115         public DeployResult checkDeployStatus(String processId) {
116             DeployResult result = new DeployResult();
117             result.setSuccess(true);
118             return result;
119         }
120         
121         public void setExpectedPurgeOnDelete(boolean purgeOnDelete) {
122             purgeOnDeleteShouldBeSet = purgeOnDelete;
123         }
124     }
125     
126     @Test
127     public void testDeleteSchema() {
128         Map<String, Object> props = new HashMap<String, Object>();
129         props.put("force.deleteSchema", true);
130         
131         mockMetadataConn.setExpectedPurgeOnDelete(false);
132         Persistence.createEntityManagerFactory("testDeleteSchema", props);
133 
134         props.put("force.purgeOnDeleteSchema", true);
135         mockMetadataConn.setExpectedPurgeOnDelete(true);
136         Persistence.createEntityManagerFactory("testDeleteSchema", props);
137 
138     }
