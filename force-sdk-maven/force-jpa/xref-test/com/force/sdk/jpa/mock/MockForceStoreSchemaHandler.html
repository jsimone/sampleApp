53  /**
54   * A mock ForceStoreSchemaHandler class.
55   * <p>
56   * This class is useful for query tests.

62  @MockClass(realClass = ForceStoreSchemaHandler.class,
63             stubs = { "initialize", "cacheDescribeSObjects", "clearDescribeSObjects" })
64  public final class MockForceStoreSchemaHandler {
65      
66      private Map<String, TableImpl> tables = new HashMap<String, TableImpl>();
67      
68      @Mock
69      public TableImpl getTable(AbstractClassMetaData cmd) {
70          // Any time the query builder wants table information, we'll
71          // serve it up here
72          return tables.get(PersistenceUtils.getEntityName(cmd));
73      }
74      
75      @Mock
76      public TableImpl addTable(AbstractClassMetaData acmd, ForceManagedConnection conn) {
77          return getTable(acmd);
78      }

80      // Iterates through the com.force.sdk.jpa.entities package and registers
81      // all JPA entities found within
82      @SuppressWarnings({ "rawtypes", "unchecked" })
83      public void registerAllTables() throws IOException, ClassNotFoundException {
84          ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
85          assertNotNull(classLoader, "Unable to load ClassLoader to register JPA entities.");
86          
87          // We'll load all JPA entities from the com.force.sdk.jpa.entities package
88          Enumeration<URL> entityFileURLs = classLoader.getResources("com/force/sdk/jpa/entities/");
89          assertTrue(entityFileURLs.hasMoreElements(), "Cannot find com/force/sdk/jpa/entities resource");
90  
91          // There should only be one base resource directory
92          File entityFileDir = new File(entityFileURLs.nextElement().getFile());
93          // load all classes in subpackages of the resource directory
94          List<File> entityList = getListEntityClasses(entityFileDir);
95          
96          // regex pattern for package path from class file path
97          Pattern pat = Pattern.compile("^.+/jpa/target/test-classes/(.+)/(.+)\\.class$");
98          
99          // Register all of our JPA entities here.
100         // This would normally be done by PersistenceUtils in createSchema or loadSchema
101         for (File entityFile : entityList) {
102             // get package name and class name from entity file path
103             Matcher mat = pat.matcher(entityFile.getCanonicalPath());
104             mat.matches();
105             String packageName = mat.group(1).replace("/", ".") + ".";
106             String entityClassName = packageName + mat.group(2);
107             Class entityClass = Class.forName(entityClassName);
108 
109             if (entityClass.isAnnotationPresent(Entity.class)) {
110                 registerTable(entityClass);
111             }
112         }
113         
114         // All persistence units load the Owner class so make sure it's registered
115         // (see ForceMetaDataManager.loadPersistenceUnit)
116         registerTable(Owner.class);
117     }
118     
119     @SuppressWarnings("rawtypes")
120     public void registerTable(Class<?> entityClass) {
121         TableImpl tableImpl;
122         CustomObject customObjectAnnotation = entityClass.getAnnotation(CustomObject.class);
123         if (customObjectAnnotation != null && customObjectAnnotation.virtualSchema()) {
124             tableImpl = MockPersistenceUtils.constructVirtualTableImpl(entityClass);
125         } else {
126             tableImpl = MockPersistenceUtils.constructTableImpl(entityClass);
127         }
128         
129         tables.put(entityClass.getSimpleName(), tableImpl);
130     }
131     
132     // search recursively and get a list of all subpackage entity class files
133     private List<File> getListEntityClasses(File file) {
134         List<File> entityList = new ArrayList<File>();
135         if (file.isDirectory()) {
136             for (File subFile : file.listFiles()) {
137                 entityList.addAll(getListEntityClasses(subFile));
138             }
139         } else if (file.getPath().endsWith(".class")) {
140             entityList.add(file);
141         }
142         return entityList;
143     }
