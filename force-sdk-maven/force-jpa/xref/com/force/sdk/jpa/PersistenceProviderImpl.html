27  package com.force.sdk.jpa;
28  
29  import static javax.jdo.Constants.PROPERTY_PERSISTENCE_MANAGER_FACTORY_CLASS;
30  
31  import java.util.*;
32  
33  import javax.persistence.EntityManagerFactory;
34  import javax.persistence.spi.PersistenceUnitInfo;
35  
36  import org.datanucleus.jpa.EntityManagerFactoryImpl;
37  import org.datanucleus.jpa.exceptions.*;
38  
39  /**
40   * 
41   * This class allows us to provide the ForceEntityManagerFactory.
42   *
43   * @author Jill Wetzler
44   */
45  public class PersistenceProviderImpl extends org.datanucleus.jpa.PersistenceProviderImpl {
46      
47      private static final String PERSISTENCE_PROVIDER_PROPERTY = "javax.persistence.provider";
48  
49      /**
50       * Method to create an EntityManagerFactory when running in J2EE.
51       * The container will have parsed the persistence.xml files to provide this PersistenceUnitInfo.
52       * @param unitInfo The "persistence-unit"
53       * @param properties EntityManagerFactory properties to override those in the persistence unit
54       * @return The EntityManagerFactory
55       */
56      @Override
57      public EntityManagerFactory createContainerEntityManagerFactory(PersistenceUnitInfo unitInfo, Map properties) {
58          try {
59              EntityManagerFactoryImpl emf = new ForceEntityManagerFactory(unitInfo, getOverrideMap(properties));
60              return emf;
61          } catch (NotProviderException npe) {
62              return null;
63          } catch (NoPersistenceUnitException npue) {
64              return null;
65          } catch (NoPersistenceXmlException npxe) {
66              return null;
67          }
68      }
69  
70      /**
71       * Method to create an EntityManagerFactory when running in J2SE.
72       * @param unitName Name of the "persistence-unit"
73       * @param properties EntityManagerFactory properties to override those in the persistence unit
74       * @return The EntityManagerFactory
75       */
76      @Override
77      public EntityManagerFactory createEntityManagerFactory(String unitName, Map properties) {
78          try {
79              EntityManagerFactoryImpl emf = new ForceEntityManagerFactory(unitName, getOverrideMap(properties));
80              return emf;
81          } catch (NotProviderException npe) {
82              return null;
83          } catch (NoPersistenceUnitException npue) {
84              return null;
85          } catch (NoPersistenceXmlException npxe) {
86              return null;
87          }
88      }
89  
90      @SuppressWarnings("unchecked")
91      private Map getOverrideMap(Map properties) {
92          Map<Object, Object> m = new HashMap<Object, Object>(properties != null ? properties : Collections.EMPTY_MAP);
93          m.put(PERSISTENCE_PROVIDER_PROPERTY, org.datanucleus.jpa.PersistenceProviderImpl.class.getName());
94          m.put(PROPERTY_PERSISTENCE_MANAGER_FACTORY_CLASS, "com.force.sdk.jdo.ForceJDOPersistenceManagerFactory");
95          return m;
96      }
97  }
