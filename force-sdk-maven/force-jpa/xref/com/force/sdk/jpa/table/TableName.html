27  package com.force.sdk.jpa.table;
28  
29  import org.datanucleus.exceptions.NucleusUserException;
30  import org.datanucleus.metadata.AbstractClassMetaData;
31  import org.datanucleus.metadata.InheritanceStrategy;
32  
33  import com.force.sdk.jpa.PersistenceUtils;

35  /**
36   * 
37   * Force.com objects have names that consist of a namespace, the base name of the object,
38   * and a suffix (like __c).  This class manages all of these elements.
39   *
40   * @author Fiaz Hossain
41   */
42  public final class TableName {
43  
44      private static final String NAME_SEPARATOR = "__";
45      
46      /**
47       * Enum of all possible table types.
48       */
49      public enum Type {
50          /**
51           * Standard Force.com table.
52           */
53          Standard(""),
54          /**
55           * Custom Force.com table.
56           */
57          Custom("c"),
58          /**
59           * Custom Force.com relationship.
60           */
61          CustomRelationship("r"),
62          /**
63           * Custom Force.com component.
64           */
65          CustomComponent("s"),
66          /**
67           * Force.com knowledge article.
68           */
69          KnowledgeArticle("ka"),
70          /**
71           * Force.com knowledge article version.
72           */
73          KnowledgeArticleVersion("kav"),
74          /**
75           * Force.com Chatter feed.
76           */
77          Feed("feed");
78          
79          private String suffix;
80          
81          Type(String suffix) {
82              this.suffix = suffix;
83          }
84          
85          String getSuffix() {
86              return suffix;
87          }
88          
89          static Type parse(String suffix) {
90              switch(suffix.length()) {
91              case 1:
92                  if (suffix.equals(Custom.suffix)) return Custom;
93                  else if (suffix.equals(CustomRelationship.suffix)) return CustomRelationship;
94                  else if (suffix.equals(CustomComponent.suffix)) return CustomComponent;
95                  break;
96              case 2:
97                  if (suffix.equals(KnowledgeArticle.suffix)) return KnowledgeArticle;
98                  break;
99              default:
100                 if (suffix.equals(KnowledgeArticleVersion.suffix)) return KnowledgeArticleVersion;
101                 if (suffix.equals(Feed.suffix)) return Feed;
102                 break;
103             }
104             throw new NucleusUserException("Unsupported custom object suffix: " + suffix);
105         }
106     }
107     
108     private final String name; // always the short name
109     private final boolean isCustom;
110     private final Type type;
111     private final String namespace;
112 
113     /**
114      * 
115      * creates a TableName object with all the components necessary for Force.com
116      * API access, including the name of the object, the namespace, and the suffix
117      * if needed.
118      * 
119      * @param namespace in most cases this will be the namespace of the org holding the API connection
120      * @param acmd  the metadata of the entity
121      * @return  The name to use for Force.com API access
122      */
123     public static TableName createTableName(String namespace, AbstractClassMetaData acmd) {
124         String tableName = null;
125         if (acmd.getTable() != null) {
126             tableName = acmd.getTable();
127         }
128         if (tableName == null) {
129             if (acmd.getInheritanceMetaData() != null
130                     && acmd.getInheritanceMetaData().getStrategy() == InheritanceStrategy.SUPERCLASS_TABLE) {
131                 AbstractClassMetaData sacmd;
132                 while ((sacmd = acmd.getSuperAbstractClassMetaData()) != null) {
133                     acmd = sacmd;
134                     if (acmd.getTable() != null) {
135                         tableName = acmd.getTable();
136                         break;
137                     }
138                 }
139             }
140         }
141         if (tableName != null) {
142             return new TableName(namespace, tableName, true);
143         } else {
144             return new TableName(namespace, PersistenceUtils.getEntityName(acmd), false);
145         }
146     }
147     
148     /**
149      * Creates a TableName object with the option to parse the full name to determine the suffix.
150      * 
151      * @param namespace  the namespace of the org, for defaulting purposes if the fullname
152      *                   does not contain a namespace
153      * @param fullName  the full name of an object, possibly includes a suffix and namespace
154      * @param parseForSuffix  whether we should parse the fullname or take the elements as is
155      */
156     private TableName(String namespace, String fullName, boolean parseForSuffix) {
157         if (parseForSuffix) {
158             String[] tokens = fullName.split(NAME_SEPARATOR); //either namespace__entity__suffix or entity__suffix
159             if (tokens.length > 3) {
160                 throw new NucleusUserException("Could not parse table: " + fullName);
161             }
162             
163             switch (tokens.length) {
164                 case 1:
165                     this.type = Type.Standard;
166                     this.name = fullName;
167                     this.namespace = namespace;
168                     break;
169                 case 2: //name__suffix
170                     String suffix = tokens[1].toLowerCase();
171                     this.type = Type.parse(suffix);
172                     this.name = tokens[0];
173                     this.namespace = namespace;
174                     break;
175                 case 3: //namespace__name__suffix
176                 default:
177                     String suf = tokens[2].toLowerCase();
178                     this.type = Type.parse(suf);
179                     this.namespace = tokens[0];
180                     this.name = tokens[1];
181                     break;
182             }
183         } else {
184             this.type = Type.Custom;
185             this.name = fullName;
186             this.namespace = namespace;
187         }
188         this.isCustom = this.type != Type.Standard;
189     }
190     
191     public String getName() {
192         return name;
193     }
194     
195     public String getNamespace() {
196         return namespace;
197     }
198     
199     /**
200      * Returns fully qualified Force.com API name for this {@code TableName}.
201      * 
202      * @return fully qualified Force.com API name
203      */
204     public String getForceApiName() {
205         if (isCustom) {
206             return PersistenceUtils.prependNamespace(namespace, String.format("%s__%s", name, type.suffix));
207         } else {
208             return name;
209         }
210     }
211     
212     /**
213      * for denoting whether an object is user created or standard in Force.com.
214      * 
215      * @return true if this object is custom
216      */
217     public boolean isCustom() {
218         return isCustom;
219     }
220     
221     public Type getType() {
222         return type;
223     }
224         
225     @Override
226     public String toString() {
227         return getName();
228     }
