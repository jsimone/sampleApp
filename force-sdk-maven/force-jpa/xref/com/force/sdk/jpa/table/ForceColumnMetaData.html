55  /**
56   * 
57   * Metadata for fields on Force.com objects. This class is what knows how to turn fields
58   * in your @Entity java classes into CustomFields that can be read by the Force.com
59   * metadata API. Fields are stored off on the ForceTableMetaData object and optionally
60   * handed to the ForceSchemaWriter
61   *
62   * @author Fiaz Hossain
63   * @author Jill Wetzler
64   */
65  public class ForceColumnMetaData extends ForceMetaData {
66      
67      private static final Pattern PICK_PATTERN = Pattern.compile(",");

69      /**
70       * fields that are standard on all Force.com objects.
71       */
72      public static final Set<String> STANDARD_FIELDS = new HashSet<String>(Arrays.<String>asList(
73              "id",
74              "name",
75              "ownerid",
76              "isdeleted",
77              "createdbyid",
78              "createddate",
79              "lastmodifiedbyid",
80              "lastmodifieddate",
81              "systemmodstamp"));
82      
83      private final List<AbstractMemberMetaData> fieldsToAdd = new ArrayList<AbstractMemberMetaData>();
84      private final ForceStoreManager storeManager;
85      private final String fieldFormatString;
86  
87      private final List<CustomField> fields = new ArrayList<CustomField>();

89      /**
90       * Create the column metadata for field registration and possible schema creation.
91       * 
92       * @param cmd  the class this column belongs to
93       * @param table  the table where this column is found
94       * @param storeManager  the store manager
95       */
96      public ForceColumnMetaData(AbstractClassMetaData cmd, TableImpl table, ForceStoreManager storeManager) {
97          super(cmd, table);
98          fieldFormatString = tableImpl.getTableName().getForceApiName() + ".%s";
99          this.storeManager = storeManager;
100     }
101     
102     /**
103      * Add all of the fields for this table to the object.  We may need to merge with any existing fields
104      * in the case of Entities that extend other entities.  There is only one CustomObject representing all
105      * of the entities and we need the intersection of their fields
106      * 
107      * @param customObject CustomObject
108      */
109     public void addFieldsToObject(CustomObject customObject) {
110         if (fields.size() > 0) {
111             Collection<CustomField> mergedResult = mergeFields(customObject.getFields(), fields);
112             customObject.setFields(mergedResult.toArray(new CustomField[mergedResult.size()]));
113             fields.clear();
114         }
115     }
116     
117     private Collection<CustomField> mergeFields(CustomField[] existing, List<CustomField> toAdd) {
118         if (existing == null || existing.length == 0) return toAdd;
119         Map<String, CustomField> result = new HashMap<String, CustomField>();
120         for (CustomField cf : existing) {
121             result.put(String.format(fieldFormatString, cf.getFullName()), cf);
122         }
123         for (CustomField cf : toAdd) {
124             result.put(String.format(fieldFormatString, cf.getFullName()), cf);
125         }
126         return result.values();
127     }

129     /**
130      * 
131      * Checks to see if columns should be created according to persistence properties, and
132      * if so, creates CustomField objects and stores them on the parent table to
133      * be created later by the {@link com.force.sdk.jpa.schema.ForceSchemaWriter}.
134      * 
135      * @param namespace  the namespace of the organization creating schema
136      */
137     public void createFieldSchema(String namespace) {
138         updateListOfMissingFields();
139         if (fieldsToAdd.size() > 0) {
140             if (storeManager.isAutoCreateColumns()) {
141                 List<AbstractClassMetaData> invalidateList = createColumns(namespace);
142                 for (AbstractClassMetaData tcmd : invalidateList) {
143                     storeManager.getTable(tcmd).setIsValid(false);
144                 }
145             } else {
146                 StringBuilder msg = new StringBuilder(1024);
147                 msg.append("Field does not exist in force.com table and datanucleus.autoCreateColumns is false, entity: ")
148                    .append(cmd.getName()).append(" table: ")
149                    .append(tableImpl.getTableName().getForceApiName()).append(" fields: [");
150                 int count = 0;
151                 for (AbstractMemberMetaData ammd : fieldsToAdd) {
152                     if (count++ > 0) msg.append(", ");
153                     msg.append(ammd.getName());
154                 }
155                 msg.append("]");
156                 if (storeManager.isAutoCreateWarnOnError()) {
157                     LOGGER.warn(msg.toString());
158                 } else {
159                     throw new NucleusUserException(msg.toString());
160                 }
161             }
162         }
163     }

165     private void validateColumn(AbstractMemberMetaData ammd, OMFContext omf) {
166         Map<String, String> extensions = PersistenceUtils.getForceExtensions(ammd);
167         FieldType fieldType = PersistenceUtils.getFieldTypeFromForceAnnotation(extensions);
168         AccessibleObject ao = (AccessibleObject) ammd.getMemberRepresented();
169         if (ao.isAnnotationPresent(OneToOne.class)) {
170             throw new NucleusUserException("@OneToOne relationship is not supported. Please combine fields into a single object."
171                                             + " Offending field: " + ammd.getFullFieldName());
172         } else if (ao.isAnnotationPresent(OneToMany.class)) {
173             if (ammd.getMappedBy() == null) {
174                 throw new NucleusUserException("@OneToMany relationship requires the 'mappedBy' attribute."
175                                                 + " Please add a foreign key field on child object and use that field name"
176                                                 + " for mappedBy attribute."
177                                                 + " Offending field: " + ammd.getFullFieldName());
178             }
179         } else if (ao.isAnnotationPresent(ManyToMany.class)) {
180             throw new NucleusUserException("@ManyToMany relationship is not supported."
181                                             + " Please use a junction object with two @ManyToOne relationships."
182                                             + " Offending field: " + ammd.getFullFieldName());
183         }
184         if (ammd.getJoinMetaData() != null) {
185             throw new NucleusUserException("@JoinTable is not supported. Offending field: " + ammd.getFullFieldName());
186         }
187         if (ammd.getColumnMetaData() != null && ammd.getColumnMetaData().length > 0) {
188             for (ColumnMetaData md : ammd.getColumnMetaData()) {
189                 if (md.getTarget() != null && !md.getTarget().equalsIgnoreCase("id")) {
190                     throw new NucleusUserException("Referenced foreign key colum name must be 'ID'."
191                                                     + " Offending field: " + ammd.getFullFieldName());
192                 } else if ("CLOB".equals(md.getJdbcType())) {
193                     throw new NucleusUserException("@Clob field type is not supported."
194                                                     + " Offending field: " + ammd.getFullFieldName());
195                 }
196             }
197             if (ammd.getTable() != null) {
198                 throw new NucleusUserException("Table cannot be specified at a column level."
199                                                 + " Offending field: " + ammd.getFullFieldName());
200             }
201         }
202         if (FieldType.Picklist == fieldType || ao.isAnnotationPresent(Enumerated.class)) {
203             if (!ammd.getType().isEnum() && !(ammd.getType().isArray() && ammd.getType().getComponentType().isEnum())) {
204                 // This is a non-strict picklist. Non-strict picklists allow only String or String[]
205                 if (ammd.getType() != String.class && ammd.getType() != String[].class) {
206                     throw new NucleusUserException("Non-strict picklist can be of type String or String[] only."
207                                                     + " Offending field: " + ammd.getFullFieldName());
208                 }
209                 String allValues = extensions.get(PersistenceUtils.PICKLIST_VALUE_FIELD_PREFIX + "value");
210                 if (allValues == null) {
211                     throw new NucleusUserException("Non-strict picklist requires @PicklistValue annotation."
212                                                     + " Offending field: " + ammd.getFullFieldName());
213                 }
214                 if (PersistenceUtils.isOrdinalEnum(ammd)) {
215                     throw new NucleusUserException("Non-strict picklist does not support EnumType.ORDINAL."
216                                                     + " Offending field: " + ammd.getFullFieldName());
217                 }
218             }
219         }
220         if (ammd.getCollection() != null || ammd.getMap() != null) {
221             /**
222              * We do not want JPA set cascade delete for MD relationships.
223              * It's a no-op but quite inefficient anyways since JPA tries to delete all children individually too.
224              * Also, In the case of AllOrNothing transactions trying to delete all children
225              * is an error not just an inefficient call.
226              * 
227              * So check if this is a MasterDetail relationship. If it is then unset Cascade delete.
228              */
229             FieldType type =
230                 PersistenceUtils.getFieldTypeFromForceAnnotation(
231                         PersistenceUtils.getForceExtensions(ammd.getRelatedMemberMetaData(omf.getClassLoaderResolver(null))[0]));
232             if (type != null && type == FieldType.MasterDetail) {
233                 if (ammd.getCollection() != null) {
234                     ammd.getCollection().setDependentElement(false);
235                 } else if (ammd.getMap() != null) {
236                     ammd.getMap().setDependentValue(false);
237                 }
238             }
239         }
240     }

242     private void updateListOfMissingFields() {
243         AbstractMemberMetaData[] membersInThisClass =  cmd.getManagedMembers();
244         if (membersInThisClass != null && membersInThisClass.length > 0) {
245             for (AbstractMemberMetaData ammd : membersInThisClass) {
246                 validateColumn(ammd, storeManager.getOMFContext());
247                 AbstractClassMetaData relatedEntity =
248                     storeManager.getMetaDataManager().getMetaDataForEntityName(ammd.getType().getSimpleName());
249                 if (relatedEntity != null && relatedEntity.isEmbeddedOnly()) {
250                     // Add all embedded fields to embedding entity
251                     for (AbstractMemberMetaData eammd : ammd.getEmbeddedMetaData().getMemberMetaData()) {
252                         addFieldToListIfAbsent(eammd, storeManager);
253                     }
254                 } else {
255                     addFieldToListIfAbsent(ammd, storeManager);
256                 }
257             }
258         }
259         // If there is a discriminator column add that too
260         if (cmd.getSuperAbstractClassMetaData() == null && cmd.getDiscriminatorMetaData() != null
261                 && cmd.getDiscriminatorMetaData().getColumnMetaData() != null) {
262             ColumnMetaData colmd = cmd.getDiscriminatorMetaData().getColumnMetaData();
263             addFieldToListIfAbsent(new ForceMemberMetaData(colmd, colmd.getName()) {
264                 @Override
265                 public Class<?> getType() {
266                     return String.class;
267                 }
268             }, storeManager);
269         }
270     }
271     
272     private void addFieldToListIfAbsent(AbstractMemberMetaData ammd, ForceStoreManager storeMgr) {
273         if (PersistenceUtils.isNonPersistedColumn(ammd)) return;
274         if (!tableImpl.isValid() || storeManager.isForDelete()
275                 || tableImpl.getColumnByForceApiName(
276                         PersistenceUtils.getForceApiName(ammd, storeManager.getOMFContext())) == null) {
277             fieldsToAdd.add(ammd);
278         }
279     }

281     /**
282      * registers the missing columns for creations.
283      * 
284      * @param namespace  the namespace of the organization creating the schema
285      * @return all the AbstractClassMetaData objects that need to invalidated
286      */
287     private List<AbstractClassMetaData> createColumns(String namespace) {
288         
289         List<AbstractClassMetaData> ret = new ArrayList<AbstractClassMetaData>();
290         if (fieldsToAdd != null && fieldsToAdd.size() > 0) {
291             ret.add(cmd);
292 
293             for (int column = 0; column < fieldsToAdd.size(); column++) {
294                 AbstractMemberMetaData ammd = fieldsToAdd.get(column);
295                 if (Collection.class.isAssignableFrom(ammd.getType()) || Map.class.isAssignableFrom(ammd.getType())) {
296                     // skip this, we'll make the relationship the other way
297                     // but we need to add the relationship column for @OneToMany
298                     // however, can't poke into the child object yet because the filed might not have been created yet
299                     continue;
300                 }
301                 Map<String, String> extensions = PersistenceUtils.getForceExtensions(ammd);
302                 String fieldName = PersistenceUtils.getForceApiName(ammd, storeManager.getOMFContext());
303                 AbstractClassMetaData relatedClass = ammd.getType() != null
304                         ? storeManager.getMetaDataManager().getMetaDataForClass(ammd.getType(),
305                                 storeManager.getOMFContext().getClassLoaderResolver(null)) : null;
306                 // Try the first column if specified
307                 ColumnMetaData[] colmds = ammd.getColumnMetaData();
308                 ColumnMetaData colmd = null;
309                 FieldType type = PersistenceUtils.getFieldTypeFromForceAnnotation(extensions);
310                 Integer scale = getIntegerFromForceAnnotation(extensions, "scale");
311                 Integer precision = getIntegerFromForceAnnotation(extensions, "precision");
312                 Integer length = getIntegerFromForceAnnotation(extensions, "length");
313                 Integer startValue = getIntegerFromForceAnnotation(extensions, "startValue");
314                 boolean enableFeeds = getBooleanFromForceAnnotation(extensions, "enableFeeds");
315                 boolean externalId = getBooleanFromForceAnnotation(extensions, "externalId");
316                 String childRelationshipName = extensions.get("childRelationshipName");
317                 String label = extensions.get("label");
318                 String description = extensions.get("description");
319                 String formula = extensions.get("formula");
320                 TreatBlanksAs treatBlanksAs = getTreatBlanksAsFromForceAnnotation(extensions);
321                 boolean allowNulls = true;
322                 boolean unique = false;
323                 if (colmds != null && colmds.length > 0) {
324                     colmd = colmds[0];
325                     if (type == null) {
326                         type = getFieldType(colmd, ammd);
327                     }
328                     if (precision == null) {
329                         scale = colmd.getScale();
330                         precision = colmd.getLength();
331                     }
332                     if (length == null) {
333                         length = colmd.getLength();
334                     }
335                     allowNulls = colmd.getAllowsNull();
336                     if (!allowNulls) {
337                         // Too bad DataNucleus tries to be smart and add this annotation for various types.
338                         // So, we go back to the actual annotations and make sure the user actually added
339                         // the @Column(nullable=true)
340                         Column columnAnnotation = PersistenceUtils.getMemberAnnotation(ammd.getMemberRepresented(), Column.class);
341                         if (columnAnnotation != null) {
342                             allowNulls = columnAnnotation.nullable();
343                         } else {
344                             allowNulls = true;
345                         }
346                     }
347                     unique = colmd.getUnique();
348                 }
349                 
350                 if (startValue != null && type != FieldType.AutoNumber) {
351                     throw new NucleusUserException("startValue attribute is only supported for AutoNumber fields types,"
352                                                     + " field: " + ammd.getName()
353                                                     + " on entity: " + tableImpl.getTableName().getName());
354                 }
355                 
356                 // This will handle bare bones @ManyToOne definitions
357                 if (type == null && PersistenceUtils.isRelationship(ammd)) {
358                     type = FieldType.Lookup;
359                 }
360 
361                 if (STANDARD_FIELDS.contains(fieldName.toLowerCase())) {
362                     continue;
363                 }
364 
365                 boolean isLookupToStandardObject = false;
366                 if (type == FieldType.Lookup || type == FieldType.MasterDetail) {
367                     if (relatedClass == null) {
368                         throw new NucleusDataStoreException("Undefined entity type for"
369                                                             + " field: " + ammd.getName()
370                                                             + " on entity: " + tableImpl.getTableName().getName());
371                     }
372                     TableImpl table1 = ((ForceStoreSchemaHandler) storeManager.getSchemaHandler()).getTable(relatedClass);
373                     isLookupToStandardObject = !table1.getTableName().isCustom();
374                     ret.add(relatedClass);
375                 }
376 
377                 // we don't want to add __c if this is a lookup to a standard object, because that will screw up the relationship
378                 TableName columnTypeName = relatedClass != null ? TableName.createTableName(namespace, relatedClass) : null;
379                 if (isLookupToStandardObject) {
380                     fieldName = removeCustomThingSuffix(fieldName);
381                 }
382 
383                 CustomField field = new CustomField();
384 
385                 if (type != null) {
386                     field.setType(type);
387                 } else {
388                     setFieldType(field, ammd);
389                 }
390                 setOtherValues(field, length, precision, scale, ammd, columnTypeName, fieldName,
391                                     tableImpl.getTableName().getName(), childRelationshipName, formula, extensions);
392                 field.setTrackFeedHistory(enableFeeds);
393                 field.setExternalId(externalId);
394                 
395                 if (isLookupToStandardObject) {
396                     //yet we need to have __c on the actual name of the field
397                     field.setFullName(String.format("%s%s", fieldName, ColumnImpl.CUSTOM_THING_SUFFIX));
398                 } else {
399                     field.setFullName(fieldName);
400                 }
401                 
402                 field.setLabel(label != null ? label : fieldName);
403                 field.setDescription(description != null ? description : field.getLabel());
404                 if (!allowNulls) {
405                     field.setRequired(true);
406                 }
407                 if (unique || ammd.isUnique()) {
408                     field.setUnique(true);
409                 }
410                 if (startValue != null) {
411                     field.setStartingNumber(startValue);
412                 }
413                 if (formula != null) {
414                     field.setFormula(formula);
415                     field.setFormulaTreatBlanksAs(treatBlanksAs);
416                 }
417                 addField(field);
418             }
419         }
420         return ret;
421     }

423     private void addField(CustomField field) {
424         if (isReadOnlyFields) {
425             throw new NucleusUserException("Cannot add custom"
426                                             + " field: " + field.getFullName()
427                                             + " to readOnlySchema object: " + tableImpl.getTableName().getName());
428         }
429         fields.add(field);
430     }

432     private FieldType getFieldType(ColumnMetaData md, AbstractMemberMetaData ammd) {
433         if (PersistenceUtils.isRelationship(ammd)) {
434             return FieldType.Lookup;
435         }
436         
437         Class<?> clazzType = ammd.getType();
438         if (clazzType.isEnum()) {
439             return FieldType.Picklist;
440         }
441         String type = md.getJdbcType() != null ? md.getJdbcType() : md.getSqlType();
442         if (type == null) {
443             return null;
444         } else {
445             type = type.toLowerCase();
446         }
447         
448         if (type.equals("varchar")) {
449             return FieldType.Text;
450         } else if (type.equals("integer") || type.equals("number") || type.equals("numeric")) {
451             return FieldType.Number;
452         } else if (type.equals("date")) {
453             return FieldType.Date;
454         } else if (type.equals("timestamp") || type.equals("time")) {
455             return FieldType.DateTime;
456         } else if (type.equals("boolean")) {
457             return FieldType.Checkbox;
458         } else if (type.equals("clob")) {
459             return FieldType.LongTextArea;
460         } else {
461             throw new NucleusUserException("Unsupported column type: " + type);
462         }
463     }

465     private void setFieldType(CustomField field, AbstractMemberMetaData ammd) {
466         Class<?> type = ammd.getType();
467         if (type == String.class) {
468             field.setType(FieldType.Text);
469         } else if (type == Integer.class || type == Integer.TYPE
470                 || type == Long.class || type == Long.TYPE
471                 || type == Double.class || type == Double.TYPE
472                 || type == Float.class || type == Float.TYPE
473                 || type == Short.class || type == Short.TYPE
474                 || type == BigInteger.class) {
475             field.setType(FieldType.Number);
476         } else if (type == Date.class) {
477             field.setType(FieldType.Date);
478         } else if (type == Calendar.class || type == GregorianCalendar.class || type == Time.class) {
479             field.setType(FieldType.DateTime);
480         } else if (type == Boolean.class || type == Boolean.TYPE) {
481             field.setType(FieldType.Checkbox);
482             field.setDefaultValue("false");
483         } else if (type == BigDecimal.class) {
484             field.setType(FieldType.Currency);
485         } else if (type == URL.class) {
486             field.setType(FieldType.Url);
487         } else if (PersistenceUtils.isMultiPicklist(ammd)) {
488             field.setType(FieldType.MultiselectPicklist);
489             field.setVisibleLines(3);  // TODO: need to make this a variable
490         } else if (type == Byte.class || type == Byte.TYPE) {
491             field.setType(FieldType.Text);
492         } else if (type == Character.class || type == Character.TYPE) {
493             field.setType(FieldType.Text);
494         } else {
495             throw new NucleusUserException("Unsupported column for"
496                                             + " Java type: " + type
497                                             + " for entity: " + cmd.getName()
498                                             + " field: " + ammd.getName());
499         }
500     }

502     private void setOtherValues(CustomField field, Integer length, Integer precision, Integer scale,
503             AbstractMemberMetaData ammd, TableName columnTypeName, String fieldName,
504             String tableName, String childRelationshipName, String formula, Map<String, String> extensions) {
505         Class<?> type = ammd.getType();
506         switch (field.getType()) {
507         case Text:
508             if (formula == null) {
509                 field.setLength(length != null ? length : 255);
510             }
511             break;
512 
513         case LongTextArea:
514         case Html:
515             field.setLength(length != null ? length : 32000);
516             field.setVisibleLines(field.getType() == FieldType.Html ? 25 : 3);
517             break;
518             
519         case Number:
520         case Currency:
521         case Percent:
522          if (type == Integer.class || type == Integer.TYPE) {
523              field.setPrecision(11);
524              field.setScale(0);
525          } else if (type == Long.class || type == Long.TYPE) {
526              field.setPrecision(18);
527              field.setScale(0);
528          } else if (type == Double.class || type == Double.TYPE) {
529              field.setPrecision(16);
530              field.setScale(2);
531          } else if (type == Float.class || type == Float.TYPE) {
532              field.setPrecision(16);
533              field.setScale(2);
534          } else if (type == Short.class || type == Short.TYPE) {
535              field.setPrecision(6);
536              field.setScale(0);
537          } else if (type == BigInteger.class) {
538              field.setPrecision(18);
539              field.setScale(0);
540          } else if (type == BigDecimal.class) {
541              field.setPrecision(16);
542              field.setScale(2);
543          }
544          break;
545 
546         case Picklist:
547             setPicklistType(field, type, extensions, PersistenceUtils.isOrdinalEnum(ammd));
548             break;
549 
550         case MultiselectPicklist:
551             setPicklistType(field, type.getComponentType(), extensions, PersistenceUtils.isOrdinalEnum(ammd));
552             break;
553             
554         case Lookup:
555         case MasterDetail:
556             String fieldNameNoSuffix = fieldName;
557             if (fieldName.endsWith(ColumnImpl.CUSTOM_THING_SUFFIX)) {
558                 fieldNameNoSuffix = fieldName.substring(0, fieldName.lastIndexOf(ColumnImpl.CUSTOM_THING_SUFFIX));
559             }
560             field.setReferenceTo(columnTypeName.getForceApiName());
561             // if we just made the childRelationshipName == tableName
562             // the MD API would fail to create the field if you have more than one lookup on that object
563             // so for a lookup from entity1__c to entity2__c, your childRelationshipName is entity2_entity1s
564             // so we're less likely to collide
565             String relationshipName = childRelationshipName != null
566                     ? childRelationshipName : String.format("%s_%ss", fieldNameNoSuffix, tableName);
567             if (relationshipName.length() > 40) {
568                 relationshipName = relationshipName.substring(0, 40);
569             }
570             field.setRelationshipName(relationshipName);
571             field.setRelationshipLabel(fieldNameNoSuffix);
572             break;
573 
574         default:
575         }
576     }

578     private void setPicklistType(CustomField field, Class<?> type, Map<String, String> extensions, boolean isOrdinal) {
579         String[] values;
580         if (type == String.class || type == String[].class) {
581             // This is for non-strict picklists, we already validated in ForceMetaDataListener
