81      /**
82       * create a store manager for use with the Force.com API. Setup the API connection
83       * configs, set some default properties and read in values from persistence.xml
84       * 
85       * @param clr  the class loader resolver
86       * @param omfContext the corresponding object manager factory context
87       * @throws NoSuchFieldException thrown if there is a problem setting up the plugin manager
88       * @throws IllegalAccessException thrown if there is a problem setting up the plugin manager
89       */
90      public ForceStoreManager(ClassLoaderResolver clr, OMFContext omfContext) throws NoSuchFieldException, IllegalAccessException {
91          super(FORCE_KEY, clr, omfContext);
92      
93          String endpoint = getConnectionURL();
94          
95          // Grab the connection information from persistence.xml.  If none exists,
96          // we'll use the unit name (see ForceMetaDataManager.loadPersistenceUnit
97          // and ForceConnectionFactory.createManagedConnection).
98          if (endpoint != null) {
99              config = new ForceConnectorConfig();
100             
101             // Treat any url starting with force:// as a connection url
102             if (endpoint.startsWith(FORCE_PREFIX)) {
103                 config.setConnectionUrl(endpoint);
104             // Any other endpoint, we'll treat as a normal Force.com API url
105             } else {
106                 config.setAuthEndpoint(endpoint);
107                 config.setUsername(getConnectionUserName());
108                 config.setPassword(getConnectionPassword());
109             }
110         }
111         
112         setCustomPluginManager();
113         
114         // Handler for metadata
115         metadataListener = new ForceMetaDataListener(this);
116         omfContext.getMetaDataManager().registerListener(metadataListener);
117 
118         // Handler for persistence process
119         persistenceHandler2 = new ForcePersistenceHandler(this);
120         
121         // Store schema handler
122         schemaHandler = new ForceStoreSchemaHandler(this);
123         
124         PersistenceConfiguration conf = omfContext.getPersistenceConfiguration();
125         boolean autoCreateSchema = conf.getBooleanProperty("datanucleus.autoCreateSchema");
126         if (autoCreateSchema) {
127             autoCreateTables = true;
128             autoCreateColumns = true;
129         } else {
130             autoCreateTables = conf.getBooleanProperty("datanucleus.autoCreateTables");
131             autoCreateColumns = conf.getBooleanProperty("datanucleus.autoCreateColumns");
132         }
133         autoCreateWarnOnError = conf.getBooleanProperty("datanucleus.autoCreateWarnOnError");
134         forDelete = conf.getBooleanProperty("force.deleteSchema");
135         boolean purgeOnDelete = conf.getBooleanProperty("force.purgeOnDeleteSchema");
136         
137         schemaWriter = new ForceSchemaWriter(new SchemaDeleteProperty(forDelete, purgeOnDelete));
138         schemaCreateClient = conf.getBooleanProperty("force.schemaCreateClient");
139         
140         // how often should the evictor run
141         poolTimeBetweenEvictionRunsMillis = conf.getIntProperty("datanucleus.connectionPool.timeBetweenEvictionRunsMillis");
142         if (poolTimeBetweenEvictionRunsMillis == 0) {
143             poolTimeBetweenEvictionRunsMillis = 15 * 1000; // default, 15 secs
144         }
145          
146         // how long may a connection sit idle in the pool before it may be evicted
147         poolMinEvictableIdleTimeMillis = conf.getIntProperty("datanucleus.connectionPool.minEvictableIdleTimeMillis");
148         if (poolMinEvictableIdleTimeMillis == 0) {
149             poolMinEvictableIdleTimeMillis = 30 * 1000; // default, 30 secs
150         }
151         
152         // setup optimistic enabled
153         Object isEnabled = omfContext.getPersistenceConfiguration().getProperty("datanucleus.Optimistic");
154         enableOptimisticTransactions = isEnabled != null && (isEnabled instanceof Boolean && isEnabled.equals(Boolean.TRUE)
155                 || isEnabled instanceof String && "true".equals(isEnabled));
156         
157         logConfiguration();
158     }

160     private void setCustomPluginManager() throws NoSuchFieldException, IllegalAccessException {
161         PluginManager pluginMgr = omfContext.getPluginManager();
162         Field registryField = PluginManager.class.getDeclaredField("registry");
163         registryField.setAccessible(true);
164         registryField.set(pluginMgr, new ForcePluginRegistry((PluginRegistry) registryField.get(pluginMgr)));
165     }

300     /**
301      * Before entities are registered or created we do describes on the entities to determine
302      * entity shape and whether the entities already exist.
303      * 
304      * @param classes a collection of class metadata to cache describe results for
305      */
306     void preInitialiseFileMetaData(Collection<AbstractClassMetaData> classes) {
307         ForceManagedConnection mconn = createConnection();
308         try {
309             ((ForceStoreSchemaHandler) getSchemaHandler()).cacheDescribeSObjects(classes, mconn);
310         } finally {
311             mconn.release();
312         }
313     }

315     /**
316      * clears out the describe calls for all entities in this application.
317      */
318     void postInitialiseFileMetaData() {

320     }
