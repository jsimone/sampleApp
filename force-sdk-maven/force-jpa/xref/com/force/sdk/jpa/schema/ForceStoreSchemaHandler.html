93      public StoreSchemaData getSchemaData(Object conn, String objName, Object[] values) {
94          final TableImpl table = getTable(objName);
95          return table == null ? null : new StoreSchemaData() {

99                  return table.getColumnByForceApiName(name);

110     /**
111      * makes a describe call for each class in the collection and caches the results.
112      * 
113      * @param classes a collection of class metadata to describe
114      * @param mconn the managed connection to the Force.com APIs
115      */
116     public void cacheDescribeSObjects(Collection<AbstractClassMetaData> classes, ForceManagedConnection mconn) {
117         ArrayList<String> sObjectNames = new ArrayList<String>(classes.size());
118         try {
119             Iterator<AbstractClassMetaData> iter = classes.iterator();
120             while (iter.hasNext()) {
121                 AbstractClassMetaData cmd = iter.next();
122                 // Similar to PersistenceUtils.createObjectSchema
123                 if (PersistenceUtils.hasNoSchema(cmd)) continue;
124                 TableName tableName = TableName.createTableName(mconn.getNamespace(), cmd);
125                 if (cmd.getSuperAbstractClassMetaData() == null) {
126                    sObjectNames.add(tableName.getForceApiName());
127                 }
128             }
129             // Setup the initialization cache
130             sObjectResults = new HashMap<String, DescribeSObjectResult>(sObjectNames.size());
131 
132             for (int siz = 0; siz < sObjectNames.size(); siz += MAX_DESCRIBE_SOBJECT) {
133                 int len = sObjectNames.size() - siz > MAX_DESCRIBE_SOBJECT ? MAX_DESCRIBE_SOBJECT : sObjectNames.size() - siz;
134                 List<String> nameList = sObjectNames.subList(siz, siz + len);
135                 if (LOGGER.isDebugEnabled()) {
136                     StringBuilder sb = new StringBuilder();
137                     for (String s : nameList) {
138                         if (sb.length() > 0) sb.append(", ");
139                         sb.append(s);
140                     }
141                     LOGGER.debug("DescribeSObjects: " + sb);
142                 }
143                 DescribeSObjectResult[] results =
144                     ((PartnerConnection) mconn.getConnection()).describeSObjects(nameList.toArray(new String[nameList.size()]));
145                 for (int i = 0; i < results.length; i++) {
146                     sObjectResults.put(nameList.get(i), results[i]);
147                 }
148             }
149         } catch (InvalidSObjectFault ie) {
150             // Well we seem to be missing some objects so we just fall back to single item describe as we go along
151             LOGGER.trace("Describes sObjects failed: ", ie);
152         } catch (ConnectionException x) {
153             throw new NucleusException(x.getMessage(), x);
154         }
155     }

165     /**
166      * get the table impl for a particular entity name from the cache.
167      * 
168      * @param objName the name of the object to retrieve a table for
169      * @return the tableImpl, null if it hasn't been registered

171     public TableImpl getTable(String objName) {
172         return getTable(storeManager.getMetaDataManager().getMetaDataForEntityName(objName));
173     }

175     /**
176      * get the table impl for a particular entity from the cache.
177      * 
178      * @param acmd the metadata to retrieve table info for 
179      * @return the tableImpl, null if it hasn't been registered
180      */
181     public TableImpl getTable(AbstractClassMetaData acmd) {
182         if (acmd == null) return null;
183         String objName = PersistenceUtils.getEntityName(acmd);
184         return entityTables.get(objName.toLowerCase());
185     }
186     
187     /**
188      * register a table with the cache, create its TableName and TableImpl.
189      * 
190      * @param acmd the class metadata of the entity to register
191      * @param conn the managed connection to Force.com APIs, used to get at the namespace
192      * @return the created TableImpl
193      */
194     public TableImpl addTable(AbstractClassMetaData acmd, ForceManagedConnection conn) {
195         String entityName = PersistenceUtils.getEntityName(acmd).toLowerCase();
196         try {
197             TableName tableName = TableName.createTableName(conn.getNamespace(), acmd);
198             TableImpl tableImpl = tables.get(tableName.getForceApiName().toLowerCase());
199             if (tableImpl == null) {
200                 tableImpl = new TableImpl(conn.getNamespace(), tableName, sObjectResults.get(tableName.getForceApiName()), conn);
201                 tables.put(tableName.getForceApiName().toLowerCase(), tableImpl);
202             }
203             entityTables.put(entityName, tableImpl);
204             return tableImpl;
205         } catch (ConnectionException e) {
206             throw new NucleusException("Could not initialize table: " + entityName, e);
207         }
208     }

210     /**
211      * register a virtual table (a table not backed by an actual object in Force.com, like owner).
212      * 
213      * @param acmd the class metadata of the virtual table to register with the cache
214      * @return the created TableImpl
215      */
216     public TableImpl addVirtualTable(AbstractClassMetaData acmd) {
217         String entityName = PersistenceUtils.getEntityName(acmd).toLowerCase();
218         TableName tableName = TableName.createTableName("", acmd);
219         TableImpl tableImpl = tables.get(tableName.getForceApiName().toLowerCase());
220         if (tableImpl == null) {
221             tableImpl = new TableImpl(tableName, acmd);
222             tables.put(tableName.getForceApiName().toLowerCase(), tableImpl);
223         }
224         entityTables.put(entityName, tableImpl);
225         return tableImpl;
226     }
