83      /**
84       * 
85       * Create a manager that will retrieve the specified field values of a particular entity with a known id.
86       * 
87       * @param objectProvider  the object provider
88       * @param storeManager  the store manager
89       * @param mconn  the managed connection for Force.com API connections
90       * @param pkValue  the id of the object we're fetching
91       * @param fieldNumbers  the numbers of the fields we want to fetch
92       * @param query the query object
93       * @throws ConnectionException  thrown if the query to Force.com fails
94       */
95      public ForceFetchFieldManager(ObjectProvider objectProvider, ForceStoreManager storeManager,
96          ForceManagedConnection mconn, Object pkValue,  int[] fieldNumbers, Query query)
97          throws ConnectionException {
98          
99          this.objectProvider = objectProvider;
100         this.ec = objectProvider.getExecutionContext();
101         this.acmd = objectProvider.getClassMetaData();
102         this.storeManager = storeManager;
103         this.mconn = mconn;
104         this.table = storeManager.getTable(objectProvider.getClassMetaData());
105         if (pkValue != null) {
106             QueryResult qr = ((PartnerConnection) mconn.getConnection()).query(
107                     new ForceQueryUtils(objectProvider.getExecutionContext(), mconn, null, null, null, null)
108                     .buildQueryWithPK(table, objectProvider.getClassMetaData(), fieldNumbers, (String) pkValue, 0));
109             if (qr.getSize() == 0) {
110                 throw new NucleusObjectNotFoundException();
111             }
112             this.sobject = qr.getRecords()[0];
113         } else {
114             this.sobject = new SObject();
115         }
116         
117         this.query = query;
118     }

234     /**
235      * 
236      * For fields containing objects rather than strings or primitive types this method will return the proper type of
237      * object depending on the field type. On occasion we use this method to fetch the proper objects for use in expressions
238      * and must override the actual value of the object, in that case use the override value instead of getting it from the
239      * sObject
240      * 
241      * @param ammd  the member metadata for the field/property holding the value
242      * @param valueOverride the value to use in case of an override where you don't want what's in the sObject
243      *                      (used for expressions)
244      * @return  the value cast to it's proper object type
245      */
246     public Object fetchObjectField(AbstractMemberMetaData ammd, Object valueOverride) {
247         if (ammd.getCollection() != null || ammd.getMap() != null) {
248             // we have a collection (@OneToMany relationship) of child objects, recursively populate them
249             final AbstractClassMetaData cmd =
250                 PersistenceUtils.getMemberElementClassMetaData(ammd, ec.getClassLoaderResolver(), ec.getMetaDataManager());
251             Collection childrenColl = null;
252             Map childrenMap = null;
253             int mapKeyPosition = (cmd != null ? cmd.getPKMemberPositions()[0] : -1);
254             if (ammd.getCollection() != null) {
255                 childrenColl = Set.class.isAssignableFrom(ammd.getType()) ?  new LinkedHashSet() : new ArrayList();
256             } else {
257                 childrenMap = new LinkedHashMap();
258                 if (ammd.getKeyMetaData() != null) {
259                     // Read the @MapKey(name="name") data and use that to get mapKeyPosition
260                     mapKeyPosition = cmd.getAbsolutePositionOfMember(ammd.getKeyMetaData().getMappedBy());
261                 }
262             }
263 
264             // Reset column to the related field
265             ColumnImpl column = table.getColumnFor(cmd, ammd);
266             
267             String relationshipName = column.getSelectFieldName();
268             
269             Iterator<XmlObject> rel = sobject.getChildren(relationshipName);
270             while (rel.hasNext()) {
271                 Iterator<XmlObject> subs = rel.next().getChildren("records");
272                 
273                 // This is used for Native query
274                 int[] fieldsToLoad = cmd != null ? cmd.getDFGMemberPositions() : null;
275                 
276                 try {
277                     while (subs.hasNext()) {
278                         final XmlObject child = subs.next();
279                         Object value = ec.findObjectUsingAID(new Type(ec.getClassLoaderResolver().
280                                 classForName(cmd.getFullClassName())),
281                                 ForceQueryUtils.getFieldValues2(cmd, fieldsToLoad, mconn, storeManager, child, query),
282                                 query == null ? true : query.getIgnoreCache(), true);
283                         
284                         if (childrenColl != null) {
285                             childrenColl.add(value);
286                         } else {
287                             childrenMap.put(PersistenceUtils.getMemberValue(cmd, mapKeyPosition, value), value);
288                         }
289                     }
290                 } catch (Exception e) {
291                     throw new NucleusUserException(e.getMessage(), e);
292                 }
293             }
294             if (objectProvider != null) {
295                   Object ret = childrenColl != null ? childrenColl : childrenMap;
296                 return SCOUtils.newSCOInstance(objectProvider, ammd, ammd.getType(),
297                         (ret != null ? ret.getClass() : null), ret, false, false, false);
298             } else {
299                 // This is for Collection MEMBEROF and Map Key, Value, Entry operation only. The collection will be read only.
300                 return childrenColl != null ? Collections.unmodifiableCollection(childrenColl)
301                                                 : Collections.unmodifiableMap(childrenMap);
302             }
303         } else if (ammd.getEmbeddedMetaData() != null) {
304             return fetchEmbeddedObject(ammd);
305         }
306 
307         ColumnImpl column = table.getColumnFor(acmd, ammd);
308         Object o = valueOverride != null ? valueOverride : sobject.getField(column.getSelectFieldName());
309         if (o == null) return null;
310         Calendar cal;
311         Object tvalue;
312         switch (column.getType()) {
313         case _boolean:
314             return Boolean.parseBoolean((String) o);
315         case _int:
316         case percent:
317             if (ammd.getType() == Long.class || ammd.getType() == long.class) {
318                 try {
319                     return Long.parseLong((String) o);
320                 } catch (NumberFormatException ne) {
321                     return Double.valueOf((String) o).longValue();
322                 }
323             } else if (ammd.getType() == Short.class || ammd.getType() == short.class)
324                 return Double.valueOf((String) o).shortValue();
325             else if (ammd.getType() == Float.class || ammd.getType() == float.class)
326                 return Float.parseFloat((String) o);
327             else if (ammd.getType() == Double.class || ammd.getType() == double.class)
328                 return Double.parseDouble((String) o);
329             else if (ammd.getType() == BigInteger.class) {
330                 DecimalFormat f = new DecimalFormat("0.##################E0");
331                 f.setGroupingUsed(false);
332                 return new BigInteger(f.parse((String) o, new ParsePosition(0)).toString());
333             } else {
334                 return Double.valueOf((String) o).intValue();
335             }
336         case _double:
337             if (ammd.getType() == Float.class || ammd.getType() == float.class) {
338                 return Float.parseFloat((String) o);
339             } else {
340                 return Double.parseDouble((String) o);
341             }
342         case currency:
343             return new BigDecimal((String) o, new MathContext(column.getField().getPrecision()))
344                         .setScale(column.getField().getScale(), RoundingMode.HALF_DOWN);
345         case date:
346             tvalue = new DateCodec().deserialize((String) o).getTime();
347             return objectProvider != null ? SCOUtils.newSCOInstance(objectProvider, ammd, ammd.getType(),
348                     (tvalue != null ? tvalue.getClass() : null), tvalue, false, false, false) : tvalue;
349         case datetime:
350             cal = new CalendarCodec().deserialize((String) o);
351             if (ammd.getType() == Date.class) {
352                 tvalue = cal.getTime();
353             } else if (ammd.getType() == Time.class) {
354                 tvalue = new Time(cal);
355             } else {
356                 tvalue = cal;
357             }
358             return objectProvider != null ? SCOUtils.newSCOInstance(objectProvider, ammd, ammd.getType(),
359                     (tvalue != null ? tvalue.getClass() : null), tvalue, false, false, false) : tvalue;
360         case reference:
361             // This is from @ManyToOne relationship, Create an entity and return it
362             if (o instanceof XmlObject) {
363                 try {
364                     AbstractClassMetaData cmd =
365                         ec.getMetaDataManager().getMetaDataForClass(ammd.getTypeName(), ec.getClassLoaderResolver());
366                     // We use the same sobject but with relationship name prefix
367                     return ec.findObjectUsingAID(new Type(ec.getClassLoaderResolver().
368                             classForName(cmd.getFullClassName())),
369                             ForceQueryUtils.getFieldValues2(cmd, cmd.getDFGMemberPositions(),
370                                                                 mconn, storeManager, (XmlObject) o, query),
371                             query == null ? true : query.getIgnoreCache(), true);
372                 } catch (Exception e) {
373                     throw new NucleusUserException(e.getMessage(), e);
374                 }
375             } else {
376                 // We return just a hollow object with ID that will have it's fields fetched later
377                 return ec.findObject(new StringIdentity(ammd.getType(), (String) o), false, false, ammd.getTypeName());
378 
379             }
380         case picklist:
381             if (ammd.getType().isEnum()) {
382                 if (PersistenceUtils.isOrdinalEnum(ammd))
383                     return ammd.getType().getEnumConstants()[Integer.parseInt((String) o)];
384                 
385                 // PicklistValueEnums should have a static fromValue method to load up enum values
386                 else if (PicklistValueEnum.class.isAssignableFrom(ammd.getType())) {
387                     try {
388                         return ammd.getType().getMethod("fromValue", String.class).invoke(ammd.getType(), (String) o);
389                     } catch (Exception e) {
390                         throw new NucleusUserException("Unable to invoke fromValue(String) for enum " + ammd.getType(), e);
391                     }
392                 }
393                 return Enum.valueOf(ammd.getType(), (String) o);
394             }
395             break;
396         case multipicklist:
397             String[] values = MULTI_SELECT_VALUE_SEPARATOR.split((String) o);
398             if (ammd.getType().getComponentType().isEnum()) {
399                 Class type = ammd.getType().getComponentType();
400                 Enum<?>[] enumObjects = (Enum<?>[]) Array.newInstance(type, values.length);
401                 Enum<?>[] enumValues = (Enum<?>[]) ammd.getType().getComponentType().getEnumConstants();
402                 boolean isOrdinal = PersistenceUtils.isOrdinalEnum(ammd);
403                 for (int i = 0; i < values.length; i++) {
404                     enumObjects[i] = isOrdinal ? enumValues[Integer.parseInt(values[i])] : Enum.valueOf(type, values[i]);
405                 }
406                 return enumObjects;
407             } else {
408                 return values;
409             }
410         case url:
411             try {
412                 return new URL((String) o);
413             } catch (MalformedURLException me) {
414                 try {
415                     // Try replacing any leading ':' and '/' with a proper protocol
416                     return new URL(NO_PROTOCOL_URL_PATTERN.matcher((String) o).replaceAll("http://"));
417                 } catch (MalformedURLException me2) {
418                     throw new NucleusUserException(me2.getMessage());
419                 }
420             }
421         case string:
422             if (ammd.getType() == Byte.class || ammd.getType() == byte.class) {
423                 return Byte.parseByte((String) o);
424             } else if (ammd.getType() == Character.class || ammd.getType() == char.class) {
425                 if (o != null) {
426                     return ((String) o).charAt(0);
427                 }
428             }
429             break;
430         case base64:
431             if (ammd.getType() == byte[].class || ammd.getType() == Byte[].class) {
432                 return Base64.decode(((String) o).getBytes());
433             } else {
434                 throw new NucleusUserException("Bad datatype for base64 encoding: " + ammd.getTypeName());
435             }
436         default:
437         }
438         return o;
439     }

441     private Object fetchEmbeddedObject(AbstractMemberMetaData ammd) {
442         if (objectProvider == null) return null;
443         AbstractClassMetaData cmd =
444             storeManager.getMetaDataManager().getMetaDataForClass(ammd.getType(), ec.getClassLoaderResolver());
445         Object obj = null;
446         try {
447             obj = ec.getClassLoaderResolver().classForName(cmd.getFullClassName(), true).newInstance();
448             for (AbstractMemberMetaData eammd : ammd.getEmbeddedMetaData().getMemberMetaData()) {
449                 PersistenceUtils.setFieldValue(ammd.getType(), cmd, cmd.getAbsolutePositionOfMember(eammd.getName()),
450                                                 obj, fetchObjectField(eammd, null));
451             }
452             ec.findObjectProviderForEmbedded(obj, objectProvider, ammd);
453         } catch (Exception e) {
454             throw new NucleusUserException(e.getMessage(), e);
455         }
456         return obj;
457     }
