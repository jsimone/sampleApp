52  /**
53   * 
54   * Field manager for updating records.  This class takes updated fields and stores them on the
55   * SObject which will eventually be pushed to Force.com via a SOQL update (or create or delete)
56   *
57   * @author Fiaz Hossain
58   */
59  public class ForceInsertFieldManager extends AbstractFieldManager {
60  
61      private static final Random PART_B = new Random(System.nanoTime());
62  
63      private final ObjectProvider objectProvider;
64      private final ForceStoreManager storeManager;
65      private final SObject sobject;
66      private final TableImpl table;
67      private final int pkIndexToSkip;
68      private final int versionIndexToSkip;
69      private boolean dirty;
70      
71      /**
72       * Create a manager that will upsert the specified field values of a particular entity.
73       *
74       * @param objectProvider the object provider for this entity
75       * @param storeManager the store manager
76       * @param pkValue the id object (string for Force.com entities) for this entity, null if this is an insert
77       */
78      public ForceInsertFieldManager(ObjectProvider objectProvider, ForceStoreManager storeManager, Object pkValue) {
79          this.objectProvider = objectProvider;
80          this.storeManager = storeManager;
81          this.sobject = new SObject();
82          AbstractClassMetaData acmd = objectProvider.getClassMetaData();
83          this.table = storeManager.getTable(acmd);
84          if (this.table == null) {
85              throw new NucleusUserException("Entity does not exist in Force.com datastore: " + acmd.getEntityName());
86          }
87          sobject.setType(this.table.getTableName().getForceApiName());
88          if (pkValue != null) {
89              sobject.addField(table.getPKFieldName(acmd), pkValue);
90          } else {
91              // If PK value is null then this is an insert and we consider this object to be dirty all the time
92              dirty = true;
93          }
94          // During insert this field index should not be added as it is the "id" field which should be null.
95          // During update it has already been added above so it needs to be skipped too to avoid duplicate fields in statement
96          pkIndexToSkip = acmd.getPKMemberPositions()[0];
97          if (acmd.getVersionMetaData() != null && pkValue != null) {
98              versionIndexToSkip = acmd.getAbsolutePositionOfMember(acmd.getVersionMetaData().getFieldName());
99              Object version = objectProvider.provideField(versionIndexToSkip);
100             if (version == null && storeManager.isEnableOptimisticTransactions()) {
101                 throw new NucleusUserException("Version field must be set to non-null value, field: "
102                                                 + acmd.getVersionMetaData().getFieldName() + " object id:" + sobject.getId());
103             }
104             objectProvider.setVersion(version);
105         } else {
106             versionIndexToSkip = -1;
107         }
108     }

110     /**
111      * Retrieves the Force.com object (SObject) for this {@code ForceInsertFieldManager}.
112      * 
113      * @param appendExternalId flag indicating if an external id should be added to
114      *                         the Force.com object (SObject) is none exists
115      * @return the Force.com object (SObject) for this {@code ForceInsertFieldManager}
116      */
117     public SObject getSObject(boolean appendExternalId) {
118         if (appendExternalId) {
119             if (table.getExternalIdColumn() != null) {
120                 String name = table.getExternalIdColumn().getFieldName();
121                 if (sobject.getField(name) == null) {
122                     sobject.setField(name, "" + System.nanoTime() + PART_B.nextInt());
123                 }
124             }
125         }
126         return sobject;
127     }

129     /**
130      * The entity is considered dirty if fields have been updated (meaning an update is necessary).
131      * Entities with no ids, meaning they are for insert, are always marked as dirty.
132      * @return true if the entity is considered dirty and needs to be upserted
133      */
134     public boolean isDirty() {
135         return dirty;

138     /**
139      * All store methods delegate to this one, to do the proper transformation on the value so it
140      * can be stored in the sObject and upserted properly by the API.
141      * 
142      * @param fieldNumber the index of the field to update
143      * @param value the value that should be stored in the field
144      */
145     private void storeField(int fieldNumber, Object value) {
146         if (value == null || fieldNumber == pkIndexToSkip || fieldNumber == versionIndexToSkip) return;
147         AbstractMemberMetaData ammd =
148             objectProvider.getClassMetaData().getMetaDataForManagedMemberAtAbsolutePosition(fieldNumber);
149         if (ammd.getEmbeddedMetaData() != null) {
150             // we need to add all the embedded fields
151             storeEmbeddedFields(ammd, value);
152             return;
153         }
154         ColumnImpl column = table.getColumnFor(objectProvider.getClassMetaData(), ammd);
155         if (column == null || column.getField() == null || !column.getField().isCreateable()) return;
156         Object actualValue = value;
157         String actualFieldName = column.getFieldName();
158         try {
159             switch (column.getType()) {
160             case datetime:
161                 if (value instanceof Date) {
162                     ColumnMetaData[] cmds = ammd.getColumnMetaData();
163                     Calendar cal = Calendar.getInstance();
164                     cal.setTime((Date) value);
165                     if (cmds != null && cmds.length > 0 && "TIME".equals(cmds[0].getJdbcType())) {
166                         actualValue = new Time(cal);
167                     } else {
168                         actualValue = cal;
169                     }
170                 } else if (value instanceof Time) {
171                     actualValue = value;
172                 }
173                 break;
174             case currency:
175                 actualValue = value.toString();
176                 break;
177             case reference:
178                 AbstractClassMetaData acmd =
179                     storeManager.getMetaDataManager().getMetaDataForClass(ammd.getType(),
180                                                         objectProvider.getExecutionContext().getClassLoaderResolver());
181                 if (acmd != null) {
182                     actualValue = PersistenceUtils.getMemberValue(acmd, acmd.getPKMemberPositions()[0], value);
183                     if (actualValue == null) {
184                         ObjectManager om = ((ObjectProviderImpl) objectProvider).getStateManager().getObjectManager();
185                         if (((ForceObjectManagerImpl) om).isInAllOrNothingMode()) {
186                             /**
187                              * This is instance of AllOrNothing transaction
188                              * Since the parent object has not been saved to the db yet
189                              * we do not have an id yet so we link objects by extId
190                              */
191                             SObject parentRef = new SObject();
192                             TableImpl parent = storeManager.getTable(acmd);
193                             if (parent.getExternalIdColumn() == null) {
194                                 throw new NucleusUserException("EntityManager in persistence.xml has 'force.AllOrNothing'"
195                                                                + " set to true. In this mode all top parent Entities must"
196                                                                + " have an externalId field. Offending entity: "
197                                                                + parent.getTableName().getName());
198                             }
199                             parentRef.setType(parent.getTableName().getForceApiName());
200                             SObject parentSObject = ((ForceObjectManagerImpl) om).getParentSObject(value);
201                             parentRef.setField(parent.getExternalIdColumn().getFieldName(),
202                                                 parentSObject.getField(parent.getExternalIdColumn().getFieldName()));
203                             actualValue = parentRef;
204                             actualFieldName = column.getForceApiRelationshipName();
205                         } else {
206                             throw new NucleusUserException("Child entity cannot be saved before parent entity.");
207                         }
208                     }
209                 } else {
210                     if (ammd.getCollection() != null || ammd.getMap() != null) {
211                         /**
212                          * There is no data to set on the parent as the relationship is only maintained from the child's end
213                          */
214                         return;
215                     }
216                 }
217                 break;
218             case multipicklist:
219                 if (value.getClass().isArray()) {
220                     if (ammd.getType().getComponentType().isEnum()) {
221                         Enum<?>[] enums = (Enum<?>[]) value;
222                         boolean isOrdinal = PersistenceUtils.isOrdinalEnum(ammd);
223                         boolean isPicklistValue = PicklistValueEnum[].class.isAssignableFrom(enums.getClass());
224                         
225                         StringBuilder sb = new StringBuilder(enums.length * 20);
226                         for (Enum<?> e : enums) {
227                             if (sb.length() > 0) sb.append(";");
228                             if (isOrdinal) {
229                                 sb.append(e.ordinal());
230                             } else if (isPicklistValue) {
231                                 sb.append(((PicklistValueEnum) e).value());
232                             } else {
233                                 sb.append(e.name());
234                             }
235                         }
236                         actualValue = sb.toString();
237                     } else {
238                         // else treat it like a non-strict picklist value
239                         String[] values = (String[]) value;
240                         StringBuilder sb = new StringBuilder(values.length * 20);
241                         for (String v : values) {
242                             if (sb.length() > 0) sb.append(";");
243                             sb.append(v);
244                         }
245                         actualValue = sb.toString();
246                     }
247                     break;
248                 }
249                 // else treat it like a single picklist value
250             case picklist:
251                 if (value.getClass().isEnum()) {
252                     Enum e = (Enum) value;
253                     if (PersistenceUtils.isOrdinalEnum(ammd)) {
254                         actualValue = "" + e.ordinal();
255                     } else if (e instanceof PicklistValueEnum) {
256                         actualValue = ((PicklistValueEnum) e).value();
257                     } else {
258                         actualValue = e.name();
259                     }
260                 }
261                 break;
262             case url:
263                 actualValue = value.toString();
264                 break;
265             case base64:
266                 if (ammd.getType() == byte[].class || ammd.getType() == Byte[].class) {
267                     /**
268                      * Odd that I did not have to encode this, yet when reading I have to decode it.
269                      */
270                     //actualValue = Base64.encode((byte[]) value);
271                     actualValue = value;
272                 } else {
273                     throw new NucleusUserException("Bad datatype for base64 encoding: " + ammd.getTypeName());
274                 }
275                 break;
276             default:
277             }
278         } catch (InvocationTargetException x) {
279             throw new NucleusException(x.getMessage(), x);
280         } catch (IllegalAccessException x) {
281             throw new NucleusException(x.getMessage(), x);
282         }
283         sobject.addField(actualFieldName, actualValue);
284         dirty = true;
285     }

287     private void storeEmbeddedFields(AbstractMemberMetaData ammd, Object value) {
288         AbstractClassMetaData cmd = objectProvider.getClassMetaData();
289         for (AbstractMemberMetaData eammd : ammd.getEmbeddedMetaData().getMemberMetaData()) {
290             if (eammd.getPersistenceModifier() != FieldPersistenceModifier.PERSISTENT) continue;
291             try {
292                 Object newValue = PersistenceUtils.getMemberValue(eammd.getMemberRepresented(), value);
293                 if (eammd.getEmbeddedMetaData() != null) {
294                     storeEmbeddedFields(eammd, newValue);
295                 } else {
296                         ColumnImpl column = table.getColumnFor(cmd, eammd);
297                         if (column != null) {
298                             sobject.addField(column.getFieldName(), newValue);
299                             dirty = true;
300                         }
301                 }
302             } catch (Exception e) {
303                 throw new NucleusUserException(e.getMessage());
304             }
305         }
306     }

343     @Override
344     public void storeObjectField(int fieldNumber, Object value) {
345         if (value instanceof Long) {
346             storeField(fieldNumber, value.toString());
347         } else if (value instanceof Float) {
348             storeField(fieldNumber, ((Float) value).doubleValue());
349         } else if (value instanceof Short) {
350             storeField(fieldNumber, ((Short) value).intValue());
351         } else if (value instanceof Byte) {
352             storeField(fieldNumber, ((Byte) value).toString());
353         } else if (value instanceof BigInteger) {
354             storeField(fieldNumber, value.toString());
355         } else if (value instanceof Character) {
356             storeField(fieldNumber, ((Character) value).toString());
357         } else {
358             storeField(fieldNumber, value);
359         }
360     }
