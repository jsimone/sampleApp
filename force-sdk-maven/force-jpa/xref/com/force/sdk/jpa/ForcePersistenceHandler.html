74      /**
75       * Deletes a persistent object from the datastore.
76       *
77       * @param op The ObjectProvider of the object to be deleted.
78       */
79      @Override
80      public void deleteObject(ObjectProvider op) {
81          // Check if read-only so update not permitted
82          storeManager.assertReadOnlyForUpdateOfObject(op);
83  
84          ForceManagedConnection mconn = (ForceManagedConnection) storeManager.getConnection(op.getExecutionContext());
85          ObjectManager om = ((ObjectProviderImpl) op).getStateManager().getObjectManager();
86          boolean isAllOrNothingMode = om instanceof ForceObjectManagerImpl && ((ForceObjectManagerImpl) om).isInAllOrNothingMode();
87          try {
88              Object pkValue = op.provideField(op.getClassMetaData().getPKMemberPositions()[0]);
89              if (!isAllOrNothingMode) {
90                  if (LOGGER.isDebugEnabled()) {
91                      LOGGER.debug("Deleting object: " + pkValue);
92                  }
93                  DeleteResult[] results = ((PartnerConnection) mconn.getConnection()).delete(new String[]{(String) pkValue});
94                  checkForErrors(results);
95              } else {
96                  if (LOGGER.isDebugEnabled()) {
97                      LOGGER.debug("Queuing for A-O-N delete object: " + pkValue);
98                  }
99                  ((ForceObjectManagerImpl) om).addToDeleteList((String) pkValue);
100             }
101         } catch (ApiFault af) {
102             throw ForceApiExceptionMap.mapToNucleusException(af, false /* isQuery */,
103                     storeManager.isEnableOptimisticTransactions());
104         } catch (NucleusOptimisticException noe) {
105             throw noe;
106         } catch (ConnectionException x) {
107             throw new NucleusDataStoreException(x.getMessage(), x);
108         } finally {
109             mconn.release();
110         }
111     }

113     /**
114      * Fetches a persistent object from the database.
115      *
116      * @param op           The ObjectProvider of the object to be fetched.
117      * @param fieldNumbers The numbers of the fields to be fetched.
118      */
119     @Override
120     public void fetchObject(ObjectProvider op, int[] fieldNumbers) {
121         ForceManagedConnection mconn = (ForceManagedConnection) storeManager.getConnection(op.getExecutionContext());
122         try {
123             int pkPosition = op.getClassMetaData().getPKMemberPositions()[0];
124             /**
125              * Check is we are being asked for PK only. If so return the ID that has been passed in
126              */
127             ForceFetchFieldManager fm;
128             if (fieldNumbers.length == 1 && fieldNumbers[0] == pkPosition) {
129                 XmlObject sObject = new XmlObject();
130                 sObject.addField("Id", op.provideField(pkPosition));
131                 fm = new ForceFetchFieldManager(op, storeManager, mconn, sObject, null);
132             } else {
133                 fm = new ForceFetchFieldManager(op, storeManager, mconn,
134                     op.provideField(pkPosition), fieldNumbers, null);
135             }
136             op.replaceFields(fieldNumbers, fm);
137         } catch (ApiFault af) {
138             throw ForceApiExceptionMap.mapToNucleusException(af, false /* isQuery */,
139                     storeManager.isEnableOptimisticTransactions());
140         } catch (NucleusObjectNotFoundException onf) {
141             throw onf;
142         } catch (Exception x) {
143             throw new NucleusDataStoreException(x.getMessage(), x);
144         } finally {
145             mconn.release();
146         }
147     }

171     @Override
172     public void locateObject(ObjectProvider op) {
173         ForceManagedConnection mconn = (ForceManagedConnection) storeManager.getConnection(op.getExecutionContext());
174         try {
175             AbstractClassMetaData acmd = op.getClassMetaData();
176             TableImpl table = storeManager.getTable(acmd);
177             QueryResult qr = ((PartnerConnection) mconn.getConnection())
178                 .query("select count() from " + table.getTableName().getForceApiName() + " where id='"
179                         + op.provideField(op.getClassMetaData().getPKMemberPositions()[0]) + "'");
180             if (qr.getSize() == 0) {
181                 throw new NucleusObjectNotFoundException();
182             }
183         } catch (ConnectionException x) {
184             throw new NucleusDataStoreException(x.getMessage(), x);
185         } finally {
186             mconn.release();
187         }

201     /**
202      * This method is used for AllOrNothing operations.
203      *
204      * @param objects the objects to be created
205      * @param objectProviders the object providers for each object
206      * @param ec the execution context for this transaction
207      */
208     public void createObjects(Collection<SObject> objects, Collection<ObjectProvider> objectProviders, ExecutionContext ec) {
209         ForceManagedConnection mconn = (ForceManagedConnection) storeManager.getConnection(ec);
210         try {
211             SObject[] toSave = objects.toArray(new SObject[objects.size()]);
212             if (LOGGER.isDebugEnabled()) {
213                 LOGGER.debug("Creating objects: " + toString(toSave, false, null));
214             }
215             PartnerConnection connection = (PartnerConnection) mconn.getConnection();
216             connection.setAllOrNoneHeader(true);
217             try {
218                 //connection.getConfig().setTraceMessage(true);
219                 SaveResult[] results = connection.create(toSave);
220                 checkForErrors(results);
221                 int i = 0;
222                 for (ObjectProvider op : objectProviders) {
223                     op.setPostStoreNewObjectId(results[i++].getId());
224                 }
225                 if (LOGGER.isDebugEnabled()) {
226                     LOGGER.debug("Created objects: " + toString(results, false, null));
227                 }
228             } finally {
229                 connection.setAllOrNoneHeader(false);
230             }
231         } catch (NucleusOptimisticException noe) {
232             throw noe;
233         } catch (NucleusUserException nue) {
234             throw nue;
235         } catch (Exception x) {
236             throw new NucleusDataStoreException(x.getMessage(), x);
237         } finally {
238             mconn.release();
239         }
240     }

315     private void upsert(ObjectProvider op, int[] fieldNumbers) {
316         if (op.getClassMetaData().isEmbeddedOnly()) {
317             // Embedded entities will be saved by the parent
318             return;
319         }
320         // Check if read-only so update not permitted
321         storeManager.assertReadOnlyForUpdateOfObject(op);
322 
323         if (!storeManager.managesClass(op.getClassMetaData().getFullClassName())) {
324             storeManager.addClass(op.getClassMetaData().getFullClassName(), op.getExecutionContext().getClassLoaderResolver());
325         }
326 
327         ForceManagedConnection mconn = (ForceManagedConnection) storeManager.getConnection(op.getExecutionContext());
328         try {
329             ForceInsertFieldManager fm = new ForceInsertFieldManager(op, storeManager,
330                     fieldNumbers != null ? op.provideField(op.getClassMetaData().getPKMemberPositions()[0]) : null);
331             op.provideFields(fieldNumbers != null ? fieldNumbers : op.getClassMetaData().getAllMemberPositions(), fm);
332             /**
333              * It is possible that this field is just a parent whose children have been updated only. In that case
334              * the current object will not be dirty and we have nothing else to do.
335              */
336             if (!fm.isDirty()) return;
337             ObjectManager om = ((ObjectProviderImpl) op).getStateManager().getObjectManager();
338             boolean isAllOrNothingMode =
339                 om instanceof ForceObjectManagerImpl && ((ForceObjectManagerImpl) om).isInAllOrNothingMode();
340             SObject toSave;
341             if (!isAllOrNothingMode) {
342                 PartnerConnection connection = getPartnerConnection(mconn, op);
343                 try {
344                     toSave = fm.getSObject(false);
345                     if (LOGGER.isDebugEnabled()) {
346                         if (fieldNumbers != null) {
347                             LOGGER.debug("Updating object: " + toSave.getType() + " id: " + toSave.getId());
348                         } else {
349                             LOGGER.debug("Creating object: " + toSave.getType());
350                         }
351                     }
352                     SaveResult[] results = fieldNumbers != null ? connection.update(new SObject[]{toSave})
353                                                 : ((PartnerConnection) mconn.getConnection()).create(new SObject[]{toSave});
354                     checkForErrors(results);
355                     if (fieldNumbers == null) {
356                         op.setPostStoreNewObjectId(results[0].getId());
357                         if (LOGGER.isDebugEnabled()) {
358                             LOGGER.debug("Created object id: " + results[0].getId());
359                         }
360                     }
361                 } finally {
362                     connection.clearConditionalRequestHeader();
363                 }
364             } else {
365                 if (fieldNumbers != null) {
366                     //When we do a-o-n + optimistic, per jpa spec we need to save properly even if some objs are missong @Version,
367                     //so if op.getVersion is null, we give a Calendar set to System.currentTimeMilis + 1 HOUR so
368                     // the if-modified-before check for the object without @Version will always succeed
369                     toSave = fm.getSObject(false);
370                     ((ForceObjectManagerImpl) om).addToUpdateList(toSave,
371                             op.getVersion() != null ? (Calendar) op.getVersion() : getVersionForUnversioned());
372                 } else {
373                     toSave = fm.getSObject(true);
374                     ((ForceObjectManagerImpl) om).addToCreateList(toSave, op);
375                 }
376                 if (LOGGER.isDebugEnabled()) {
377                     if (fieldNumbers != null) {
378                         LOGGER.debug("Queuing for A-O-N update object: " + toSave.getType() + " id: " + toSave.getId());
379                     } else {
380                         LOGGER.debug("Queuing for A-O-N create object: " + toSave.getType());
381                     }
382                 }
383             }
384         } catch (ApiFault af) {
385             throw ForceApiExceptionMap.mapToNucleusException(af, false /* isQuery */,
386                 storeManager.isEnableOptimisticTransactions());
387         } catch (NucleusException ne) {
388             throw ne;
389         } catch (Exception x) {
390             throw new NucleusDataStoreException(x.getMessage(), x);
391         } finally {
392             mconn.release();
393         }
394     }

396     private static Calendar getVersionForUnversioned() {
397         long time = System.currentTimeMillis() + (60 * 60 * 1000);
398         Calendar cal = Calendar.getInstance();
399         cal.setTimeInMillis(time);
400         return cal;
401     }

403     private PartnerConnection getPartnerConnection(ForceManagedConnection mconn, ObjectProvider op) {
404         PartnerConnection connection = (PartnerConnection) mconn.getConnection();
405         if (op.getVersion() != null && storeManager.isEnableOptimisticTransactions()) {
406             ConditionalRequestHeader_element ch = new ConditionalRequestHeader_element();
407             ch.setIfModifiedBefore((Calendar) op.getVersion());
408             connection.__setConditionalRequestHeader(ch);
409             if (LOGGER.isDebugEnabled()) {
410                 LOGGER.debug("Conditional header set to: " + new CalendarCodec().getValueAsString(ch.getIfModifiedBefore()));
411             }
412         }
413         return connection;
414     }

416     private PartnerConnection getPartnerConnection(ForceManagedConnection mconn, Calendar[] versions) {
417         PartnerConnection connection = (PartnerConnection) mconn.getConnection();
418         if (versions.length > 0 && storeManager.isEnableOptimisticTransactions()) {
419             ConditionalRequestHeader_element ch = new ConditionalRequestHeader_element();
420             ch.setIfModifiedBeforeArray(versions);
421             connection.__setConditionalRequestHeader(ch);
422             if (LOGGER.isDebugEnabled()) {
423                 LOGGER.debug("Conditional header set to: " + toString(ch.getIfModifiedBeforeArray(), false, new CalendarCodec()));
424             }
425         }
426         return connection;
427     }

429     private void checkForErrors(SaveResult[] results) {
430         List<Error> failures = null;
431         boolean optimisticFailure = false;
432         for (SaveResult sr : results) {
433             if (!sr.getSuccess()) {
434                 if (failures == null) {
435                     failures = new ArrayList<Error>();
436                 }
437                 optimisticFailure = handleError(failures, sr.getErrors()[0]);
438             }
439         }
440         handleFailures(failures, optimisticFailure);
441     }

443     private static boolean handleError(List<Error> failures, Error error) {
444         failures.add(error);
445         return error.getStatusCode() == StatusCode.ENTITY_FAILED_IFLASTMODIFIED_ON_UPDATE;
446     }

448     private static void handleFailures(List<Error> failures, boolean optimisticFailure) {
449         if (failures != null) {
450             Iterator<Error> iter = failures.iterator();
451             while (iter.hasNext()) {
452                 Error error = iter.next();
453                 if (error.getStatusCode() == StatusCode.ALL_OR_NONE_OPERATION_ROLLED_BACK && failures.size() > 1) {
454                     iter.remove();
455                 }
456             }
457             if (optimisticFailure) {
458                 throw new NucleusOptimisticException(failures.get(0).getMessage(),
459                         failures.size() == 1 ? failures.get(0) : failures.toArray(new Error[failures.size()]));
460             } else {
461                 throw new NucleusUserException(failures.get(0).getMessage(),
462                         failures.size() == 1 ? failures.get(0) : failures.toArray(new Error[failures.size()]));
463             }
464         }
465     }

467     /**
468      * checks for errors in a delete call and handles the results properly.
469      * 
470      * @param results the results from the delete API call
471      */
472     public static void checkForErrors(DeleteResult[] results) {
473         List<Error> failures = null;
474         boolean optimisticFailure = false;
475         for (DeleteResult dr : results) {
476             if (!dr.getSuccess()) {
477                 if (failures == null) {
478                     failures = new ArrayList<Error>();
479                 }
480                 optimisticFailure = handleError(failures, dr.getErrors()[0]);
481             }
482         }
483         handleFailures(failures, optimisticFailure);

505     private static String toString(Object[] objects, boolean isUpdate, CalendarCodec cCodec) {
506         StringBuilder sb = new StringBuilder(objects.length * 40);
507         sb.append("[");
508         for (Object obj : objects) {
509             if (sb.length() > 1) sb.append(", ");
510             if (obj instanceof SObject) {
511                 SObject s = (SObject) obj;
512                 sb.append("entity: ").append(s.getType());
513                 if (isUpdate) {
514                     sb.append(" id: ").append(s.getId());
515                 }
516             } else if (obj instanceof SaveResult) {
517                 sb.append(((SaveResult) obj).getId());
518             } else if (obj instanceof Calendar) {
519                 sb.append(cCodec.getValueAsString(obj));
520             }
521         }
522         sb.append("]");
523         return sb.toString();
524     }
