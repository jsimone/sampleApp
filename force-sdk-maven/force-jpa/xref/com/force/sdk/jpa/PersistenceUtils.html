67      /**
68       * 
69       * Return the name of the entity, first checking the name attribute of the annotations and then
70       * defaulting to the entity name on the metadata object.
71       * 
72       * @param acmd  the metadata object of the JPA entity
73       * @return  The name of the entity to use in JPA (entity name or JPQL queries)

75      public static String getEntityName(AbstractClassMetaData acmd) {
76          Map<String, String> extensions = getForceExtensions(acmd);
77          String name = extensions.get("name") != null ? extensions.get("name") : acmd.getEntityName();
78          return name != null ? name : acmd.getName();

82       * 
83       * An entity does not have schema if it would not directly represent an object in Force.com,
84       * e.g. if the entity is embedded or if it's a subclass of another entity
85       * 
86       * @param acmd  the metadata object of the JPA entity
87       * @return  true if the entity does not have schema

89      public static boolean hasNoSchema(AbstractClassMetaData acmd) {
90          return acmd.isEmbeddedOnly() || acmd.getInheritanceMetaData() != null
91                  && acmd.getInheritanceMetaData().getStrategy() == InheritanceStrategy.SUBCLASS_TABLE;

94      /**
95       * 
96       * Performs various checks to see if the schema is read only (i.e. schema should not be
97       * created, updated, or deleted when the application starts up, regardless of whether schema
98       * creation is enabled).
99       * 
100      * @param acmd  the metadata object of the JPA entity
101      * @param checkHierarchy  generally, we don't need to check the hierarchy if we're creating fields, but
102      *                        if we're creating tables we need to check the parent to see if the parent object
103      *                        is read only
104      * @return  true if the schema is read only.
105      */
106     public static boolean isReadOnlySchema(AbstractClassMetaData acmd, boolean checkHierarchy) {
107         boolean isReadOnlySchema = false;
108         do {
109             isReadOnlySchema = isReadOnlySchemaInternal(acmd);
110             acmd = acmd.getSuperAbstractClassMetaData();
111         } while (!isReadOnlySchema && checkHierarchy && acmd != null);
112         return isReadOnlySchema;
113     }

115     private static boolean isReadOnlySchemaInternal(AbstractClassMetaData acmd) {
116         Map<String, String> extensions = getForceExtensions(acmd);
117         String value = extensions.get("readOnlySchema");
118         if (value == null) value = extensions.get("virtualSchema");  // Virtual schema is always read only
119         
120         return value != null && Boolean.valueOf(value);
121     }
122     
123     private static boolean isReadOnlyFieldSchema(AbstractMemberMetaData ammd, OMFContext omf) {
124         // Load up the class metadata for the owning class of this member (e.g. if an inherited field,
125         // the super class which has declared it)
126         AbstractClassMetaData owningAcmd =
127             omf.getMetaDataManager().getMetaDataForClass(ammd.getClassName(), omf.getClassLoaderResolver(null));
128         return isReadOnlySchemaInternal(owningAcmd);
129     }

158     /**
159      * 
160      * Utility method that returns only the Force.com specific extensions for a piece of
161      * metadata (either an entity or a field).
162      * 
163      * @param md  The metadata containing extensions
164      * @return    a map of the extension key and value of the extension, limited to Force.com extensions
165      */
166     public static Map<String, String> getForceExtensions(MetaData md) {
167         Map<String, String> map = new HashMap<String, String>(4);
168         ExtensionMetaData[] extensions = md.getExtensions();
169         if (extensions != null && extensions.length > 0) {
170             for (ExtensionMetaData e : extensions) {
171                 if (ForceStoreManager.FORCE_KEY.equals(e.getVendorName())) {
172                     map.put(e.getKey(), e.getValue());
173                 }
174             }
175         }
176         return map;
177     }

191     /**
192      * 
193      * A check to see if a field has an ordinal enum type, i.e. it's annotated with
194      * \@Enumerated(EnumType.ORDINAL).  Callers of this method should have already verified
195      * ammd to be a picklist field.
196      * 
197      * @param ammd  the metadata object of a picklist field
198      * @return      true if the field is an ordinal enum
199      */
200     public static boolean isOrdinalEnum(AbstractMemberMetaData ammd) {
201         ColumnMetaData[] cmds = ammd.getColumnMetaData();
202         if (cmds != null && cmds.length > 0 && "INTEGER".equals(cmds[0].getJdbcType())) return true;
203         // If the fast method fails take the slow route
204         Enumerated enumerated = ((AccessibleObject) ammd.getMemberRepresented()).getAnnotation(Enumerated.class);
205         return enumerated != null && (enumerated.value() == null || enumerated.value() == EnumType.ORDINAL);
206     }

208     /**
209      * 
210      * A check for whether a field is a multiselect picklist or a single select picklist.
211      * Callers of this method ahould have already verified ammd to be a picklist field
212      * 
213      * @param ammd  the metadata object of a picklist field
214      * @return      true if the field represents a multi select picklist, false otherwise.
215      */
216     public static boolean isMultiPicklist(AbstractMemberMetaData ammd) {
217         // Somehow when we create an Enum[] and annotate with @Enumerated
218         // it is not picked up by Datanucleus and included into ColumnMetadata
219         if (ammd.getMemberRepresented() == null) return false;
220         return ((AccessibleObject) ammd.getMemberRepresented()).isAnnotationPresent(Enumerated.class)
221                 && (ammd.getType().isArray() && ammd.getType().getComponentType().isEnum() || ammd.getType() == String[].class);
222     }

255     /**
256      * 
257      * Retrive the value of a field on an entity via reflection.
258      * 
259      * @param member  the Member object for a particular field on an entity
260      * @param entity  the entity that contains the value
261      * @return  the value stored in the entity's member object
262      * @throws IllegalAccessException thrown if the member cannot be accessed
263      * @throws InvocationTargetException thrown if the member's get method cannot be invoked
264      */
265     public static Object getMemberValue(Member member, Object entity) throws IllegalAccessException, InvocationTargetException {
266         if (member instanceof Method) {
267             Method m = (Method) member;
268             if (!m.isAccessible()) {
269                 m.setAccessible(true);
270             }
271             return m.invoke(entity, (Object[]) null);
272         } else if (member instanceof Field) {
273             Field f = (Field) member;
274             if (!f.isAccessible()) {
275                 f.setAccessible(true);
276             }
277             return f.get(entity);
278         }
279         return null;
280     }

283      * 
284      * Returns the annotation of type T on a field or method, e.g. you could use this method to get
285      * the @Column or @JoinFilter annotation on a getter method for a column
286      * 
287      * @param <T>  an Annotation type
288      * @param member the member representing a particular field or method
289      * @param annotationClass  the class of the annotation object that should be returned
290      * @return  the Annotation on this field or method
291      */
292     public static <T extends Annotation> T getMemberAnnotation(Member member, Class<T> annotationClass) {
293         if (member instanceof Method) {
294             Method m = (Method) member;
295             return m.getAnnotation(annotationClass);
296         } else if (member instanceof Field) {
297             Field f = (Field) member;
298             return f.getAnnotation(annotationClass);
299         }
300         return null;
301     }

304      * 
305      * Sets the value of a field on an entity.
306      * 
307      * @param clazz  The entity class
308      * @param acmd  metadata object for the entity
309      * @param position  the position of the field that should be set
310      * @param entity  the instance of the entity being updated
311      * @param value  the value that should be set on the field
312      * @throws SecurityException  thrown if the field cannot be accessed
313      * @throws NoSuchFieldException  thrown if the field does not exist
314      * @throws IllegalArgumentException  thrown if the set fails
315      * @throws IllegalAccessException  thrown if the set fails
316      */
317     public static void setFieldValue(Class clazz, AbstractClassMetaData acmd, int position, Object entity, Object value)
318         throws SecurityException, NoSuchFieldException, IllegalArgumentException, IllegalAccessException {
319         Field field = clazz.getDeclaredField(acmd.getMetaDataForManagedMemberAtAbsolutePosition(position).getName());
320         if (!field.isAccessible()) {
321             field.setAccessible(true);
322         }
323         field.set(entity, value);
324     }

326     private static String getForceApiRelationshipName(AbstractMemberMetaData ammd, Map<String, String> extensions,
327             OMFContext omf) {
328         AbstractClassMetaData acmd =
329             PersistenceUtils.getMemberElementClassMetaData(ammd, omf.getClassLoaderResolver(null), omf.getMetaDataManager());
330         AbstractMemberMetaData mmd = acmd.getMetaDataForMember(ammd.getMappedBy());
331         if (mmd == null) {
332             // The mappedBy field does not exist
333             throw new NucleusUserException("Cannot access mappedBy field " + ammd.getMappedBy()
334                                             + " on entity " + acmd.getName() + " at " + ammd.getFullFieldName());
335         }
336         String tableName = acmd.getEntityName();
337         String fieldNameNoSuffix = getFieldName(mmd, extensions);
338         extensions = getForceExtensions(mmd);
339         String childRelationshipName = extensions.get("childRelationshipName");
340         String relationshipName =
341             childRelationshipName != null ? childRelationshipName : String.format("%s_%ss", fieldNameNoSuffix, tableName);
342         if (relationshipName.length() > 40) {
343             relationshipName = relationshipName.substring(0, 40);
344         }
345         return relationshipName;
346     }

348     /**
349      * 
350      * Utility method to determine the name of a field based on the member metadata. Field names can be set
351      * via JPA or Force.com annotations, and relationship fields call for special handling
352      * 
353      * @param ammd  the metadata object for the field we're retrieving a name for
354      * @param omf   the Object Manager Factory context
355      * @return the name of a field/column as it would appear in the Force.com API
356      */
357     public static String getForceApiName(AbstractMemberMetaData ammd, OMFContext omf) {
358         Map<String, String> extensions = getForceExtensions(ammd);
359         
360         if (ammd.getCollection() != null || ammd.getMap() != null) {
361             String relationshipName = getForceApiRelationshipName(ammd, extensions, omf);
362             
363             // If the relationship already exists
364             if (isReadOnlyFieldSchema(ammd, omf)) {
365                 return relationshipName;
366             }
367             
368             // Append the custom suffix if it's missing
369             if (!relationshipName.endsWith(ColumnImpl.CUSTOM_RELATIONSHIP_SUFFIX)) {
370                 return relationshipName + ColumnImpl.CUSTOM_RELATIONSHIP_SUFFIX;
371             }
372             
373             // Not a standard relationship and already ends with __r
374             return relationshipName;
375         } else {
376             // Get the java field name
377             String fieldName = getFieldName(ammd, extensions);
378             
379             // Standard fields names are the same as the java field name
380             // Also, if the entity is read only, the schema already exists in Force.com
381             // so we should just use whatever java field name is loaded
382             // (e.g. standard field name, @Column annotation)
383             if (ForceColumnMetaData.STANDARD_FIELDS.contains(fieldName.toLowerCase())
384                     || isReadOnlyFieldSchema(ammd, omf)) {
385                 return fieldName;
386             }
387             
388             // Append the custom suffix if it's missing
389             if (!fieldName.endsWith(ColumnImpl.CUSTOM_THING_SUFFIX)) {
390                 return fieldName + ColumnImpl.CUSTOM_THING_SUFFIX;
391             }
392             
393             // Not a standard field and already ends with __c
394             return fieldName;
395         }
396     }

430     /**
431      * 
432      * Get the name of a field for use with JPA. First try to get the name from the Force.com annotations, the
433      * try to get the name from standard JPA annotations, otherwise use the name of the field in Java
434      * 
435      * @param ammd  the metadata object for the field we're retrieving a name for
436      * @param extensions  the Force.com extensions for this metadata (see getForceExtensions}
437      * @return  the name of the field for JPQL
438      */
439     public static String getFieldName(AbstractMemberMetaData ammd, Map<String, String> extensions) {
440         // First, try to get the field name from the @CustomObject annotation
441         String fieldName = extensions.get("name");
442         
443         if (fieldName != null) {
444             return fieldName;
445         }
446         
447         // Next, try to get the field name from JPA annotations
448         fieldName = getFieldNameFromJPAAnnotation(ammd);
449         if (fieldName == null) {
450             // Fallback to the field/property name
451             fieldName = ammd.getName();
452         }
453         if (fieldName.indexOf(":") > -1) {
454             fieldName = fieldName.substring(fieldName.indexOf(":") + 1);
455         }
456         
457         return fieldName;
458     }

460     /**
461      * 
462      * Returns the name of a column as specified in a JPA annotation (can possibly be null).
463      * 
464      * @param ammd  the metadata object for the field we're retrieving a name for
465      * @return  null if the field name is not defined by a JPA annotation, otherwise return the field name
466      */
467     public static String getFieldNameFromJPAAnnotation(AbstractMemberMetaData ammd) {
468         String fieldName = null;
469         
470         // Try the first column if specified
471         ColumnMetaData[] colmds = ammd.getColumnMetaData();
472         ElementMetaData elemmd;
473         if (colmds != null && colmds.length > 0) {
474             fieldName = colmds[0].getName();
475             
476         // In some cases (e.g. Collections and Arrays)
477         // the ColumnMetaData will be embedded in ElementMetaData
478         } else if ((elemmd = ammd.getElementMetaData()) != null) {
479             colmds = elemmd.getColumnMetaData();
480             if (colmds != null && colmds.length > 0) {
481                 fieldName = colmds[0].getName();
482             }
483         }
484         
485         return fieldName;
486     }
