53   * 
54   * Helper class for building the WHERE clause in SOQL queries.
55   *
56   * @author Fiaz Hossain
57   */
58  public class ExpressionBuilderHelper {
59      ForceQueryUtils forceQuery;
60      StringBuilder sb;
61      TableImpl table;
62      AbstractClassMetaData acmd;
63      int level;
64      boolean isJoin;
65      QueryCompilation compilation;
66      FetchPlan fetchPlan;
67      int fetchDepth;
68      private final int maxFetchDepth;
69      Map<String, Expression> aliasToFilterMappings; // alias => where expression
70      Map<TupleName, String> relatedJoinAliases; // fieldname => alias
71      boolean isInSelect;
72      private final boolean isTopLevel;
73      
74      ExpressionBuilderHelper(ForceQueryUtils forceQuery, int length, TableImpl table,
75              AbstractClassMetaData acmd, boolean isJoin, QueryCompilation compilation, FetchPlan fetchPlan,
76              int fetchDepth, ExpressionBuilderHelper parent) {
77          this.forceQuery = forceQuery;
78          this.sb = new StringBuilder(length);
79          this.table = table;
80          this.acmd = acmd;
81          this.level = 0;
82          this.isJoin = isJoin;
83          this.compilation = compilation;
84          this.fetchPlan = fetchPlan;
85          this.fetchDepth = fetchDepth;
86          Object mfd = forceQuery.getHints(QueryHints.MAX_FETCH_DEPTH);
87          /**
88           * For maxFetchDepth use the following priority -
89           * First preference given to QueryHints.
90           * Second preference to passed in fetchPlan.
91           * Finally use the default configured "datanucleus.maxFetchDepth" property
92           */
93          int maxDepth = mfd != null ? (Integer) mfd : fetchPlan != null ? fetchPlan.getMaxFetchDepth()
94              : forceQuery.getExecutionContext().getOMFContext()
95                                                  .getPersistenceConfiguration().getIntProperty("datanucleus.maxFetchDepth");
96          this.maxFetchDepth = maxDepth >= 0 ? maxDepth : 1;
97          if (parent != null) {
98              this.aliasToFilterMappings = parent.aliasToFilterMappings;
99              this.relatedJoinAliases = parent.relatedJoinAliases;
100         } else {
101             initRelatedAliases(compilation);
102         }
103         this.isInSelect = true;
104         this.isTopLevel = parent == null;
105     }

107     /**
108      * Appends a relationship via a join to this query builder.
109      * 
110      * @param colCmd the class metadata of the relationship column
111      * @param fieldNum the field number of the relationship column
112      * @param col the column object for the relationship field
113      * @param prefix the prefix to use for the join
114      * @param isQuery true if we're appending the relationship field but not all of the fields of the related object
115      */
116     public void appendRelationship(AbstractClassMetaData colCmd, int fieldNum, ColumnImpl col, String prefix, boolean isQuery)  {
117         AbstractMemberMetaData ammd = colCmd.getMetaDataForManagedMemberAtAbsolutePosition(fieldNum);
118         AbstractClassMetaData cmd =
119             PersistenceUtils.getMemberElementClassMetaData(ammd, forceQuery.getExecutionContext().getClassLoaderResolver(),
120                                                             forceQuery.getExecutionContext().getMetaDataManager());
121         
122         if (cmd == null) {
123             /**
124              * This can be true if there are Foreign key fields in Force.com that are not fully mapped in Java, e.g.
125              * User has a ProfileId field but there is no Profile entity in Java. Treat it just as a string field
126              */
127             sb.append(col.getFieldName());
128             return;
129         }
130 
131         TableImpl joinTable = ((ForceStoreManager) forceQuery.getExecutionContext().getStoreManager()).getTable(cmd);
132 
133         TableImpl parentTable = this.table;
134         AbstractClassMetaData parentCmd = this.acmd;
135         this.table = joinTable;
136         this.acmd = cmd;
137         this.fetchDepth++;
138 
139         try {
140             if (isQuery) {
141                 forceQuery.appendRelationshipQuery(this, ammd, col);
142             } else {
143                 forceQuery.appendRelationshipFields(this, col, prefix);
144             }
145         } finally {
146             this.table = parentTable;
147             this.acmd = parentCmd;
148             this.fetchDepth--;
149         }
150     }

152     public StringBuilder getBuilder() {
153         return sb;

165     /**
166      * First looks in filter mappings cache for the filter expression associated
167      * with the given alias, else returns the expression from the compilation.
168      * @param alias the alias of the queried object
169      * @return  the filter expression of a query
170      */
171     public Expression getFilterExpression(String alias) {
172         return aliasToFilterMappings != null ? aliasToFilterMappings.get(alias)
173                                                 : isTopLevel && compilation != null ? compilation.getExprFilter() : null;
174     }

177      * this method tells us whether to skip querying for relationship fields
178      * by comparing the current depth of the query to the maximum.

186     private void initRelatedAliases(QueryCompilation queryCompilation) {
187         if (queryCompilation == null || queryCompilation.getExprFilter() == null) return;
188         Map<String, Expression> ret = new HashMap<String, Expression>();
189         Set<String> mappingAliases = new HashSet<String>();
190         createAliasToFilterMappings(queryCompilation.getExprFilter(), ret, mappingAliases);
191         if (ret.size() == 0) return;
192         this.aliasToFilterMappings = ret;
193         if (mappingAliases.size() == 0) return;
194         
195         // Now setup the relatedJoinAliases
196         this.relatedJoinAliases = new HashMap<TupleName, String>();
197         for (Expression fromExpr : queryCompilation.getExprFrom()) {
198             for (Expression expr = fromExpr.getRight(); expr != null; expr = expr.getRight()) {
199                 if (expr instanceof JoinExpression && mappingAliases.contains(expr.getAlias())) {
200                     List<String> t = ((JoinExpression) expr).getPrimaryExpression().getTuples();
201                     relatedJoinAliases.put(new TupleName(t), expr.getAlias());
202                     /**
203                      * Datanucleus seems to have incorrect alias mapping for this in symbols
204                      */
205                     AbstractMemberMetaData mmd = acmd.getMetaDataForMember(t.get(t.size() - 1));
206                     AbstractClassMetaData cmd = PersistenceUtils.getMemberElementClassMetaData(mmd,
207                                                                     forceQuery.getExecutionContext().getClassLoaderResolver(),
208                                                                     forceQuery.getExecutionContext().getMetaDataManager());
209                     PropertySymbol ps =
210                         new PropertySymbol(expr.getAlias(),
211                             forceQuery.getExecutionContext().getClassLoaderResolver().classForName(cmd.getFullClassName()));
212                     queryCompilation.getSymbolTable().removeSymbol(ps);
213                     queryCompilation.getSymbolTable().addSymbol(ps);
214                 }
215             }
216         }
217     }

219     private String createAliasToFilterMappings(Expression expr, Map<String, Expression> map, Set<String> mappingAliases) {
220         String leftAlias = null;
221         String rightAlias = null;
222         if (expr.getLeft() instanceof DyadicExpression) {
223             leftAlias = createAliasToFilterMappings(expr.getLeft(), map, mappingAliases);
224             if (expr.getRight() instanceof DyadicExpression) {
225                 rightAlias = createAliasToFilterMappings(expr.getRight(), map, mappingAliases);
226             } else {
227                 rightAlias = getAlias(expr.getRight(), mappingAliases);
228             }
229         } else if (expr instanceof DyadicExpression) {
230             leftAlias = getAliasFromDyadictExpression(expr, mappingAliases);
231         } else {
232             leftAlias = getAlias(expr, mappingAliases);
233         }
234         
235         if (leftAlias != null) {
236             if (rightAlias == null || leftAlias.equals(rightAlias)) {
237                 // Left and right are the same so current expression is a potential top level expression
238                 map.put(leftAlias, expr);
239                 return leftAlias;
240             } else {
241                 // Left and right are not the same so we split them and toss the current expression and look further right
242                 map.put(leftAlias, expr.getLeft());
243                 map.put(rightAlias, expr.getRight());
244                 return rightAlias;
245             }
246         }
247         if (rightAlias != null) {
248             // Here Left alias is null but right has an alias current expression is a potential top level expression
249             map.put(rightAlias, expr);
250             return rightAlias;
251         }
252         return null;
253     }

255     private String getAliasFromDyadictExpression(Expression expr, Set<String> mappingAliases) {
256         String leftAlias = getAlias(expr.getLeft(), mappingAliases);
257         String rightAlias = getAlias(expr.getRight(), mappingAliases);
258         if (leftAlias != null && rightAlias != null && !leftAlias.equals(rightAlias)) {
259             throw new NucleusDataStoreException("There cannot be two different aliases in a leaf DyadictExpression: " + expr);
260         }
261         return leftAlias != null ? leftAlias : rightAlias;
262     }
263     
264     private String getAlias(Expression expr, Set<String> mappingAliases) {
265         if (expr instanceof PrimaryExpression) {
266             List<String> tuples = ((PrimaryExpression) expr).getTuples();
267             return tuples.size() > 1 ? tuples.get(0) : null;
268         } else if (expr instanceof InvokeExpression) {
269             return getAliasIfMapExpression((InvokeExpression) expr, mappingAliases);
270         }
271         return null;
272     }
273     
274     private String getAliasIfMapExpression(InvokeExpression expr, Set<String> mappingAliases) {
275         if ("mapValue".equals(expr.getOperation()) || "mapKey".equals(expr.getOperation())
276                 || "mapEntry".equals(expr.getOperation())) {
277             String alias = expr.getLeft().getSymbol().getQualifiedName();
278             mappingAliases.add(alias);
279             return alias;
280         } else {
281             return getAlias(expr.getLeft(), mappingAliases);
282         }
283     }
