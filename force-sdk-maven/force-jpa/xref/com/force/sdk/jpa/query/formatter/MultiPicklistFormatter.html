27  package com.force.sdk.jpa.query.formatter;
28  
29  import java.util.regex.Pattern;
30  
31  /**
32   * This is a convenience class to format a multipicklist value
33   * i.e. 'AAA;BBB,CCC' will be formatted as 'AAA;BBB','CCC'
34   * for use in the expression of a query.
35   *
36   * @author Fiaz Hossain
37   */
38  public class MultiPicklistFormatter {
39  
40      private static final Pattern COMMA_PATTERN = Pattern.compile(",");
41      
42      private String value;
43      
44      /**
45       * Create a formatter for the given string.
46       * 
47       * @param value  the string to be formatted.
48       */
49      public MultiPicklistFormatter(String value) {
50          this.value = value;
51      }
52      
53      /**
54       * Formats the string.
55       * 
56       * @return a string that separates groups of picklist values by ','
57       */
58      public String getFormattedString() {
59          if (value == null || value.length() == 0) return value;
60          String[] values = COMMA_PATTERN.split(value);
61          StringBuilder sb = new StringBuilder(value.length() + values.length * 4);
62          for (int i = 0; i < values.length; i++) {
63              if (i > 0) sb.append(",");
64              sb.append("'").append(values[i]).append("'");
65          }
66          return sb.toString();
67      }
68      
69      @Override
70      public String toString() {
71          return getFormattedString();
72      }
73  }
