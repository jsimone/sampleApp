44  /**
45   * ResultObjectFactory that operates using a QueryResultMetaData and returns objects based on the definition.
46   * A QueryResultMetaData allows for a row of a ResultSet to be returned as a mix of :-
47   * <ul>
48   * <li>a number of persistent objects (made up of several ResultSet columns)</li>
49   * <li>a number of Objects (from individual ResultSet columns)</li>
50   * </ul>
51   * Each call to getObject() will then return a set of objects as per the MetaData definition.
52   * <h3>ResultSet to object mapping</h3>
53   * Each row of the SObject has a set of columns, and these columns are either used for direct outputting
54   * back to the user as a "simple" object, or as a field in a persistent object.
55   * 
56   * @author Fiaz Hossain
57   */
58  public class ResultMetaDataROF {
59      
60      /** MetaData defining the result from the Query. */
61      QueryResultMetaData queryResultMetaData = null;
62  
63      /**
64       * Constructor.
65       * @param qrmd MetaData defining the results from the query.
66       */
67      public ResultMetaDataROF(QueryResultMetaData qrmd) {
68          this.queryResultMetaData = qrmd;
69      }
70  
71      /**
72       * @param ec ExecutionContext
73       * @param mconn ForceManagedConnection
74       * @param query AbstractJavaQuery
75       * @param sobject SObject
76       * @return The object(s) for this row of the SObject. We return persistent object first followed by single columns
77       * in the order they are described.
78       */
79      public Object getObject(final ExecutionContext ec, final ForceManagedConnection mconn,
80              final AbstractJavaQuery query, SObject sobject) {
81          
82          List returnObjects = new ArrayList();
83  
84          // A). Process persistent types
85          PersistentTypeMapping[] persistentTypes = queryResultMetaData.getPersistentTypeMappings();
86          ForceStoreManager storeMgr = (ForceStoreManager) ec.getStoreManager();
87          if (persistentTypes != null) {
88              for (int i = 0; i < persistentTypes.length; i++) {
89                  AbstractClassMetaData acmd =
90                      ec.getMetaDataManager().getMetaDataForClass(persistentTypes[i].getClassName(), ec.getClassLoaderResolver());
91                  returnObjects.add(ec.findObjectUsingAID(
92                          new Type(ec.getClassLoaderResolver().classForName(acmd.getFullClassName())),
93                                      ForceQueryUtils.getFieldValues2(acmd, acmd.getDFGMemberPositions(),
94                                                                      mconn, storeMgr, sobject, query),
95                                      query.getIgnoreCache(), true));
96              }
97          }
98  
99          // B). Process simple columns
100         String[] columns = queryResultMetaData.getScalarColumns();
101         if (columns != null) {
102             for (int i = 0; i < columns.length; i++) {
103                 Object obj = getResultObject(sobject, columns[i]);
104                 if (obj == null) {
105                     throw new NucleusUserException("No data found for column: " + columns[i]);
106                 }
107                 returnObjects.add(obj);
108             }
109         }
110 
111         if (returnObjects.size() == 0) {
112             // No objects so user must have supplied incorrect MetaData
113             return null;
114         } else if (returnObjects.size() == 1) {
115             // Return Object
116             return returnObjects.get(0);
117         } else {
118             // Return Object[]
119             return returnObjects.toArray(new Object[returnObjects.size()]);
120         }
121     }
122 
123     /**
124      * Convenience method to read the value of a column out of an SObject.
125      * @param rs XmlObject
126      * @param columnName Name of the column
127      * @return Value for the column for this row.
128      */
129     private Object getResultObject(final XmlObject rs, String columnName) {
130         if (rs == null || columnName == null) return null;
131         int index = columnName.indexOf('.');
132         if (index > 0) {
133             return getResultObject(rs.getChild(columnName.substring(0, index)), columnName.substring(index + 1));
134         }
135         return rs.getField(columnName);
136     }
