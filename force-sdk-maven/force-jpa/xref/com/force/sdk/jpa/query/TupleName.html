27  package com.force.sdk.jpa.query;
28  
29  import java.util.Collections;
30  import java.util.List;
31  
32  /**
33   * This is a conveneient class to maintain short and long names for a tuple i.e. c as short and a.b.c
34   * as long name.
35   *
36   * @author Fiaz Hossain
37   */
38  public class TupleName {
39  
40      List<String> tuple;
41      
42      /**
43       * Create a tuple name out of a list.
44       * 
45       * @param tuple  A list of strings that make up the name
46       */
47      public TupleName(List<String> tuple) {
48          this.tuple = tuple;
49      }
50      
51      /**
52       * Create a tuple name from just one string.
53       * 
54       * @param name  the name of the tuple
55       */
56      public TupleName(String name) {
57          this.tuple = Collections.singletonList(name);
58      }
59      
60      public List<String> getTuple() {
61          return tuple;
62      }
63      
64      /**
65       * Return tuple short name.
66       * 
67       * @return tuple short name
68       */
69      public String getShortName() {
70          return tuple.get(tuple.size() - 1);
71      }
72      
73      /**
74       * Return the prefix that when combined with the tuple
75       * short name creates the tuple long name.
76       * <p>
77       * E.g. If tuple long name is a.b.c, the short name prefix is a.b
78       * 
79       * @return short name prefix
80       */
81      public String getShortNamePrefix() {
82          StringBuilder sb = new StringBuilder();
83          for (int i = 0; i < tuple.size() - 1; i++) {
84              if (sb.length() > 0) sb.append(".");
85              sb.append(tuple.get(i));
86          }
87          return sb.toString();
88      }
89      
90      /**
91       * Return the tuple long name.
92       * 
93       * @return fully qualified tuple name
94       */
95      public String getLongName() {
96          StringBuilder sb = new StringBuilder();
97          for (String s : tuple) {
98              if (sb.length() > 0) sb.append(".");
99              sb.append(s);
100         }
101         return sb.toString();
102     }
103     
104     @Override
105     public int hashCode() {
106         return getShortName().hashCode();
107     }
108     
109     @Override
110     public boolean equals(Object obj) {
111         if (this == obj) return true;
112         if (obj instanceof TupleName)  {
113             TupleName other = (TupleName) obj;
114             return tuple.get(tuple.size() - 1).equals(other.tuple.get(other.tuple.size() - 1));
115         }
116         return false;
117     }
118     
119     @Override
120     public String toString() {
121         return getLongName();
122     }
123 }
