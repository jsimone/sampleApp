27  package com.force.sdk.jpa.exception;
28  
29  import org.datanucleus.exceptions.*;
30  import org.datanucleus.store.query.NoQueryResultsException;
31  
32  import com.sforce.soap.partner.fault.ApiFault;
33  
34  /**
35   * Maps Force.com API Exceptions to appropriate JPA Exceptions.
36   *
37   * @author Tim Kral
38   */
39  public final class ForceApiExceptionMap {
40  
41      private ForceApiExceptionMap() {  }
42      
43      /**
44       * Method to map specific api exception codes to the proper DataNucleus
45       * exception.
46       * 
47       * @param af  the Force.com ApiFault exception to map
48       * @param isQuery whether the exception happened during a query or not
49       * @param isOptimistic whether JPA transactions are in optimistic mode
50       * @return the properly mapped NucleusException
51       */
52      public static NucleusException mapToNucleusException(ApiFault af, boolean isQuery, boolean isOptimistic) {
53          switch (af.getExceptionCode()) {
54          case INVALID_QUERY_FILTER_OPERATOR:
55              if (isQuery) {
56                  // For queries with an invalid ID field message, treat them
57                  // just like any other query for which we have no results.
58                  String exceptionMsg = af.getExceptionMessage();
59                  if (exceptionMsg != null && exceptionMsg.contains("invalid ID field")) {
60                      return new NoQueryResultsException("invalid ID field");
61                  }
62                  
63                  return new NucleusUserException(af.toString(), af);
64              } else {
65                  // For non-queries (e.g. em.find), the SOQL query is pre-defined
66                  // which means we likely have an invalid ID field.
67                  // So treat this case as if we found no data. 
68                  return new NucleusObjectNotFoundException();
69              }
70          case INVALID_SOAP_HEADER:
71              if (isOptimistic) {
72                  return new NucleusDataStoreException("Your organization does not have the necessary permissions "
73                      + "to use optimistic transactions. Please contact salesforce.com to have this permission enabled.");
74              }
75              // fall through to default
76          default:
77              return new NucleusDataStoreException(af.toString(), af);
78          }
79      }
80  }
