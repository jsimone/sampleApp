46  /**
47   * 
48   * Object manager that manages the objects being created, updated, or deleted.
49   * Special handling for all or nothing operations.
50   *
51   * @author Fiaz Hossain
52   */
53  public class ForceObjectManagerImpl extends ObjectManagerImpl {
54  
55      private final boolean allOrNothingEnabled;
56      private boolean inAllOrNothingMode;
57      private LinkedHashMap<ObjectProvider, SObject> createObjectList;
58      private IdentityHashMap<Object, SObject> pcToSObject;
59      private List<SObject> updateObjectList;
60      private List<Calendar> versionList;
61      private List<String> deleteObjectList;
62      
63      /**
64       * create an object manager with datastore credentials.
65       * 
66       * @param omf  the object manager factory
67       * @param owner  the owning persistence manager or entity manager
68       * @param userName  the username to the datastore
69       * @param password  the password to the datastore
70       */
71      public ForceObjectManagerImpl(ObjectManagerFactoryImpl omf, Object owner, String userName, String password) {
72          super(omf, owner, userName, password);
73          this.allOrNothingEnabled = omf.getOMFContext().getPersistenceConfiguration().getBooleanProperty("force.AllOrNothing");
74      }
75      
76      /**
77       * check for whether an active transaction is currently flushing data to the datastore in all or nothing mode.
78       * 
79       * @return true if all or nothing mode is enabled and we are currently flushing data in this mode
80       */
81      public boolean isInAllOrNothingMode() {
82          return allOrNothingEnabled && this.inAllOrNothingMode;
83      }
84      
85      /**
86       * This method flushes all dirty, new, and deleted instances to the
87       * datastore. It has no effect if a transaction is not active. If a
88       * datastore transaction is active, this method synchronizes the cache with
89       * the datastore and reports any exceptions. If an optimistic transaction is
90       * active, this method obtains a datastore connection and synchronizes the
91       * cache with the datastore using this connection. The connection obtained
92       * by this method is held until the end of the transaction.
93       * 
94       * @param flushToDatastore Whether to ensure any changes reach the datastore
95       *     Otherwise they will be flushed to the datastore manager and leave it to
96       *     decide the opportune moment to actually flush them to the datastore
97       */
98      @Override
99      public synchronized void flushInternal(boolean flushToDatastore) {
100         if (flushToDatastore && allOrNothingEnabled) {
101             inAllOrNothingMode = true;
102             try {
103                 super.flushInternal(flushToDatastore);
104                 if (createObjectList != null) {
105                     ((ForcePersistenceHandler) getStoreManager().getPersistenceHandler()).createObjects(createObjectList.values(),
106                             createObjectList.keySet(), getExecutionContext());
107                 }
108                 if (updateObjectList != null) {
109                     ((ForcePersistenceHandler) getStoreManager().getPersistenceHandler())
110                         .updateObjects(updateObjectList.toArray(new SObject[updateObjectList.size()]),
111                             versionList.toArray(new Calendar[versionList.size()]), getExecutionContext());
112                 }
113                 if (deleteObjectList != null) {
114                     ((ForcePersistenceHandler) getStoreManager().getPersistenceHandler())
115                         .deleteObjects(deleteObjectList.toArray(new String[deleteObjectList.size()]), getExecutionContext());
116                 }
117             } catch (NucleusOptimisticException noe) {
118                 throw new NucleusOptimisticException(LOCALISER.msg("010031"), noe.getFailedObject());
119             } finally {
120                 inAllOrNothingMode = false;
121                 createObjectList = null;
122                 updateObjectList = null;
123                 versionList = null;
124                 deleteObjectList = null;
125             }
126         } else {
127             super.flushInternal(flushToDatastore);
128         }
129     }

131     /**
132      * Retrieve the Force.com object (SObject) for the given parent.
133      * 
134      * @param parent Object
135      * @return the Force.com object (SObject) corresponding to the given parent
136      */
137     public synchronized SObject getParentSObject(Object parent) {
138         SObject parentSObject;
139         if (pcToSObject == null || (parentSObject = pcToSObject.get(parent)) == null) {
140             throw new NucleusDataStoreException("Parent entity has not been saved");
141         }
142         return parentSObject;
143     }
144     
145     /**
146      * add to the current list of entities to be created.
147      * 
148      * @param object  the object to be created
149      * @param op  the object provider
150      */
151     public synchronized void addToCreateList(SObject object, ObjectProvider op) {
152         if (createObjectList == null) {
153             createObjectList = new LinkedHashMap<ObjectProvider, SObject>();
154             pcToSObject = new IdentityHashMap<Object, SObject>();
155         }
156         createObjectList.put(op, object);
157         pcToSObject.put(op.getObject(), object);
158     }
159     
160     /**
161      * add to the current list of entities to be updated.
162      * 
163      * @param object  the object to update (complete with updated fields) 
164      * @param version  this should be the time of modification. Pass in a version if we're checking if-modified-before headers
165      */
166     public synchronized void addToUpdateList(SObject object, Calendar version) {
167         if (updateObjectList == null) {
168             updateObjectList = new ArrayList<SObject>();
169             versionList = new ArrayList<Calendar>();
170         }
171         updateObjectList.add(object);
172         if (version != null) {
173             versionList.add(version);
174         }
175     }
176     
177     /**
178      * add to the current list of entities to be deleted.
179      * 
180      * @param id the id of the entity to delete
181      */
182     public synchronized void addToDeleteList(String id) {
183         if (deleteObjectList == null) {
184             deleteObjectList = new ArrayList<String>();
185         }
186         deleteObjectList.add(id);
187     }

189     /**
190      * Method to mark an object (StateManager) as dirty.
191      * @param sm The StateManager
192      * @param directUpdate Whether the object has had a direct update made on it (if known)
193      */
194     @Override
195     public synchronized void markDirty(StateManager sm, boolean directUpdate) {
196         if (!(sm instanceof ForceJPAStateManagerImpl)) {
197             super.markDirty(sm, directUpdate);
198         }
199     }

201     /**
202      * There are some cases when a postrollback tries to detach an object that has not been persisted.
203      * In the case of detachment the object is attempted to be reloaded which throws an NucleusObjectNotFoundException
204      * since the object was never stored in the db. We can ignore that exception.
205      */
206     @Override
207     public synchronized void postRollback() {
208         try {
209             super.postRollback();
210         } catch (NucleusObjectNotFoundException ne) {
211             // We can ignore this exception
212         }
213     }
214     
215     /**
216      * We add this method so that we can support detach of a newly created persistent object that has never been saved to the db.
217      * {@inheritDoc}
218      */
219     @Override
220     public synchronized void detachObject(Object obj, FetchPlanState state) {
221         StateManager sm = findStateManager(obj);
222         super.detachObject(obj, state);
223         if (sm != null) {
224             clearDirty(sm);
225         }
226     }
