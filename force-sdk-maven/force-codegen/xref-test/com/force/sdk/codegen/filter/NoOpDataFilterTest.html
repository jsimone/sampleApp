45      @Test
46      public void testFilterWithNullValue() {
47          assertNull(new NoOpDataFilter().filter(null), "A no op filter of a null value should be null");

49      
50      @Test
51      public void testFilterWithNonNullValue() {
52          DescribeSObjectResult dsr = new DescribeSObjectResult();
53          dsr.setName("Object_Name__c");
54          
55          List<DescribeSObjectResult> dsrs =
56              new NoOpDataFilter().filter(Collections.<DescribeSObjectResult>singletonList(dsr));
57          
58          assertNotNull(dsrs, "A no op filter of a non-null value should be non-null");
59          assertEquals(dsrs.size(), 1, "Unexpected number of DescribeSObjectResults after no op filter");
60          assertEquals(dsrs.get(0).getName(), "Object_Name__c", "Unexpected DescribeSObjectResult after no op filter");
61      }
