41  public class ForcePicklistEnumBuilderTest {
42  
43      @DataProvider
44      protected Object[][] picklistEntryProvider() {
45          
46          return new Object[][] {
47                  {true, false, "Basic Picklist Value", "BasicPicklistValue",
48                      "BASICPICKLISTVALUE(true,false,\"Basic Picklist Value\",\"BasicPicklistValue\"),"},
49                  {false, true, "Inactive With Default", "InactiveWithDefault",
50                      "INACTIVEWITHDEFAULT(false,true,\"Inactive With Default\",\"InactiveWithDefault\"),"},
51                  {true, false, "Non White Space", "Non!White@Space#$",
52                      "NON_WHITE_SPACE__(true,false,\"Non White Space\",\"Non!White@Space#$\"),"},
53                  {true, false, null, "NullLabel",
54                      "NULLLABEL(true,false,null,\"NullLabel\"),"},
55                  {true, false, "Short Value", "S",
56                      "SHORT_VALUE(true,false,\"Short Value\",\"S\"),"},
57                  {true, false, "0", "0",
58                      "VALUE_0(true,false,\"0\",\"0\"),"},
59                  {true, false, "0123", "0123",
60                      "VALUE_0123(true,false,\"0123\",\"0123\"),"},
61                  {true, false, "5 Minute Down Time", "5MinuteDownTime",
62                      "VALUE_5MINUTEDOWNTIME(true,false,\"5 Minute Down Time\",\"5MinuteDownTime\"),"},
63              };
64      }
65      
66      @Test(dataProvider = "picklistEntryProvider")
67      public void testBasicEnumValue(boolean isActive, boolean isDefaultValue, String label, String value,
68              String expectedRenderedString) {
69          
70          PicklistEntry picklistEntry = new PicklistEntry();
71          picklistEntry.setActive(isActive);
72          picklistEntry.setDefaultValue(isDefaultValue);
73          picklistEntry.setLabel(label);
74          picklistEntry.setValue(value);
75          
76          ForcePicklistEnumBuilder builder = new ForcePicklistEnumBuilder(0);
77          builder.add(picklistEntry);
78          assertEquals(builder.toString(), expectedRenderedString, "Unexpected result for ForcePicklistEnumBuilder");
79      }
80      
81      @Test
82      public void testDuplicateEnumValues() {
83          PicklistEntry picklistEntry1 = new PicklistEntry();
84          picklistEntry1.setLabel("label");
85          picklistEntry1.setValue("value");
86          
87          PicklistEntry picklistEntry2 = new PicklistEntry();
88          picklistEntry2.setLabel("label");
89          picklistEntry2.setValue("value");
90          
91          ForcePicklistEnumBuilder builder = new ForcePicklistEnumBuilder(0);
92          builder.add(picklistEntry1);
93          builder.add(picklistEntry2);
94          
95          assertEquals(builder.toString(),
96                  "VALUE(false,false,\"label\",\"value\"),\n"
97                  + "//VALUE(false,false,\"label\",\"value\"),",
98                  "Unexpected result for ForcePicklistEnumBuilder");
99      }
