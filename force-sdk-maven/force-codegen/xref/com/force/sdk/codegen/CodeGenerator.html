42  /**
43   * Basic Code Generator.
44   * <p>
45   * The generator will load up all Force.com schema object (SObject) information 
46   * for a given Force.com store (organization). It then filters and selects
47   * the data that is necessary for code generation and writes
48   * this to a {@code Template}.
49   * <p>
50   * Notice that code generation is analogous to MVC (well, at
51   * least the MV parts): 
52   * <p>
53   * <ol>
54   *   <li>
55   *    Model = {@code DataFilter}, {@code DataSelector}. This provides the data that will
56   *    be used in the template.
57   *    </li>
58   *    <li>
59   *    View = {@code Template}.  This represents the physical layout of
60   *    the code being generated.
61   *    </li>
62   *    <li>
63   *    A {@code WriterProvider} provides Jave {@code Writer}s to the {@code CodeGenerator}
64   *    to tell it where to write the generated code. 
65   *    </li>
66   * </ol>
67   *
68   * @author Tim Kral
69   */
70  public final class CodeGenerator {
71  
72      private static final int MAX_BATCH_DESCRIBE_SIZE = 100;
73      
74      final DataFilter filter;
75      final DataSelector selector;
76      final Template template;
77      final WriterProvider writerProvider;
78      
79      private CodeGenerator(Builder b) {
80          this.filter = b.filter == null ? new NoOpDataFilter() : b.filter;
81          this.selector = b.selector == null ? new DefaultDataSelector() : b.selector;
82          
83          if (b.template == null)
84              throw new IllegalArgumentException("Cannot build a CodeGenerator with a null template");
85          this.template = b.template;
86          
87          if (b.writerProvider == null)
88              throw new IllegalArgumentException("Cannot build a CodeGenerator with a null writer provider");
89          this.writerProvider = b.writerProvider;
90      }

92      /**
93       * Generates code based on a connection to a Force.com store (organization).
94       * <p>
95       * This will make various Force.com API calls to a Force.com store.  First,
96       * it will run a global describe.  Next, it will run full describes on the
97       * global describe results.  These calls will be batched in the most efficient
98       * way possible (that is, making the least number of Force.com API calls).
99       * <p>
100      * The full describe results will be filtered and injected into a {@code Template}
101      * whose contents will then be written to a Java {@code Writer}.
102      * 
103      * @param conn a Force.com API connection to the Force.com store (organization)
104      *             that is to be described for code generation
105      * @return the number of Force.com objects for which code is generated
106      * @throws ConnectionException if an error occurs while connecting to the Force.com
107      *                             store (organization)
108      * @throws IOException if an error occurs while writing the generated code
109      */
110     public int generateCode(PartnerConnection conn) throws ConnectionException, IOException {
111         
112         // Get all known SObjects in the org
113         List<String> allOrgObjectNames = new ArrayList<String>();
114         for (DescribeGlobalSObjectResult dgsr : conn.describeGlobal().getSobjects()) {
115             allOrgObjectNames.add(dgsr.getName());
116         }
117         
118         // Describe all known SObjects in the org.  Batch by the appropriate size.
119         // TODO: Is this ok to read entirely into memory?
120         List<DescribeSObjectResult> allOrgObjects = new ArrayList<DescribeSObjectResult>();
121         for (int i = 0; i < allOrgObjectNames.size(); i += MAX_BATCH_DESCRIBE_SIZE) {
122             // Ensure that our range only goes to the end of the allOrgObjectNames array
123             int endIndex = i + MAX_BATCH_DESCRIBE_SIZE;
124             if (endIndex > allOrgObjectNames.size()) endIndex = allOrgObjectNames.size();
125             
126             List<String> objectNameBatch = allOrgObjectNames.subList(i, endIndex);
127             DescribeSObjectResult[] dsrs =
128                 conn.describeSObjects(objectNameBatch.toArray(new String[objectNameBatch.size()]));
129             
130             allOrgObjects.addAll(Arrays.<DescribeSObjectResult>asList(dsrs));
131         }
132         
133         // Get the user information
134         GetUserInfoResult userInfo = conn.getUserInfo();
135         
136         // Write filtered data to the template
137         int numGeneratedCode = 0;
138         for (DescribeSObjectResult dsr : filter.filter(allOrgObjects)) {
139             // Before we write a new source file, make sure the template is reset
140             template.reset();
141             
142             // Select the data that we're interested in 
143             selector.select(userInfo, dsr, template);
144             
145             Writer writer = null;
146             try {
147                 writer = writerProvider.getWriter(userInfo, dsr);
148                 template.write(writer);
149                 numGeneratedCode++;
150             } finally {
151                 if (writer != null) writer.close();
152             }
153         }
154         
155         return numGeneratedCode;
156     }

158     /**
159      * Builder pattern for {@code CodeGenerator}.
160      */
161     public static class Builder {
162         DataFilter filter;
163         DataSelector selector;
164         Template template;
165         WriterProvider writerProvider;
166         
167         /**
168          * Injects a {@code DataFilter} into this {@code Builder}.
169          * <p>
170          * When this {@code Builder} builds a {@code CodeGenerator} object, 
171          * that object will use the given {@code DataFilter}.
172          * 
173          * @param filter the {@code DataFilter} to be used in the built
174          *               {@code CodeGenerator} object
175          * @return instance of this {@code Builder}
176          */
177         public Builder filter(DataFilter filter) {
178             this.filter = filter;
179             return this;
180         }
181         
182         /**
183          * Injects a {@code DataSelector} into this {@code Builder}.
184          * <p>
185          * When this {@code Builder} builds a {@code CodeGenerator} object, 
186          * that object will use the given {@code DataSelector}.
187          * 
188          * @param selector the {@code DataSelector} to be used in the built
189          *                 {@code CodeGenerator} object
190          * @return instance of this {@code Builder}
191          */
192         public Builder selector(DataSelector selector) {
193             this.selector = selector;
194             return this;
195         }
196 
197         /**
198          * Injects a {@code Template} into this {@code Builder}.
199          * <p>
200          * When this {@code Builder} builds a {@code CodeGenerator} object, 
201          * that object will use the given {@code Template}.
202          * 
203          * @param template the {@code Template} to be used in the built
204          *                 {@code CodeGenerator} object
205          * @return instance of this {@code Builder}
206          */
207         public Builder template(Template template) {
208             this.template = template;
209             return this;
210         }
211         
212         /**
213          * Injects a {@code WriterProvider} into this {@code Builder}.
214          * <p>

216          * that object will use the given {@code WriterProvider}.
217          * 
218          * @param writerProvider the {@code WriterProvider} to be used in the built
219          *                       {@code CodeGenerator} object
220          * @return instance of this {@code Builder}
221          */
222         public Builder writerProvider(WriterProvider writerProvider) {
223             this.writerProvider = writerProvider;
224             return this;
225         }
226         
227         /**
228          * Builds a {@code CodeGenerator} object.
229          * 
230          * @return a {@code CodeGenerator} object with the state
231          *         contained in this {@code Builder}

233         public CodeGenerator build() {
234             return new CodeGenerator(this);
235         }
