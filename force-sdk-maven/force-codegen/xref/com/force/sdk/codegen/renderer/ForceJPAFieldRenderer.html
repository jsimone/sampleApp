46  /**
47   * A StringTemplate {@code AttributeRenderer} that renders Java field level {@code String}s.
48   * <p>
49   * This {@code AttributeRenderer} effectively translates between a Force.com API {@code Field}
50   * object and Java field level code.  Within StringTemplate, it is meant to be registered
51   * as an {@code AttributeRenderer} for a Force.com API {@code Field} object.
52   *
53   * @author Tim Kral
54   */
55  public class ForceJPAFieldRenderer implements AttributeRenderer {
56  
57      private static final Pattern ANNOTATION_PATTERN = Pattern.compile("^(enum|getter|setter)Annotation(\\d+)$");
58      private static final Pattern ENUM_VALUES_PATTERN = Pattern.compile("^enumValues(\\d+)$");
59      private static final Pattern FIELD_COMMENTS_PATTERN = Pattern.compile("^fieldComments(\\d+)$");
60      
61      @Override
62      public String toString(Object o) {
63          return "";
64      }
65  
66      @Override
67      public String toString(Object o, String format) {
68          Field field = (Field) o;
69          Matcher matcher;
70          
71          if ("fieldName".equals(format)) {
72              return renderJavaName(field, true /*firstCharLowerCase*/);
73          } else if ("fieldType".equals(format)) {
74              return renderFieldType(field);
75          } else if ((matcher = FIELD_COMMENTS_PATTERN.matcher(format)).matches()) {
76              // The number at the end of an fieldComments format tells
77              // us the number of indents to add to the fieldComments
78              String numIndentsStr = matcher.group(1);
79              JavaCommentBuilder builder = new JavaCommentBuilder(Integer.valueOf(numIndentsStr));
80              
81              return renderFieldComments(field, builder);
82          } else if (format != null && format.contains("fieldComments")) {
83              throw new IllegalArgumentException("Unrecognized format: " + format + ". "
84                  + "fieldComments formats must end in the number of indents required (e.g. fieldComments1");
85          } else if ((matcher = ANNOTATION_PATTERN.matcher(format)).matches()) {
86              String annotationType = matcher.group(1);
87              
88              // The number at the end of an annotation format tells
89              // us the number of indents to add to the annotations
90              String numIndentsStr = matcher.group(2);
91              JPAAnnotationBuilder builder = new JPAAnnotationBuilder(Integer.valueOf(numIndentsStr));
92              
93              if ("enum".equals(annotationType)) {
94                  return renderEnumAnnotation(field, builder);
95              } else if ("getter".equals(annotationType)) {
96                  return renderGetterAnnotation(field, builder);
97              } else if ("setter".equals(annotationType)) {
98                  return renderSetterAnnotation(field, builder);
99              }
100         } else if (format != null && format.contains("Annotation")) {
101             throw new IllegalArgumentException("Unrecognized annotation format: " + format + ". "
102                     + "Annotation formats must be one of (enum, getter, setter) "
103                     + "and must end in the number of indents required (e.g. getterAnnotation1");
104         } else if ("methodName".equals(format)) {
105             return renderJavaName(field, false /*firstCharLowerCase*/);
106         } else if ("enumName".equals(format)) {
107             return renderEnumName(field);
108         } else if ((matcher = ENUM_VALUES_PATTERN.matcher(format)).matches()) {
109             // The number at the end of an enumValues format tells
110             // us the number of indents to add to the enumValues
111             String numIndentsStr = matcher.group(1);
112             ForcePicklistEnumBuilder builder = new ForcePicklistEnumBuilder(Integer.valueOf(numIndentsStr));
113             
114             return renderEnumValues(field, builder);
115         } else if (format != null && format.contains("enumValues")) {
116             throw new IllegalArgumentException("Unrecognized format: " + format + ". "
117                 + "enumValues formats must end in the number of indents required (e.g. enumValues1");
118         }
119         
120         return toString(o);
121     }

123     // Render the Java field type for a given API field
124     private String renderFieldType(Field field) {
125         FieldType type = field.getType();
126         if (type == FieldType._boolean
127             || type == FieldType.combobox) {
128             return boolean.class.getName();
129         } else if (type == FieldType._double
130                    || type == FieldType.percent) {
131             return double.class.getName();
132         } else if (type == FieldType._int) {
133             return int.class.getName();
134         } else if (type == FieldType.currency) {
135             return BigDecimal.class.getName();
136         } else if (type == FieldType.date
137                    || type == FieldType.time) {
138             return Date.class.getName();
139         } else if (type == FieldType.datetime) {
140             return Calendar.class.getName();
141         } else if (type == FieldType.url) {
142             return URL.class.getName();
143         } else if (type == FieldType.picklist) {
144             // Force the field type to be an enum for restricted picklists
145             // with API enabled picklist values
146             if (field.isRestrictedPicklist() && field.getPicklistValues().length > 0) return renderEnumName(field);
147             
148             return String.class.getSimpleName();
149         } else if (type == FieldType.multipicklist) {
150             // Force the field type to be an enum array for restricted multipicklists
151             // with API enabled picklist values
152             if (field.isRestrictedPicklist() && field.getPicklistValues().length > 0) return renderEnumName(field) + "[]";
153             
154             return String[].class.getSimpleName();
155         } else if (type == FieldType.reference) {
156             String[] referenceTo;
157             // Return the class name of the relationship if there's only one relationship object
158             // Otherwise, we'll just use a String
159             if ((referenceTo = field.getReferenceTo()).length == 1) {
160                 DescribeSObjectResult dsr = new DescribeSObjectResult();
161                 dsr.setName(referenceTo[0]);
162             
163             return renderJavaName(dsr, false /*firstCharLowerCase*/);
164             }
165         }
166         
167         return String.class.getSimpleName();
168     }

170     private String renderFieldComments(Field field, JavaCommentBuilder builder) {
171         if (field.getType() == FieldType.reference
172             && field.getReferenceTo().length > 1) {
173             
174             builder.add(renderJavaName(field, true) + " possible references:");
175             for (String refTo : field.getReferenceTo()) {
176                 builder.add(refTo);
177             }
178         }
179         
180         return builder.toString();
181     }

183     private String renderEnumAnnotation(Field field, JPAAnnotationBuilder builder) {
184         builder.add(Generated.class,
185                 Collections.<String, String>singletonMap("value", "\"" + ForceJPAClassGenerator.class.getName() + "\""));
186         
187         return builder.toString();
188     }

190     // Render annotations that are to be add to a JPA object
191     // at the method level
192     private String renderGetterAnnotation(Field field, JPAAnnotationBuilder builder) {
193         
194         String name = field.getName();
195         FieldType type = field.getType();
196         if (type == FieldType.id) {
197             builder.add(Id.class);
198             builder.add(GeneratedValue.class,
199                     Collections.<String, String>singletonMap("strategy", "GenerationType.IDENTITY"));
200         } else if (type == FieldType.reference) {
201             builder.add(ManyToOne.class);
202             
203             // Require reference fields to be lazily loaded.
204             // Developers may override this in subclasses.
205             builder.add(Basic.class,
206                     Collections.<String, String>singletonMap("fetch", "FetchType.LAZY"));
207             
208         // The LastModifiedDate field is used for versioning
209         // so add a @Version annotation
210         } else if (type == FieldType.datetime && name != null && "lastmodifieddate".equals(name.toLowerCase())) {
211             builder.add(Version.class);
212         } else if (type == FieldType.url) {
213             builder.add(Basic.class);
214             
215         // Restricted picklists must conform to an Enum value
216         } else if (field.isRestrictedPicklist()) {
217             builder.add(Enumerated.class,
218                     Collections.<String, String>singletonMap("value", "EnumType.STRING"));
219         }
220         
221         // Add an @Column annotation to all custom fields
222         // so we know the proper Force.com name of the field
223         // Also, relationship fields will be named by their
224         // relationship name.
225         if (field.isCustom()
226                 || (type == FieldType.reference && field.getRelationshipName() != null)) {
227             builder.add(Column.class,
228                     Collections.<String, String>singletonMap("name", "\"" + name + "\""));
229         }
230         
231         return builder.toString();
232     }
233     
234     private String renderSetterAnnotation(Field field, JPAAnnotationBuilder builder) {
235         
236         // Mark non nullable, non defaulted fields as required
237         if (!field.isNillable() && !field.isDefaultedOnCreate()) {
238             builder.add(Basic.class,
239                     Collections.<String, String>singletonMap("optional", "false"));
240         }
241 
242         return builder.toString();
243     }
244     
245     private String renderEnumName(Field field) {
246         StringBuffer enumName = new StringBuffer(renderJavaName(field, false));
247         enumName.append("Enum");
248         
249         return enumName.toString();
250     }
251     
252     private String renderEnumValues(Field field, ForcePicklistEnumBuilder builder) {
253         for (PicklistEntry picklistEntry : field.getPicklistValues()) {
254             builder.add(picklistEntry);
255         }
256         
257         return builder.toString();
258     }
