41  /**
42   * Shared utilities for the Force.com JPA enabled Java class generator.
43   *
44   * @author Tim Kral
45   */
46  public final class ForceJPAClassGeneratorUtils {
47      
48      /**
49       * Represents a tab (consisting of spaces). 
50       */
51      public static final String INDENT = "    ";
52      
53      /**
54       * Represents a system independent new line character.
55       */
56      public static final String NEWLINE = System.getProperty("line.separator");
57      
58      // The number of characters in a custom postfix (e.g. __c)
59      private static final int CUSTOM_POSTFIX_LENGTH = 3;
60      
61      /**
62       * The set of fields common to all Force.com objects. 
63       */
64      public static final Set<String> ALL_OBJECT_COMMON_FIELDS =
65          ImmutableSet.<String>of("id");
66      
67      /**
68       * The set of fields common to *most* Force.com standard objects.
69       */
70      public static final Set<String> STANDARD_OBJECT_COMMON_FIELDS =
71          ImmutableSet.<String>copyOf(Sets.difference(ForceColumnMetaData.STANDARD_FIELDS,
72                  Sets.newHashSet("createdbyid", "lastmodifiedbyid", "isdeleted")));
73      
74      /**
75       * The set of fields common to all Force.com custom objects.
76       */
77      public static final Set<String> CUSTOM_OBJECT_COMMON_FIELDS =
78          ImmutableSet.<String>copyOf(Sets.difference(ForceColumnMetaData.STANDARD_FIELDS,
79                  Sets.newHashSet("createdbyid", "lastmodifiedbyid")));
80      
81      private ForceJPAClassGeneratorUtils() {  }
82      
83      /**
84       * Constructs a Java package name based on a Force.com store name
85       * (Organization name).
86       * 
87       * @param userInfo the Force.com user who is running the code generation
88       * @return a non {@code null} {@code String} which conforms
89       *         to Java class naming standards
90       */
91      public static String constructPackageName(GetUserInfoResult userInfo) {
92          String orgName = userInfo.getOrganizationName();
93          
94          if (orgName != null) {
95              String orgNameDenorm = userInfo.getOrganizationName().replaceAll("(,|\\.|\\s)", "").toLowerCase();
96              return "com." + orgNameDenorm + ".model";
97          }
98          
99          return "com.force.model";
100     }
101 
102     /**
103      * Determines if a Force.com object has all standard Force.com fields.
104      * 
105      * @param dsr a Force.com {@code DescribeSObjectResult} of the object
106      *            to be tested
107      * @return {@code true} if and only if the given object contains all
108      *         common Force.com fields
109      * @see ForceJPAClassGeneratorUtils#STANDARD_OBJECT_COMMON_FIELDS
110      */
111     public static boolean hasAllCommonFields(DescribeSObjectResult dsr) {
112         // Custom object always have all common fields
113         if (dsr.isCustom()) return true;
114         
115         // Gather up all the field names for this standard object
116         Set<String> fieldNameSet = new HashSet<String>();
117         for (Field field : dsr.getFields()) {
118             fieldNameSet.add(field.getName().toLowerCase());
119         }
120         
121         return fieldNameSet.containsAll(STANDARD_OBJECT_COMMON_FIELDS);
122     }

124     /**
125      * Constructs a valid Java name from a Force.com {@code DescribeSObjectResult} object name.
126      * 
127      * @param dsr a Force.com {@code DescribeSObjectResult} object whose name is to
128      *            to converted to a Java name
129      * @param firstCharLowerCase indicates if the first character in the constructed Java
130      *                           name should be lower case
131      * @return a non {@code null} {@code String} which conforms
132      *         to Java naming standards
133      */
134     public static String renderJavaName(DescribeSObjectResult dsr, boolean firstCharLowerCase) {
135         return renderJavaName(dsr.getName(), "Entity", firstCharLowerCase);
136     }
137 
138     /**
139      * Determines if a Java name should be constructed from the given
140      * Force.com API {@code Field} object's relationship name.
141      * 
142      * @param field the Force.com API {@code Field} object to test
143      * @return {@code true} if and only if the given {@code Field}
144      *         object is a reference field and has a non {@code null} relationship
145      *         name and only references exactly one other Force.com object
146      */
147     public static boolean useRelationshipName(Field field) {
148         return field.getType() == FieldType.reference && field.getRelationshipName() != null
149                // With more than one reference, we'll revert back to the field name (and String type)
150                && field.getReferenceTo().length == 1;
151     }
152     
153     /**
154      * Constructs a valid Java name from a Force.com API {@code Field} object name.
155      * 
156      * @param field a Force.com API {@code Field} object whose name is to
157      *              to converted to a Java name
158      * @param firstCharLowerCase indicates if the first character in the constructed Java
159      *                           name should be lower case
160      * @return a non {@code null} {@code String} which conforms
161      *         to Java naming standards
162      */
163     public static String renderJavaName(Field field, boolean firstCharLowerCase) {
164         if (useRelationshipName(field)) {
165             return renderJavaName(field.getRelationshipName(), "Field", firstCharLowerCase);
166         }
167         
168         return renderJavaName(field.getName(), "Field", firstCharLowerCase);
169     }
170     
171     // Convert the given name into a suitable Java name
172     private static String renderJavaName(String name, String keywordSuffix, boolean firstCharLowerCase) {
173         // First, strip off any custom suffix
174         if (name.endsWith("__c") || name.endsWith("__r")) {
175             name = name.substring(0, name.length() - CUSTOM_POSTFIX_LENGTH);
176         }
177 
178         // Convert to camelCase
179         name = WordUtils.capitalize(name, new char[]{'_'});
180         
181         //Remove all underscores ('_')
182         name = name.replace("_", "");
183         
184         // If the name we're going to generate is a Java keyword
185         // then we'll tweak it with the suffix argument
186         if (SourceVersion.isKeyword(name.toLowerCase())) {
187             name = name + keywordSuffix;
188         }
189         
190         // Enforce the desired case on the first character
191         if (firstCharLowerCase) {
192             name = WordUtils.uncapitalize(name);
193         } else {
194             name = WordUtils.capitalize(name);
195         }
196         
197         return name;
198     }
