27  package com.force.sdk.codegen.builder;
28  
29  import java.lang.annotation.Annotation;
30  import java.util.Map;
31  
32  /**
33   * Builds JPA annotation {@code String}s that can be included in generated source code.
34   * <p>
35   * {@code JPAAnnotationBuilder} takes in {@code Annotation} classes and converts those
36   * into {@code String}s that can be added to Java source code.    
37   * <p>
38   * Built annotations will be in the form:
39   * <p>
40   *   {@code @<annotationName>}
41   *   
42   * @author Tim Kral
43   */
44  public class JPAAnnotationBuilder extends BaseBuilder<Class<? extends Annotation>> {
45      
46      /**
47       * Initializes a {@code JPAAnnotationBuilder} with the number
48       * of intentions that should proceed each annotation.
49       * 
50       * @param numIndents the number of indentations that should
51       *                   proceed each annotation
52       */
53      public JPAAnnotationBuilder(int numIndents) {
54          super(numIndents);
55      }
56      
57      @Override
58      void append(Class<? extends Annotation> annotation) {
59          builderString.append('@');
60          
61        // javax.persistence annotations will have an import (see ForceJPAObject.st)
62        // so these do not need to be fully qualified
63        if (annotation.getPackage().getName().equals("javax.persistence")) {
64            builderString.append(annotation.getSimpleName());
65        } else {
66            builderString.append(annotation.getName());
67        }
68      }
69      
70      /**
71       * Adds an annotation with attributes to this {@code JPAAnnotationBuilder}.
72       * <p>
73       * Note that the attributes map has {@code String} values.  This method
74       * will exactly copy the attribute map values into the annotation attribute
75       * list.  Therefore callers must exactly write the attribute values as they
76       * appear in {@code String} form.  For example, to add the attribute name="FooBar"
77       * the caller would do the following:
78       * <p>
79       *   {@code map.put("name", "\"FooBar\"")}
80       * <p>
81       * Annotations with attributes will be built in the form:
82       * <p>
83       *   {@code @<annotationName>(<attr1>=<attr1Value>,<attr2>=<attr2Value>,...)}
84       * 
85       * @param annotation a subclass of {@code Annotation} that represents the annotation
86       *                   that is to be added to this builder
87       * @param attrs a {@code java.util.Map} of attribute keys and values in {@code String} form 
88       */
89      public void add(Class<? extends Annotation> annotation, Map<String, String> attrs) {
90          add(annotation);
91          
92          // Add all of the annotation's attributes in key=value pairs
93          if (attrs != null && !attrs.isEmpty()) {
94              builderString.append('(');
95              for (Map.Entry<String, String> attr : attrs.entrySet()) {
96                  builderString.append(attr.getKey()).append('=').append(attr.getValue()).append(',');
97              }
98              
99              // Get rid of the last comma (',')
100             builderString.deleteCharAt(builderString.lastIndexOf(","));
101             builderString.append(')');
102         }
103     }
