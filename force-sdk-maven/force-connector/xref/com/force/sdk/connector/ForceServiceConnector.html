53  /**
54   * A connector to the Force.com service.
55   * <p>
56   * This connector will manage and provide connections to the Force.com service. It can
57   * provide the following connection types:
58   * <p>
59   * <ul>
60   *   <li>PartnerConnection - Used for SOAP API calls</li>
61   *   <li>MetadataConnection - Used for Force.com's Metadata API</li>
62   *   <li>BulkConnection - Used for Force.com's Bulk API</li>
63   * </ul>
64   * Connectors require a {@link ForceConnectorConfig} to provide connection
65   * properties.  They can handle {@code ForceConnectorConfig}s from the following
66   * sources (Note that sources will be checked in this order):
67   * <p>
68   * <ol>
69   *   <li>An injected {@code ForceConnectorConfig}</li>
70   *   <li>A named source (see {@link ForceConnectorUtils#loadConnectorPropsFromName(String)})</li>
71   *   <li>A {@code ForceConnectorConfig} stored in the connector's {@code ThreadLocal} cache</li>
72   * </ol>
73   * Connectors are capable of caching {@code ForceConnectorConfig} objects both in a {@code ThreadLocalCache}
74   * and an in memory cache.  The {@code ThreadLocalCache} is directly controlled by the caller.  The in memory
75   * cache is controlled internally by the connector.  By default, a connector will cache a {@code ForceConnectorConfig}
76   * in memory whenever it constructs a connection.  However, this can be turned off by the caller.
77   * <p>
78   * A {@code ForceServiceConnector} provides several additional features beyond the traditional
79   * Force.com API {@code Connector}.  Namely:
80   * <p>
81   * <ul>
82   *   <li>
83   *   Force.com connection endpoint building with automatic version binding 
84   *   (see {@link ForceConnectorConfig#setAuthEndpoint(String)} and {@link ForceConnectorUtils#buildForceApiEndpoint(String)})
85   *   </li>
86   *   <li>Automatic session renewal</li>
87   *   <li>Convenient setting of Force.com connection client identifiers (clientId)</li>
88   * </ul>
89   *
90   * @author Tim Kral
91   * @author Fiaz Hossain
92   */
93  public class ForceServiceConnector implements ForceConnector, SessionRenewer {

232     private void initConnection() throws ConnectionException {
233         if (config == null) {
234             config = getConfig();
235             initConfig();
236         }
237         
238         this.connection = Connector.newConnection(config);
239 
240         CallOptions_element co = new CallOptions_element();
241         
242         // Give the connection a client id if we have one
243         if (this.clientId != null) {
244             co.setClient(this.clientId);
245             
246         // Check for any external client id we might
247         // have come across (see getConfig)
248         } else if (this.externalClientId != null) {
249             co.setClient(this.externalClientId);
250         }
251         
252         this.connection.__setCallOptions(co);
253     }
254     
255     // Constructs and returns config.
256     private ForceConnectorConfig getConfig() throws ConnectionException {
257         
258         // First, look for construction state in this object's state.
259         if (externalConfig != null) {
260             // Save the client id for possible later use (see initConnection)
261             externalClientId = externalConfig.getClientId();
262             
263             // Clone the config in case it gets cached.  This will prevent
264             // the caller from modifying the cached object.
265             return checkConfigCache((ForceConnectorConfig) externalConfig.clone());
266         }
267         
268         // Next, try to retrieve saved connection construction state using the connection name
269         if (connectionName != null) {
270             ForceConnectorConfig loadedConfig;
271             try {
272                 loadedConfig = ForceConnectorConfig.loadFromName(connectionName);
273             } catch (IOException e) {
274                 throw new ConnectionException("Unable to load ForceConnectorConfig for name " + connectionName, e);
275             }
276             
277             if (loadedConfig != null) {
278                 // Save the client id for possible later use (see initConnection)
279                 externalClientId = loadedConfig.getClientId();
280                 return checkConfigCache(loadedConfig);
281             }
282         }
283         
284         // Finally, try to get config from ThreadLocal
285         if ((config = ForceThreadLocalStore.getConnectorConfig()) != null) return config;
286         
287         // Out of options.  There's not enough here to construct config.
288         StringBuffer errorMsg = new StringBuffer();
289         errorMsg.append("No state was found to construct a connection.")
290                 .append(" Please provide a ForceConnectorConfig.");
291         
292         if (connectionName != null) {
293             errorMsg.append(" Or create a classpath properties file, environment variable or java property for the name '")
294                     .append(connectionName).append("'");
295         }
296 
297         throw new ConnectionException(errorMsg.toString());
298     }

300     private ForceConnectorConfig checkConfigCache(ForceConnectorConfig configToCheck) throws ConnectionException {
301         validateConnectorConfig(configToCheck);
302         
303         String cacheId = configToCheck.getCacheId();
304         if (cacheId != null && !skipCache) {
305             LOGGER.trace("ForceServiceConnector Cache: Checking for id: " + cacheId);
306             
307             ForceConnectorConfig cachedConfig;
308             if ((cachedConfig = CACHED_CONFIGS.get(cacheId)) != null) {
309                 LOGGER.trace("ForceServiceConnector Cache: HIT for id: " + cacheId);
310                 return cachedConfig;
311             }
312 
313             LOGGER.trace("ForceServiceConnector Cache: MISS for id: " + cacheId);
314             CACHED_CONFIGS.put(cacheId, configToCheck);
315         }
316         
317         return configToCheck;
318     }

320     private void validateConnectorConfig(ForceConnectorConfig configToValidate) throws ConnectionException {
321         if (configToValidate.getSessionId() == null && configToValidate.getAuthEndpoint() == null) {
322             throw new ConnectionException("ForceConnectorConfig must have an AuthEndpoint");
323         }
324         
325         if (configToValidate.getSessionId() == null && configToValidate.getUsername() == null) {
326             throw new ConnectionException("ForceConnectorConfig must have a Username");
327         }
328     }

330     // Initiates config.  We assume that config is non-null.
331     private void initConfig() {
332         if (config.getSessionRenewer() == null) {
333             config.setSessionRenewer(this);
334         }
335         
336         // Add proxy information to config
337         if (DEFAULT_PROXY != null && (config.getProxy() != Proxy.NO_PROXY || config.getProxy().type() == Type.DIRECT)) {
338             config.setProxy(DEFAULT_PROXY);
339         }
340         
341         // Transfer any extra state we have to config
342         if (this.timeout > 0) config.setReadTimeout(this.timeout);
343     }

369     private void initMetadataConnection() throws ConnectionException {
370         if (this.connection == null) {
371             initConnection();
372         }
373         
374         ConnectorConfig configNew = new ConnectorConfig();
375         configNew.setSessionId(config.getSessionId());
376         configNew.setServiceEndpoint(METADATA_URI_PATTERN.matcher(config.getServiceEndpoint()).replaceFirst("$1/m/$2"));
377         this.metadataConnection = new MetadataConnection(configNew);
378     
379         // Give the metadata connection a client id if we have one
380         if (this.clientId != null) {
381             this.metadataConnection.setCallOptions(this.clientId);
382             
383         // If we've constructed config from ForceServiceConnectionInfo state
384         // then we've saved any client id for use here (see getConfig)
385         } else if (this.externalClientId != null) {
386             this.metadataConnection.setCallOptions(this.externalClientId);
387         }
388     }

417         if (this.connection == null) {

421         ConnectorConfig configNew = new ConnectorConfig();

429      * Returns the namespace of the Force.com store to which this {@code ForceServiceConnector}

431      * <p>
432      * A namespace uniquely identifies a Force.com store on the Force.com service.  This {@code ForceServiceConnector}

434      * 
435      * @return the unique Force.com store namespace; {@code null} is no such namespace exists

437      *                             or otherwise constructed
438      * @throws ConnectionException if this {@code ForceServiceConnector} cannot get a connection
439      *                             to the Force.com service with its {@code ForceConnectorConfig} state

442     public String getNamespace() throws ConnectionException {
443         if (metadataConnection == null) {
444             initMetadataConnection();
445         }
446         if (!config.isNamespaceInitialized()) {
447             DescribeMetadataResult result = metadataConnection.describeMetadata(DESCRIBE_METADATA_VERSION);
448             String organizationNamespace = result.getOrganizationNamespace();
449             config.setNamespace(organizationNamespace != null && organizationNamespace.length() > 0
450                     ? organizationNamespace : null);
451         }
452         return config.getNamespace();
453     }

495     /**
496      * Automatically renews Force.com timed out sessions.
497      * <p>
498      * The connections gotten by a {@code ForceServiceConnector} are lazily
499      * constructed and stored in the {@code ForceServiceConnector}'s local state.  Thus
500      * multiple calls to get a connection will return the same connection object without
501      * re-establishing a connection to the Force.com service.  However, this presents
502      * a problem should the Force.com session expire.  In that case, stored connections
503      * will have their session automatically renewed by the {@code renewSession}.
504      * 
505      * @param connectorConfig the {@code ForceConnectorConfig} to be used to re-establish a
506      *               Force.com connection
507      * @return a session renewal SOAP header
508      * @throws ConnectionException if the Force.com connection cannot be re-established
509      * @see ForceServiceConnector#getConnection()
510      * @see ForceServiceConnector#getMetadataConnection()
511      * @see ForceServiceConnector#getBulkConnection()
512      * @see ForceServiceConnector#close()
513      */
514     @Override
515     public SessionRenewalHeader renewSession(ConnectorConfig connectorConfig) throws ConnectionException {
516         if (connectorConfig.getPassword() != null) {
517             connectorConfig.setSessionId(null);
518             
519             close();
520             setConnectorConfig((ForceConnectorConfig) connectorConfig);
521             getConnection();
522 
523             SessionRenewalHeader ret = new SessionRenewalHeader();
524             ret.name = SESSION_HEADER_QNAME;
525             SessionHeader_element se = new SessionHeader_element();
526             se.setSessionId(connectorConfig.getSessionId());
527             ret.headerElement = se;
528             return ret;
529         }
530         return null;
531     }

597     /**
598      * Indicates whether or not this {@code ForceServiceConnector} should skip
599      * in memory {@code ForceConnectorConfig} cache reads and writes.
600      * <p>
601      * By default, a {@code ForceServiceConnector} will cache {@code ForceConnectorConfig}
