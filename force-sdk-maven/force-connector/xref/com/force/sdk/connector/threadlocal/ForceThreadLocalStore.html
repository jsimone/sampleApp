27  package com.force.sdk.connector.threadlocal;
28  
29  import java.util.HashMap;
30  
31  import com.force.sdk.connector.ForceConnectorConfig;
32  
33  /**
34   * Class for storing connector configs in a {@code ThreadLocal} cache.
35   *
36   * @author John Simone
37   */
38  public final class ForceThreadLocalStore {
39  
40      // Thread Local for saved configs
41      private static final ThreadLocal<ForceConnectorConfig> TL_CONNECTOR_CONFIG = new ThreadLocal<ForceConnectorConfig>();
42      // Thread Local map for other classes
43      private static final HashMap<Class<?>, ThreadLocal<?>> TL_MAP = new HashMap<Class<?>, ThreadLocal<?>>();
44      
45      private ForceThreadLocalStore() {  }
46      
47      /**
48       * Retrieves the {@code ForceConnectorConfig} from the
49       * {@code ForceThreadLocalStore} {@code ThreadLocal} cache.
50       * 
51       * @return the {@code ForceConnectorConfig} found in the {@code ThreadLocal}
52       *         cache; {@code null} if the cache is empty
53       */
54      public static ForceConnectorConfig getConnectorConfig() {
55          return TL_CONNECTOR_CONFIG.get();
56      }
57  
58      /**
59       * Sets the given {@code ForceConnectorConfig} in the
60       * {@code ForceThreadLocalStore} {@code ThreadLocal} cache.
61       * 
62       * @param config the {@code ForceConnectorConfig} to be added
63       *               to the {@code ThreadLocal} cache
64       */
65      public static void setConnectorConfig(ForceConnectorConfig config) {
66          TL_CONNECTOR_CONFIG.set(config);
67      }
68      
69      /**
70       * Retrieve the object of type {@code clazz} from the
71       * {@code ForceThreadLocalStore} {@code ThreadLocal} cache.
72       * 
73       * @param <T> describes the object type to be searched for in the {@code ThreadLocal} cache
74       * @param clazz the object type to be searched for in the {@code ThreadLocal} cache
75       * @return the object of type {@code clazz} found in the {@code ThreadLocal}
76       *         cache; {@code null} if no such object can be found
77       */
78      @SuppressWarnings("unchecked")
79      public static <T> T getThreadLocal(Class<T> clazz) {
80          ThreadLocal<T> tl = null;
81          if (TL_MAP.containsKey(clazz)) {
82              tl = (ThreadLocal<T>) TL_MAP.get(clazz);
83          } else {
84              tl = new ThreadLocal<T>();
85              TL_MAP.put(clazz, tl);
86          }
87          
88          return tl.get();
89      }
90      
91      /**
92       * Sets an object of type {@code clazz} in the
93       * {@code ForceThreadLocalStore} {@code ThreadLocal} cache.
94       * 
95       * @param <T> describes the object type to be added to the {@code ThreadLocal} cache
96       * @param clazz the object type to be added to the {@code ThreadLocal} cache
97       * @param variable the object to be added to the {@code ThreadLocal} cache
98       */
99      @SuppressWarnings("unchecked")
100     public static <T> void setThreadLocal(Class<T> clazz, T variable) {
101         ThreadLocal<T> tl = null;
102         if (TL_MAP.containsKey(clazz)) {
103             tl = (ThreadLocal<T>) TL_MAP.get(clazz);
104         } else {
105             tl = new ThreadLocal<T>();
106             TL_MAP.put(clazz, tl);
107         }
108         tl.set(variable);
109     }
110 }
