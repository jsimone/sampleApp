29  import java.util.regex.Pattern;

31  /**
32   * Property that can be used for in a Force.com connection URL.
33   * <p>
34   * These enum values represent available Force.com connection URL
35   * properties.  Each value contains two parts: 
36   * <p>
37   * <ol>
38   *   <li>The property key name</li>
39   *   <li>A validation regular expression</li>
40   * </ol>
41   * <p>
42   * A Force.com connection URL is written using the property key name
43   * in key=value pairs which are separated by a semi-colon (';') delimiter.
44   * The exception is the Force.com endpoint property which is interpreted
45   * as the string which immediately follows the force:// protocol.
46   * For example, a Force.com connection URL with a username and password
47   * would look like this:
48   * <pre>
49   * {@code force://<endpoint value>;user=<username value>;password=<password value>}
50   * </pre>  
51   * The validation regular expression allows for the validation of
52   * property values within a Force.com connection URL.
53   * 
54   * @author Tim Kral
55   */
56  public enum ForceConnectionProperty {
57      /**
58       * The Force.com connection endpoint.
59       */
60      // The endpoint can begin with nothing, http://, or https://
61      // It can end with nothing, / or /services/Soap/u/<API version number>
62      ENDPOINT("endpoint", Pattern.compile("^(http://|https://)?([A-Za-z0-9_.\\-:]+)(/|/services/Soap/u/(\\d+|\\d+.\\d+))?$")),
63      /**
64       * The username to be used in a Force.com connection.
65       */
66      USER("user", null),
67      /**
68       * The password to be used in a Force.com connection.
69       */
70      PASSWORD("password", null),
71      /**
72       * The OAuth key to be used in a Force.com OAuth handshake.
73       */
74      OAUTH_KEY("oauth_key", null),
75      /**
76       * The OAuth secret to be used in a Force.com OAuth handshake.
77       */
78      OAUTH_SECRET("oauth_secret", Pattern.compile("(\\d+)")),
79      /**
80       * The API client id to be used in a Force.com connection.
81       */
82      CLIENTID("clientid", null),
83      /**
84       * The timeout to be used in a Force.com connection.
85       */
86      TIMEOUT("timeout", Pattern.compile("(\\d+)")),
87      ;
88      
89      private String propertyName;
90      private Pattern validationRegex;
91      
92      private ForceConnectionProperty(String propertyName, Pattern validationRegex) {
93          this.propertyName = propertyName;
94          this.validationRegex = validationRegex;
95      }
96      
97      /**
98       * Returns the connection property name.
99       * <p>
100      * The property name (with the exception of the endpoint
101      * property) is used as the key in a key=value pair within
102      * a Force.com connection URL.
103      * 
104      * @return the connection property name
105      */
106     public String getPropertyName() { return this.propertyName; }
107     
108     /**
109      * Returns the connection property validation regular expression
110      * <p>
111      * The validation regular expression is used to valid
112      * a property value that is set in a key=value pair within
113      * a Force.com connection URL.
114      * 
115      * @return the connection property validation regular expression
116      */
117     public Pattern getValidationRegex() { return this.validationRegex; }
118     
119     /**
120      * Validations a connection property value.
121      * <p>
122      * Ensures that the connection property value is not null, has a length
123      * greater than zero and matches the validation regular expression.
124      * 
125      * @param propertyValue the connection property value
126      * @throws IllegalArgumentException if the given propertyValue is null or zero length or
127      *                                  does not match the validation regular expression 
128      */
129     public void validateValue(String propertyValue) {
130         validateValue(propertyValue, "");
131     }
132     
133     /**
134      * Validations a connection property value.
135      * <p>
136      * Ensures that the connection property value is not null, has a length
137      * greater than zero and matches the validation regular expression.
138      * 
139      * @param propertyValue the connection property value
140      * @param customErrorMessage an error message to be included in the thrown exception message
141      *                           should validation fail
142      * @throws IllegalArgumentException if the given propertyValue is null or zero length or
143      *                                  does not match the validation regular expression 
144      */
145     public void validateValue(String propertyValue, String customErrorMessage) {
146         if ("endpoint".equals(this.propertyName) && propertyValue != null && propertyValue.contains("localhost")) {
147             return;
148         }
149 
150         if (propertyValue == null || propertyValue.length() == 0) {
151             StringBuffer errorMessage = new StringBuffer(customErrorMessage);
152             errorMessage.append(" The ForceConnectionProperty (" + this.propertyName + ") must have a value");
153             
154             throw new IllegalArgumentException(errorMessage.toString());
155         }
156         
157         if (this.validationRegex == null) return;
158         
159         if (!validationRegex.matcher(propertyValue).matches()) {
160             StringBuffer errorMessage = new StringBuffer(customErrorMessage);
161             errorMessage.append(" Illegal value (" + propertyValue + ") for ForceConnectionProperty (" + this.propertyName + ")");
162             
163             throw new IllegalArgumentException(errorMessage.toString());
164         }
165     }
166     
167     /**
168      * Load the ForceConnectionProperty enum value from a connection 
169      * property name.
170      * 
171      * @param propertyName A connection property name
172      * @return The ForceConnectionProperty enum value matching the given
173      *         propertyName or {@code null} if no such value exists
174      */
175     public static ForceConnectionProperty fromPropertyName(String propertyName) {
176         if (propertyName == null) return null;
177         
178         for (ForceConnectionProperty connProp : values()) {
179             if (connProp.getPropertyName().equals(propertyName.toLowerCase()))
180                 return connProp;
181         }
182         
183         return null;
184     }
