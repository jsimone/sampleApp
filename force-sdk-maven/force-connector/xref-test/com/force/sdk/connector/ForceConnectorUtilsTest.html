31  import java.net.MalformedURLException;
32  import java.net.URL;

34  import org.testng.annotations.DataProvider;
35  import org.testng.annotations.Test;

37  /**
38   * Unit Tests for ForceConnectorUtils.
39   *
40   * @author Tim Kral
41   */
42  public class ForceConnectorUtilsTest {
43      
44      @DataProvider
45      protected Object[][] endpointProvider() {
46          return new Object[][] {
47                  {"url", "https", "url", -1},
48                  {"url:80", "https", "url", 80},
49                  {"url/", "https", "url", -1},
50                  {"http://url", "https", "url", -1},
51                  {"https://url", "https", "url", -1},
52                  {"http://localhost:8080", "http", "localhost", 8080},
53                  {"https://localhost:8080", "https", "localhost", 8080},
54          };
55      }
56      
57      @Test(dataProvider = "endpointProvider")
58      public void testBuildForceApiEndpoint(String endpoint, String expectedProtocol, String expectedHost, int expectedPort)
59      throws MalformedURLException {
60          URL apiUrl = new URL(ForceConnectorUtils.buildForceApiEndpoint(endpoint));
61          assertEquals(apiUrl.getProtocol(), expectedProtocol);
62          assertEquals(apiUrl.getHost(), expectedHost);
63          assertEquals(apiUrl.getPort(), expectedPort);
64          assertEquals(apiUrl.getPath(), new URL(com.sforce.soap.partner.Connector.END_POINT).getPath());
65      }
