117     @Override
118     public BeanDefinition parse(Element element, ParserContext parserContext) {
119 
120         // Find the connection information child (either oauthInfo, connectionUrl or connectionName)
121         NodeList children = element.getElementsByTagNameNS(element.getNamespaceURI(), "*");
122 
123         // validate that exactly one connection info element is defined
124         validateConfiguration(children);
125 
126         parserContext.pushContainingComponent(new CompositeComponentDefinition(element.getTagName(), parserContext
127                 .extractSource(element)));
128 
129         // Create OAuthConnector bean first
130         BeanDefinition oauthConnector = createOAuthConnector((Element) getConnectionNode(children), parserContext);
131         // If there is a custom data retriever defined then create the necessary beans and
132         // and set onto the OAuthConnector
133         Element customDataRetrieverElement = (Element) getCustomDataRetrieverNode(children);
134         if (customDataRetrieverElement != null) {
135             parserContext.getRegistry().registerBeanDefinition(USER_DATA_RETRIEVAL_SERVICE_BEAN_NAME,
136                     createCustomUserDataRetrievalService(customDataRetrieverElement, element));
137             // set service onto OAuthConnector
138             oauthConnector.getPropertyValues().add("userDataRetrievalService",
139                     new RuntimeBeanReference(USER_DATA_RETRIEVAL_SERVICE_BEAN_NAME));
140         } else {
141             parserContext.getRegistry().registerBeanDefinition(USER_DATA_RETRIEVAL_SERVICE_BEAN_NAME,
142                     createUserDataRetrievalService(element));
143         }
144         parserContext.getRegistry().registerBeanDefinition(OAUTH_CONNECTOR_BEAN_NAME, oauthConnector);
145 
146         // Create OAuthAuthenticationProvider next
147         parserContext.getRegistry().registerBeanDefinition(AUTH_PROVIDER_BEAN_NAME,
148                 createOAuthAuthenticationProvider(parserContext));
149 
150         // Create AuthenticationManager next
151         String[] knownAlias = parserContext.getRegistry().getAliases(AUTH_MANAGER_BEAN_ALIAS);
152         if (knownAlias == null || knownAlias.length == 0) {
153             parserContext.getRegistry().registerBeanDefinition(BeanIds.AUTHENTICATION_MANAGER,
154                     createAuthenticationManager(element, parserContext));
155         }
156 
157         // Create AuthenticationProcessingFilterEntryPoint next
158         parserContext.getRegistry().registerBeanDefinition(AUTH_PROCESSING_ENTRY_POINT_BEAN_NAME,
159                 createAuthenticationProcessingEntryPoint(parserContext));
160 
161         // Create SecurityContextStorage bean
162         // this handles the storage of the security information into either a cookie or session
163         // Create AuthenticationProcessingFilterEntryPoint next
164         parserContext.getRegistry().registerBeanDefinition(CONTEXT_STORAGE_SERVICE_NAME,
165                 createSecurityContextStorageService(element));
166 
167         // Create SecurityContextService bean
168         // this handles management of security information, it is a service facade to the userDataRetrieval
169         // and securityContextStorage services
170         parserContext.getRegistry().registerBeanDefinition(CONTEXT_SERVICE_NAME, createSecurityContextService());
171 
172         // Create RememberMeServices bean
173         // This implements the remember me functionality which will handle the population of security
174         // information from a browser cookie or server side session
175         parserContext.getRegistry().registerBeanDefinition(REMEMBER_ME_SERVICES_BEAN_NAME, createRememberMeServices());
176 
177         // Create AuthenticationProcessingFilter
178         // This filter is invoked after OAuth login to exchange token for an access token
179         parserContext.getRegistry().registerBeanDefinition(AUTH_PROCESSING_FILTER_BEAN_NAME,
180                 createAuthenticationProcessingFilter(element, parserContext));
181 
182         // This filter is invoked during logout
183         // sets this as the authentication filter over the default
184         parserContext.getRegistry().registerBeanDefinition(AUTH_LOGOUT_FILTER_BEAN_NAME, createLogoutFilter(element));
185 
186         // This filter is invoked during the login process to process remember me tokens which is
187         // how the cookie based security information is stored
188         parserContext.getRegistry().registerBeanDefinition(REMEMBER_ME_FILTER_BEAN_NAME, createRememberMeFilter());
189 
190         // This filter is invoked after the remember me filter. It insures that the thread local connection variables
191         // are set.
192         parserContext.getRegistry().registerBeanDefinition(CONNECTION_STORAGE_FILTER_BEAN_NAME,
193                 createConnectionStorageFilter());
194 
195         // Now we configure http element if it's present. HTTP element should be a sibling so look for it.
196         if (element.getParentNode() != null) {
197             NodeList nodes = element.getParentNode().getChildNodes();
198             for (int i = 0; i < nodes.getLength(); i++) {
199                 Node node = nodes.item(i);
200                 if (node.getLocalName() != null && "http".equals(node.getLocalName().toLowerCase())) {
201                     if (node.getAttributes().getNamedItem(ATT_ENTRY_POINT_REF) == null) {
202                         // Add entry-point-ref since it's missing
203                         addNodeAttribute(node, ATT_ENTRY_POINT_REF, AUTH_PROCESSING_ENTRY_POINT_BEAN_NAME);
204                     }
205                     if (node.getAttributes().getNamedItem(CREATE_SESSION) == null) {
206                         // Add create-session="never" attribute
207                         addNodeAttribute(node, CREATE_SESSION, "never");
208                     }
209 
210                     /**
211                      * Sets this as the authentication filter over the default <security:custom-filter
212                      * position="FORM_LOGIN_FILTER" ref="authenticationFilter"/> <security:custom-filter
213                      * position="LOGOUT_FILTER" ref="logoutFilter"/>
214                      **/
215                     setCustomFilterIfMissing(node, "FORM_LOGIN_FILTER", AUTH_PROCESSING_FILTER_BEAN_NAME, ATT_POSITION);
216                     setCustomFilterIfMissing(node, "LOGOUT_FILTER", AUTH_LOGOUT_FILTER_BEAN_NAME, ATT_POSITION);
217                     setCustomFilterIfMissing(node, "REMEMBER_ME_FILTER", REMEMBER_ME_FILTER_BEAN_NAME, ATT_POSITION);
218                     setCustomFilterIfMissing(node, "REMEMBER_ME_FILTER", CONNECTION_STORAGE_FILTER_BEAN_NAME, ATT_AFTER);
219                     break;
220                 }
221             }
222         }
223         return null;
224     }

226     private void addNodeAttribute(Node node, String name, String value) {
227         Attr newNode = node.getOwnerDocument().createAttributeNS(node.getNamespaceURI(), name);
228         newNode.setNodeValue(value);
229         node.getAttributes().setNamedItem(newNode);
230     }

232     private void setCustomFilterIfMissing(Node httpNode, String positionUpper, String ref, String attribute) {
233         if (httpNode.hasChildNodes()) {
234             NodeList children = httpNode.getChildNodes();
235             for (int i = 0; i < children.getLength(); i++) {
236                 Node node = children.item(i);
237                 Node nn;
238                 if (node.getLocalName() != null && ELEM_CUSTOM_FILTER.equals(node.getLocalName().toLowerCase())
239                         && (nn = node.getAttributes().getNamedItem(attribute)) != null && nn.getNodeValue() != null
240                         && positionUpper.equals(nn.getNodeValue().toUpperCase())) {
241                     // The filter already exists
242                     return;
243                 }
244             }
245         }
246 
247         // Add custom-filter since it's missing
248         int prefix = httpNode.getNodeName().indexOf(':');
249         Node newNode = httpNode.getOwnerDocument().createElementNS(httpNode.getNamespaceURI(),
250                 prefix > 0 ? httpNode.getNodeName().substring(0, prefix + 1) + ELEM_CUSTOM_FILTER : ELEM_CUSTOM_FILTER);
251         httpNode.appendChild(newNode);
252         addNodeAttribute(newNode, attribute, positionUpper);
253         addNodeAttribute(newNode, ATT_REF, ref);
254     }

256     private BeanDefinition createOAuthConnector(Element connectionInfo, ParserContext parser) {
257         // Build a ForceOAuthConnectionInfo bean, if applicable
258         BeanDefinition oauthConnInfo = null;
259         if (NAME_OAUTH_INFO_ELEMENT.equals(connectionInfo.getLocalName())) {
260             // Create a new ForceOAuthConnectionInfo bean and populate it with OAuth Info
261             oauthConnInfo = new RootBeanDefinition(ForceOAuthConnectionInfo.class);
262             oauthConnInfo.getPropertyValues().add("endpoint", connectionInfo.getAttribute(ENDPOINT_ATTR));
263             oauthConnInfo.getPropertyValues().add("oauthKey", connectionInfo.getAttribute(OAUTH_KEY_ATTR));
264             oauthConnInfo.getPropertyValues().add("oauthSecret", connectionInfo.getAttribute(OAUTH_SECRET_ATTR));
265         } else if (NAME_CONN_URL_ELEMENT.equals(connectionInfo.getLocalName())) {
266             // Create a new ForceOAuthConnectionInfo bean and populate it with Connection Url
267             oauthConnInfo = new RootBeanDefinition(ForceOAuthConnectionInfo.class);
268             oauthConnInfo.getPropertyValues().add("connectionUrl", connectionInfo.getAttribute(CONNECTION_URL_ATTR));
269         }
270 
271         BeanDefinition oauthConnector = new RootBeanDefinition(ForceOAuthConnector.class);
272 
273         if (oauthConnInfo != null) {
274             // Register ForceOAuthConnectionInfo bean and give it to the connector
275             parser.getRegistry().registerBeanDefinition(OAUTH_CONNECTION_INFO_BEAN_NAME, oauthConnInfo);
276             oauthConnector.getPropertyValues().add("connectionInfo", oauthConnInfo);
277         } else if (NAME_CONN_NAME_ELEMENT.equals(connectionInfo.getLocalName())) {
278             oauthConnector.getPropertyValues().add("connectionName", connectionInfo.getAttribute(CONNECTION_NAME_ATTR));
279         } else {
280             throw new RuntimeException("Unrecognized oauth connection information child element: "
281                     + connectionInfo.getLocalName());
282         }
283 
284         return oauthConnector;
285     }

323     private BeanDefinition createAuthenticationManager(Element element, ParserContext parserContext) {
324         BeanDefinition authManager = new RootBeanDefinition(ProviderManager.class);
325         authManager.getPropertyValues().add("parent", new RootBeanDefinition(AuthenticationManagerFactoryBean.class));
326         List<BeanMetadataElement> providers = new ManagedList<BeanMetadataElement>();
327         providers.add(new RuntimeBeanReference(AUTH_PROVIDER_BEAN_NAME));
328         authManager.getPropertyValues().add("providers", providers);
329 
330         // Add the default event publisher
331         /*
332          * BeanDefinition publisher = new RootBeanDefinition(DefaultAuthenticationEventPublisher.class); String id =
333          * parserContext.getReaderContext().generateBeanName(publisher); parserContext.registerBeanComponent(new
334          * BeanComponentDefinition(publisher, id)); authManager.getPropertyValues().add("authenticationEventPublisher",
335          * publisher);
336          */
337 
338         parserContext.registerBeanComponent(new BeanComponentDefinition(authManager, AUTH_MANAGER_BEAN_NAME));
339 
340         // Add the alias
341         parserContext.getRegistry().registerAlias(AUTH_MANAGER_BEAN_NAME, AUTH_MANAGER_BEAN_ALIAS);
342         parserContext.getReaderContext().fireAliasRegistered(AUTH_MANAGER_BEAN_NAME, AUTH_MANAGER_BEAN_ALIAS,
343                 parserContext.extractSource(element));
344 
345         return authManager;
346     }

448     private void validateConfiguration(NodeList children) {
449         if (children.getLength() != 1 && children.getLength() != 2) { throw new RuntimeException(
450                 "<oauth> must specify exactly one of: <oauthInfo>, <connectionUrl> or <connectionName>");
451         }
452 
453         int conectionElementCount = 0;
454 
455         for (int i = 0; i < children.getLength(); i++) {
456             String name = children.item(i).getLocalName();
457 
458             if (isConnectionElementName(name)) {
459                 conectionElementCount++;
460             }
461         }
462 
463         if (conectionElementCount != 1) { throw new RuntimeException(
464                 "<oauth> must specify exactly one of: <oauthInfo>, <connectionUrl> or <connectionName>");
465         }
466     }
