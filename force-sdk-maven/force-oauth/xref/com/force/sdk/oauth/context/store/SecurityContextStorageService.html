27  package com.force.sdk.oauth.context.store;
28  
29  import javax.crypto.spec.SecretKeySpec;
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  
33  import com.force.sdk.oauth.context.SecurityContext;
34  
35  /**
36   * 
37   * Defines the interface for storing, retrieving, and clearing a SecurityContext to/from 
38   * a means of storage. Implementors of this interface will represent different methods of 
39   * of handling the storage of a SecurityContext.
40   *
41   * @author John Simone
42   */
43  public interface SecurityContextStorageService {
44  
45      /**
46       * Store the security context. The means of storage will vary by implementation.
47       * 
48       * @param request HttpServletRequest
49       * @param response HttpServletResponse
50       * @param securityContext SecurityContext
51       * @throws ContextStoreException {@link ContextStoreException}
52       */
53      void storeSecurityContext(
54              HttpServletRequest request, HttpServletResponse response, SecurityContext securityContext)
55              throws ContextStoreException;
56      
57      /**
58       * Retrieve the security context. The means of storage will vary by implementation.
59       * 
60       * @param request HttpServletRequest
61       * @return the stored SecurityContext
62       * @throws ContextStoreException {@link ContextStoreException}
63       */
64      SecurityContext retreiveSecurityContext(HttpServletRequest request) throws ContextStoreException;
65      
66      /**
67       * Clears the security context from storage. This won't be relevant for all storage types.
68       * 
69       * @param request HttpServletRequest
70       * @param response HttpServletResponse
71       */
72      void clearSecurityContext(HttpServletRequest request, HttpServletResponse response);
73      
74      /**
75       * Retrieves the secret key if one is used in this security context store.
76       * @return SecretKeySpec
77       * @throws ForceEncryptionException {@link ForceEncryptionException}
78       */
79      SecretKeySpec getSecureKey() throws ForceEncryptionException;
80  }
