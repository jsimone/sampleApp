41  /**
42   * 
43   * Utility for handling encryption logic with AES. 
44   *
45   * @author John Simone
46   */
47  public final class AESUtil {
48      static final String ALGORITHM  = "AES";
49      static final String PRIVATE_KEY_PROPERTY = "private-key";
50      static final String PRIVATE_KEY_PATH = "encryption.properties";
51      private static final Logger LOGGER = LoggerFactory.getLogger(AESUtil.class);
52      private static final int KEY_SIZE = 128;
53      
54      private AESUtil() {  }
55      
56      /**
57       * Encrypt the value using the passed in key.
58       * 
59       * @param value data to encrypt
60       * @param skeySpec encryption key
61       * @return The encrypted data
62       * @throws ForceEncryptionException {@link ForceEncryptionException} 
63       */
64      public static byte[] encrypt(byte[] value, SecretKeySpec skeySpec) throws ForceEncryptionException {
65          try {
66              Cipher cipher = null;
67              cipher = Cipher.getInstance(ALGORITHM);
68              cipher.init(Cipher.ENCRYPT_MODE, skeySpec);
69              byte[] encrypted = null;
70              encrypted = cipher.doFinal(value);
71              return encrypted;
72          } catch (BadPaddingException e) {
73              throw new ForceEncryptionException(e);
74          } catch (InvalidKeyException e) {
75              throw new ForceEncryptionException(e);
76          } catch (NoSuchAlgorithmException e) {
77              throw new ForceEncryptionException(e);
78          } catch (NoSuchPaddingException e) {
79              throw new ForceEncryptionException(e);
80          } catch (IllegalBlockSizeException e) {
81              throw new ForceEncryptionException(e);
82          }
83      }
84      
85      /**
86       * Decrypt the value using the passed in key.
87       * 
88       * @param value data to decrypt
89       * @param skeySpec The encryption key
90       * @return the decrypted value
91       * @throws ForceEncryptionException {@link ForceEncryptionException}
92       */
93      public static byte[] decrypt(byte[] value, SecretKeySpec skeySpec) throws ForceEncryptionException {
94          
95          try {
96              Cipher cipher = null;
97              cipher = Cipher.getInstance(ALGORITHM);
98              cipher.init(Cipher.DECRYPT_MODE, skeySpec);
99              byte[] original = null;
100             original = cipher.doFinal(value);
101             return original;
102         } catch (BadPaddingException e) {
103             throw new ForceEncryptionException(e);
104         } catch (InvalidKeyException e) {
105             throw new ForceEncryptionException(e);
106         } catch (NoSuchAlgorithmException e) {
107             throw new ForceEncryptionException(e);
108         } catch (NoSuchPaddingException e) {
109             throw new ForceEncryptionException(e);
110         } catch (IllegalBlockSizeException e) {
111             throw new ForceEncryptionException(e);
112         }
113         
114     }
115     
116     /**
117      * Create a SecretKeySpec object from an AES key. This String could come from a 
118      * properties file or other means of persistent configuration.
119      * 
120      * @param key String
121      * @return encryption key
122      * @throws ForceEncryptionException {@link ForceEncryptionException}
123      */
124     public static SecretKeySpec recreateSecretKeySpec(String key) throws ForceEncryptionException {
125         SecretKeySpec secretKey = null;
126         if (key != null && key.trim().length() > 0) {
127             secretKey = new SecretKeySpec(SecurityContextCookieStore.b64decode(key.trim()), ALGORITHM);
128         }
129         
130         if (secretKey == null) {
131             secretKey = AESUtil.generateRandomKey();
132         }
133         
134         return secretKey;
135     }
136     
137     /**
138      * call getSecretKey(String fileName) with the default filename.
139      * 
140      * @return encryption key
141      * @throws ForceEncryptionException {@link ForceEncryptionException}
142      */
143     public static SecretKeySpec getSecretKey() throws ForceEncryptionException {
144         return getSecretKey(PRIVATE_KEY_PATH);
145     }
146     
147     /**
148      * Read in a stored secret key from a properties file and create a SecretKeySpec object from it.
149      * 
150      * @param fileName String
151      * @return encryption key
152      * @throws ForceEncryptionException {@link ForceEncryptionException}
153      */
154     public static SecretKeySpec getSecretKey(String fileName) throws ForceEncryptionException {
155         if (fileName == null || fileName.length() == 0) {
156             fileName = PRIVATE_KEY_PATH;
157         }
158         
159         InputStream is = AESUtil.class.getResourceAsStream("/" + fileName);
160         String key = null;
161         if (is == null) {
162             LOGGER.warn("Could not open file at  path " + fileName + ". Generating private key... ");
163         } else {
164             Properties encryptionProps = new Properties();
165             try {
166                 encryptionProps.load(is);
167             }  catch (IOException e) {
168                 LOGGER.warn("Could not open file at  path " + fileName + ". Generating private key... ");
169             } finally {
170                 if (is != null) {
171                     try {
172                         is.close();
173                     } catch (IOException ignored) {
174                         // Exception ignored
175                     }
176                 }
177             }
178             
179             key = encryptionProps.getProperty(PRIVATE_KEY_PROPERTY);
180             if (key == null ||  key.trim().length() == 0) {
181                 LOGGER.warn(PRIVATE_KEY_PROPERTY + " property was null in file " + fileName + ". Generating private key... ");
182             }
183         }
184         
185         SecretKeySpec secretKey;
186          secretKey = AESUtil.recreateSecretKeySpec(key);
187         
188         return secretKey;
189     }
190     
191     /**
192      * Generate a random secret key.
193      * 
194      * @return encryption key
195      * @throws ForceEncryptionException {@link ForceEncryptionException}
196      */
197     public static SecretKeySpec generateRandomKey() throws ForceEncryptionException {
198         KeyGenerator keyGen;
199         try {
200             keyGen = KeyGenerator.getInstance(ALGORITHM);
201         } catch (NoSuchAlgorithmException e) {
202             throw new ForceEncryptionException(e);
203         }
204         
205         keyGen.init(KEY_SIZE);
206         return new SecretKeySpec(keyGen.generateKey().getEncoded(), ALGORITHM);
207     }
