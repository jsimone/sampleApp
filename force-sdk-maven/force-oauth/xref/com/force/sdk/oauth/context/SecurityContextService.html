27  package com.force.sdk.oauth.context;
28  
29  import javax.crypto.spec.SecretKeySpec;
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  
33  import com.force.sdk.oauth.context.store.ForceEncryptionException;
34  
35  /**
36   * 
37   * The security context service provides the basic functionality for managing a security context
38   * in the session of the authenticated user.
39   *
40   * @author John Simone
41   */
42  public interface SecurityContextService {
43      
44      /**
45       * Sets the security context to the session.
46       * 
47       * @param request HttpServletRequest
48       * @param response HttpServletResponse
49       * @param sc SecurityContext
50       */
51      void setSecurityContextToSession(HttpServletRequest request, HttpServletResponse response, SecurityContext sc);
52      
53      /**
54       * This method retrieves the security context. The security context
55       * will either come out of the session or will be built from a call
56       * to the partner api. The partner api will be called if:
57       * - There is no security context in the session, but there is 
58       * a session id available in a cookie
59       * - There is a security context in the session, but the session id
60       * that it contains doesn't match the one available in the cookie
61       * 
62       * @param request HttpServletRequest
63       * @return SecurityContext
64       */
65      SecurityContext getSecurityContextFromSession(HttpServletRequest request);
66      
67      /**
68       * Verify the passed in security context against the browser cookies. This will
69       * make sure that the necessary cookies exist and that the values match those
70       * in the security context. It will create a fresh security context with data 
71       * from the partner API if necessary.
72       * 
73       * @param sc SecurityContext
74       * @param request HttpServletRequest
75       * @return SecurityContext
76       */
77      SecurityContext verifyAndRefreshSecurityContext(SecurityContext sc, HttpServletRequest request);
78      
79      /**
80       * Clears the security context from the context store. 
81       * 
82       * @param request HttpServletRequest
83       * @param response HttpServletResponse
84       */
85      void clearSecurityContext(HttpServletRequest request, HttpServletResponse response);
86      
87      /**
88       * Return the secret key if one is being used.
89       * 
90       * @return SecretKeySpec
91       * @throws ForceEncryptionException {@link ForceEncryptionException}
92       */
93      SecretKeySpec getSecretKey() throws ForceEncryptionException;
94  
95  }
