42  /**
43   * 
44   * The security context service provides the basic functionality for managing a security context
45   * in the session of the authenticated user.
46   * 
47   * The implementation provided here is customizable because it delegates user data retrieval to a UserDataRetrievalService
48   * and the security context storage to a SecurityContextStorageService. The implementation used for each of these can vary
49   * depending on if a custom user data retriever is being used and whether server side session or browser cookie based
50   * security context storage is used.
51   *
52   * @author John Simone
53   */
54  public class SecurityContextServiceImpl implements SecurityContextService {
55  
56      private UserDataRetrievalService userDataRetrievalService = null;
57      private SecurityContextStorageService securityContextStorageService = null;
58      private static final Logger LOGGER = LoggerFactory.getLogger(SecurityContextServiceImpl.class);
59  
60      public void setUserDataRetrievalService(UserDataRetrievalService userDataRetrievalService) {
61          this.userDataRetrievalService = userDataRetrievalService;
62      }
63  
64      public void setSecurityContextStorageService(SecurityContextStorageService securityContextStorageService) {
65          this.securityContextStorageService = securityContextStorageService;
66      }
67      
68      /**
69       * Use the SecurityContextStorageService to store the security context.
70       * {@inheritDoc}
71       */
72      @Override
73      public void setSecurityContextToSession(HttpServletRequest request, HttpServletResponse response, SecurityContext sc) {
74          try {
75              securityContextStorageService.storeSecurityContext(request, response, sc);
76          } catch (ContextStoreException e) {
77              LOGGER.error("Cannot store security information: ", e);
78          }
79          //set cookies with sid and endpoint regardless of the securityContextStorageService used
80          //cookies should be secure if hose is anything other than localhost
81          boolean secure = !"localhost".equalsIgnoreCase(request.getLocalName());
82          SecurityContextUtil.setCookieValues(sc, response, secure);
83      }
84      
85      /**
86       * This method retrieves the security context. The security context
87       * will either come out of the session or will be built from a call
88       * to the partner api. The partner api will be called if:
89       * - There is no security context in the session, but there is 
90       * a session id available in a cookie
91       * - There is a security context in the session, but the session id
92       * that it contains doesn't match the one available in the cookie
93       * {@inheritDoc}
94       */
95      @Override
96      public SecurityContext getSecurityContextFromSession(HttpServletRequest request) {
97          //check for a security context
98          SecurityContext sc = null;
99          try {
100             sc = securityContextStorageService.retreiveSecurityContext(request);
101         } catch (ContextStoreException e) {
102             LOGGER.error("Cannot retrieve security infromation: ", e);
103             sc = null;
104         }
105         
106         sc = verifyAndRefreshSecurityContext(sc, request);
107         
108         return sc;
109     }
110     
111     /**
112      * Verify the passed in security context against the browser cookies. This will
113      * make sure that the necessary cookies exist and that the values match those
114      * in the security context. It will create a fresh security context with data 
115      * from the partner API if necessary.
116      * {@inheritDoc}
117      */
118     @Override
119     public SecurityContext verifyAndRefreshSecurityContext(SecurityContext sc, HttpServletRequest request) {
120         String sessionId = null;
121         String endpoint = null;
122         
123         //get the session id and endpoint from cookies
124         Map<String, String> cookieValueMap = SecurityContextUtil.getCookieValues(request);
125         sessionId = cookieValueMap.get(SecurityContextUtil.FORCE_FORCE_SESSION);
126         endpoint = cookieValueMap.get(SecurityContextUtil.FORCE_FORCE_ENDPOINT);
127         
128         //if we found a security context in the session we must make sure that it still
129         //matches the values from the cookies. Otherwise throw it away so that we can
130         //refresh it.
131         if (sc != null) {
132             if (sessionId == null || !sessionId.equals(sc.getSessionId())
133                     || endpoint == null || !endpoint.equals(sc.getEndPoint())) {
134                 sc = null;
135             }
136         }
137         
138         //populate the security context with user information
139         if (sessionId != null && endpoint != null && sc == null) {
140             //attempt to connect to the partner api and retrieve the user data
141             //if this fails then set the security context to null because we'll
142             //need to redo the oauth handshake.
143             try {
144                 sc = userDataRetrievalService.retrieveUserData(sessionId, endpoint, null);
145             } catch (ConnectionException e) {
146                 sc = null;
147             }
148         }
149         
150         return sc;
151     }
152 
153     /**
154      * Clear the security context from the security context store and use the
155      * SecurityContextUtil to clear the other security related cookies.
156      * {@inheritDoc}
157      */
158     @Override
159     public void clearSecurityContext(HttpServletRequest request, HttpServletResponse response) {
160         securityContextStorageService.clearSecurityContext(request, response);
161         SecurityContextUtil.clearCookieValues(response);
162     }
163 
164     @Override
165     public SecretKeySpec getSecretKey() throws ForceEncryptionException {
166         return securityContextStorageService.getSecureKey();
167     }
