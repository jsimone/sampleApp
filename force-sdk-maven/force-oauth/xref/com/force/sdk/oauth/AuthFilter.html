46  /**
47   * This filter can be used to add Force.com OAuth Authentication to any web application. When configuring web.xml all
48   * requests that need to be authenticated should be sent through AuthFilter. The OAuth callback (usually _auth) must
49   * also be sent through AuthFilter. To use the connector, add the following servlet filter to your application's web.xml
50   * file:
51   * <p>
52   * {@code
53   * 
54   * <!-- Enables Security -->
55   * <filter>
56   *     <filter-name>AuthFilter</filter-name>
57   *     <filter-class>com.force.sdk.oauth.AuthFilter</filter-class>
58   *          <init-param>
59   *             <param-name>connectionName</param-name>
60   *             <param-value>nameOfConnectionToUse</param-value>
61   *         </init-param>
62   * </filter>
63   * <filter-mapping>
64   *     <filter-name>AuthFilter</filter-name>
65   *     <url-pattern>/*</url-pattern>
66   * </filter-mapping>
67   * 
68   * }
69   * <p>
70   * The OAuth Connector uses the Force.com API Connector to access the Force.com APIs. The connectionName is used to look
71   * up OAuth properties defined in an environment variable, or a Java system property, or in a properties file on the
72   * classpath. See @doclink connection-url for more information. Other init parameters that can be set are:
73   * <ul>
74   * <li>securityContextStorageMethod - valid values are "cookie" or "session". Defaults to "cookie". See @doclink
75   * force-security for more information on session management and security</li>
76   * <li>secure-key-file - specify the location of the file where your AES secure key is stored.</li> For Cookie based
77   * session management.
78   * </ul>
79   * 
80   * @author Fiaz Hossain
81   * @author John Simone
82   */
83  public class AuthFilter implements Filter, SessionRenewer {

88      static final String DEFAULT_USER_PROFILE = "myProfile";
89      static final String CONTEXT_STORE_SESSION_VALUE = "session";

94      /**
95       * Initialize the filter from the init params.
96       * {@inheritDoc} 
97       */
98      @Override
99      public void init(FilterConfig config) throws ServletException {
100 
101         SecurityContextServiceImpl securityContextServiceImpl = new SecurityContextServiceImpl();
102 
103         String customDataRetrieverName = config.getInitParameter("customDataRetriever");
104         boolean storeUsername = true;
105 
106         if ("false".equals(config.getInitParameter("storeUsername"))) {
107             storeUsername = false;
108         }
109 
110         UserDataRetrievalService userDataRetrievalService = null;
111 
112         if (customDataRetrieverName != null) {
113             try {
114                 Class<?> customDataRetrievalClass = Class.forName(customDataRetrieverName);
115                 Object customDataRetrievalObject = customDataRetrievalClass.newInstance();
116 
117                 if (customDataRetrievalObject instanceof CustomUserDataRetriever) {
118                     CustomUserDataRetriever<?> customDataRetriever = (CustomUserDataRetriever<?>) customDataRetrievalObject;
119                     userDataRetrievalService = new CustomUserDataRetrievalService(customDataRetriever, storeUsername);
120                 }
121 
122             } catch (ClassNotFoundException e) {
123                 throw new ServletException("Custom user data retriever class not found: " + customDataRetrieverName, e);
124             } catch (InstantiationException e) {
125                 throw new ServletException("Custom user data retriever class could not be instantiated: "
126                         + customDataRetrieverName, e);
127             } catch (IllegalAccessException e) {
128                 throw new ServletException("Custom user data retriever class could not be instantiated: "
129                         + customDataRetrieverName, e);
130             }
131         } else {
132             userDataRetrievalService = new UserDataRetrievalService(storeUsername);
133         }
134 
135         // Now that the data retrieval service is created set the
136         securityContextServiceImpl.setUserDataRetrievalService(userDataRetrievalService);
137         oauthConnector = new ForceOAuthConnector(userDataRetrievalService);
138 
139         // Build a ForceOAuthConnectionInfo object, if applicable
140         ForceOAuthConnectionInfo connInfo = null;
141         if (config.getInitParameter("endpoint") != null) {
142             connInfo = new ForceOAuthConnectionInfo();
143             connInfo.setEndpoint(config.getInitParameter("endpoint"));
144             connInfo.setOauthKey(config.getInitParameter("oauthKey"));
145             connInfo.setOauthSecret(config.getInitParameter("oauthSecret"));
146             oauthConnector.setConnectionInfo(connInfo);
147         } else if (config.getInitParameter("url") != null) {
148             connInfo = new ForceOAuthConnectionInfo();
149             connInfo.setConnectionUrl(config.getInitParameter("url"));
150             oauthConnector.setConnectionInfo(connInfo);
151         } else if (config.getInitParameter("connectionName") != null) {
152             oauthConnector.setConnectionName(config.getInitParameter("connectionName"));
153         } else {
154             throw new IllegalArgumentException("Could not find any init state for AuthFilter. "
155                     + "Please specify an endpoint, oauthKey and oauthSecret or a connection url or a connection name.");
156         }
157 
158         if (CONTEXT_STORE_SESSION_VALUE.equals(config.getInitParameter("securityContextStorageMethod"))) {
159             securityContextServiceImpl.setSecurityContextStorageService(new SecurityContextSessionStore());
160         } else {
161             SecurityContextCookieStore cookieStore = new SecurityContextCookieStore();
162 
163             try {
164                 cookieStore.setKeyFileName(config.getInitParameter("secure-key-file"));
165             } catch (ForceEncryptionException e) {
166                 throw new ServletException(e);
167             }
168 
169             securityContextServiceImpl.setSecurityContextStorageService(cookieStore);
170         }
171 
172         securityContextService = securityContextServiceImpl;
173     }

175     /**
176      * Handle the secured requests.
177      * {@inheritDoc} 
178      */
179     @Override
180     public void doFilter(ServletRequest sreq, ServletResponse sres, FilterChain chain) throws IOException,
181             ServletException {
182         HttpServletRequest request = (HttpServletRequest) sreq;
183         HttpServletResponse response = (HttpServletResponse) sres;
184 
185         if (request.getAttribute(FILTER_ALREADY_VISITED) != null) {
186             // ensure we do not get into infinite loop here
187             chain.doFilter(request, response);
188             return;
189         }
190 
191         SecurityContext sc = null;
192 
193         // if this isn't the callback from an OAuth handshake
194         // get the security context from the session
195         if (!ForceOAuthConnector.REDIRECT_AUTH_URI.equals(request.getServletPath())) {
196             sc = securityContextService.getSecurityContextFromSession(request);
197         }
198 
199         // if there is no valid security context then initiate an OAuth handshake
200         if (sc == null) {
201             doOAuthLogin(request, response);
202             return;
203         } else {
204             securityContextService.setSecurityContextToSession(request, response, sc);
205         }
206         ForceSecurityContextHolder.set(sc);
207         
208         ForceConnectorConfig cc = new ForceConnectorConfig();
209         cc.setSessionId(sc.getSessionId());
210         cc.setServiceEndpoint(sc.getEndPoint());
211         cc.setSessionRenewer(this);
212 
213         try {
214             ForceServiceConnector.setThreadLocalConnectorConfig(cc);
215             request.setAttribute(FILTER_ALREADY_VISITED, Boolean.TRUE);
216             chain.doFilter(new AuthenticatedRequestWrapper(request, sc), response);
217         } catch (ForceOAuthSessionExpirationException e) {
218             doOAuthLogin(request, response);
219         } catch (SecurityException se) {
220             response.sendError(HttpServletResponse.SC_FORBIDDEN, request.getRequestURI());
221         } finally {
222             try {
223                 request.removeAttribute(FILTER_ALREADY_VISITED);
224             } finally {
225                 ForceSecurityContextHolder.release();
226                 ForceServiceConnector.setThreadLocalConnectorConfig(null);
227             }
228         }
229     }

231     /**
232      * Send the authentication redirect or save the security context to the session depending on which phase of the
233      * handshake we're in.
234      * 
235      * @param request
236      * @param response
237      * @throws IOException
238      */
239     private void doOAuthLogin(HttpServletRequest request, HttpServletResponse response) throws IOException {
240         if (ForceOAuthConnector.REDIRECT_AUTH_URI.equals(request.getServletPath())) {
241             securityContextService.setSecurityContextToSession(
242                     request,
243                     response,
244                     oauthConnector.getAccessToken(oauthConnector.getAccessCode(request),
245                             oauthConnector.getRedirectUri(request)));
246             // response.sendRedirect(URLEncoder.encode(request.getParameter("state"), "UTF-8"));
247             response.sendRedirect(response.encodeRedirectURL(URLEncoder.encode(request.getParameter("state"), "UTF-8")));
248         } else {
249             response.sendRedirect(oauthConnector.getLoginRedirectUrl(request));
250         }
251     }

263     /**
264      * Wrap the request and provide methods that will make the authenticated user information available.
265      */
266     private static final class AuthenticatedRequestWrapper extends HttpServletRequestWrapper {
267 
268         private final ForceUserPrincipal userP;
269         private final ForceRolePrincipal roleP;
270 
271         public AuthenticatedRequestWrapper(HttpServletRequest request, SecurityContext sc) {
272             super(request);
273             this.userP = new ForceUserPrincipal(sc.getUserName(), sc.getSessionId());
274             this.roleP = new ForceRolePrincipal(sc.getRole());
275         }
276 
277         @Override
278         public String getRemoteUser() {
279             return userP != null ? userP.getName() : super.getRemoteUser();
280         }
281 
282         @Override
283         public Principal getUserPrincipal() {
284             return userP != null ? userP : super.getUserPrincipal();
285         }
286 
287         @Override
288         public boolean isUserInRole(String role) {
289             return roleP != null ? roleP.getName().endsWith(role) : super.isUserInRole(role);
290         }
291     }
