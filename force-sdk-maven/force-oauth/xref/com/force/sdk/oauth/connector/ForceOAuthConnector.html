44  /**
45   * 
46   * This is the main actor in the OAuth handshake. The other pieces of the OAuth integration will
47   * ultimately rely on the ForceOAuthConnector to handle the details of the OAuth flow.
48   * <p>
49   * In addition to the usual OAuth 2.0 protocol a UserDataRetrievalService is passed in and used to retrieve
50   * data about the authenticated user after the handshake completes. This service can either be the standard 
51   * service or an application specific one. This allows user data to be made available to the application.
52   * <p>
53   * This object is used from a per application singleton. DO NOT store per request values with this class.

56   * @author Tim Kral

126     /**
127      * Get the access token for user. This gets called after an access code has been obtained.
128      * That access code will now be exchanged for an access token or "auth token". Once an 
129      * access token is obtained it will immediately be used to retrieve data about the user to
130      * populate a SecurityContext.
131      * 
132      * @param accessCode String
133      * @param redirectUri String
134      * @return a SecurityContext containing data about the authenticated user
135      * @throws IOException i/o error
136      */
137     public SecurityContext getAccessToken(String accessCode, String redirectUri) throws IOException {
138         StringBuffer urlParams = new StringBuffer("grant_type=authorization_code")
139         .append("&code=").append(accessCode)
140         .append("&redirect_uri=").append(URLEncoder.encode(redirectUri, "UTF-8"));
141 
142         getConnInfo().appendOauthKeyParam(urlParams);
143         getConnInfo().appendOauthSecretParam(urlParams);
144         
145         return createTokenInternal(urlParams.toString(), null /* refresh token */);
146     }
147     
148     /**
149      * Use the refresh token to obtain a new auth token for the user.
150      * 
151      * @param refreshToken String
152      * @return a SecurityContext containing data about the authenticated user
153      * @throws IOException i/o error
154      */
155     public SecurityContext refreshAccessToken(String refreshToken) throws IOException {
156         StringBuffer urlParams = new StringBuffer("grant_type=refresh_token")
157         .append("&refresh_token=").append(refreshToken);
158 
159         getConnInfo().appendOauthKeyParam(urlParams);
160         getConnInfo().appendOauthSecretParam(urlParams);
161         
162         return createTokenInternal(urlParams.toString(), refreshToken);
163     }

165     /**
166      * Obtain an access token by calling the OAuth authentication endpoint and either trading an 
167      * access code or refresh token for it.
168      * 
169      * The UserDataRetrievalService is called to retrieve data about the user after the access token is obtained.
170      * 
171      * @param params String
172      * @param refreshToken String
173      * @return SecurityContext containing data about the authenticated user
174      * @throws IOException i/o error
175      */
176     @SuppressWarnings("unchecked")
177     private SecurityContext createTokenInternal(String params, String refreshToken) throws IOException {
178         String instanceUrl = null;
179         String session = null;
180         String refresh = refreshToken;
181         try {
182             
183             String responsePayload = tokenRetrievalService.retrieveToken(
184                     getHostPort(connInfo.getEndpoint(), null), params, refreshToken, getConnInfo());
185             
186             JSONParser parser = new JSONParser();
187             Map<String, String> map = (Map) parser.parse(responsePayload);
188             
189             if ((session = map.get(ACCESS_TOKEN_JSON_KEY)) == null) {
190                 throw new IOException("Missing access token on response");
191             }
192             
193             if (refresh == null) {
194                 if ((refresh = map.get(REFRESH_TOKEN_JSON_KEY)) == null) {
195                     throw new IOException("Missing refresh token on response");
196                 }
197             }
198             
199             
200             if ((instanceUrl = map.get(INSTANCE_URL_JSON_KEY)) == null) {
201                 throw new IOException("Missing instance url on response");
202             }
203             
204             SecurityContext sc =
205                 userDataRetrievalService.retrieveUserData(
206                         session, ForceConnectorUtils.buildForceApiEndpoint(instanceUrl), refresh);
207             
208             return sc;
209         } catch (ConnectionException ce) {
210             throw new IOException("Unable to create token due to connection exception", ce);
211         } catch (ParseException pe) {
212             throw new IOException("Unable to create token due to parse exception", pe);
213         }
214     }

238      * Build the logout url.
239      * 
240      * @param request HttpServletRequest
241      * @param forceEndpoint String
242      * @param localLogoutSuccessfulPath String
243      * @return the logout url
244      */
245     public String getForceLogoutUrl(HttpServletRequest request, String forceEndpoint, String localLogoutSuccessfulPath) {
246         
247         // Replace the force endpoint path with the logout path
248         StringBuffer forceLogoutUrl =
249             new StringBuffer(forceEndpoint.substring(0, forceEndpoint.indexOf('/', 9)) + "/secur/logout.jsp");
250         if (localLogoutSuccessfulPath != null && localLogoutSuccessfulPath.length() > 0) {
251             try {
252                 forceLogoutUrl.append("?retUrl=").append(getHostPort(request))
253                         .append(request.getContextPath())
254                         .append(URLEncoder.encode(localLogoutSuccessfulPath, "UTF-8"));
255                 
256                 getConnInfo().appendOauthKeyParam(forceLogoutUrl);
257             } catch (IOException e) {
258                 // log the error
259             }
260         }
261         
262         return forceLogoutUrl.toString();
263     }

277         return getHostPort(host, request.getScheme());

280     /**
281      * Returns the host and port of the endpoint.
282      * 
283      * @param endpoint String
284      * @param protocol String
285      * @return the host and port string
286      */
287     public String getHostPort(String endpoint, String protocol) {
288         String[] parsedEndpoint = endpoint.split("://", 2);
289         String endpointWithoutProtocol = parsedEndpoint[parsedEndpoint.length - 1];
290         
291         if (protocol == null && parsedEndpoint.length == 2) {
292             protocol = parsedEndpoint[0];
293         }
294         
295         if (endpointWithoutProtocol.startsWith("localhost") || endpointWithoutProtocol.contains("internal")) {
296             if (protocol != null) return protocol + "://" + endpointWithoutProtocol;
297             
298             return "http://" + endpointWithoutProtocol;
299         }
300         
301         return "https://" + endpointWithoutProtocol;
302     }

304     /**
305      * get the url that the user will be redirected to for authentication. This url should send the user to a login
306      * screen so that they can enter their credentials. Once the user sucessfully authenticates a callback will 
307      * be received with the access code.
308      * 
309      * @param request HttpServletRequest
310      * @return the url to redirect the user to
311      * @throws IOException i/o error
312      */
313     public String getLoginRedirectUrl(HttpServletRequest request) throws IOException {
314         String redirectAttribute = (String) request.getAttribute(LOGIN_REDIRECT_URL_ATTRIBUTE);
315         
316         // Build the state url parameter
317         StringBuffer state;
318         if (redirectAttribute != null) {
319             state = new StringBuffer(redirectAttribute);
320         } else {
321             
322             // host:post/contextPath/servletPath
323             state = new StringBuffer(getHostPort(request)).append(request.getContextPath())
324                             .append(request.getServletPath());
325             
326             if (request.getPathInfo() != null)
327                 state.append(request.getPathInfo());
328         }
329         
330         // Build the login redirect url
331         StringBuffer loginRedirectUrl =
332             new StringBuffer(getHostPort(getConnInfo().getEndpoint(), null)).append("/services/oauth2/authorize")
333                 .append("?response_type=code")
334                 .append("&redirect_uri=").append(URLEncoder.encode(getRedirectUri(request), "UTF-8"))
335                 .append("&state=").append(URLEncoder.encode(state.toString(), "UTF-8"));
336     
337         // Add the oauth key parameter
338         getConnInfo().appendOauthKeyParam(loginRedirectUrl);
339         
340         return loginRedirectUrl.toString();
341     }

353     /**
354      * Creates a connection info object. This returns the same output regardless
355      * of whether the connector was supplied with a connection name that must be
356      * resolved or directly given the oauth key, secret, and endpoint.
357      * 
358      * @return The force oauth connection info
359      * @throws IOException i/o error
360      */
361     private ForceOAuthConnectionInfo getConnInfo() throws IOException {
362         if (connInfo == null) {
363             if (this.externalConnInfo != null) {
364                 connInfo = this.externalConnInfo;
365             } else if (this.connectionName != null) {
366                 connInfo = ForceOAuthConnectionInfo.loadFromName(this.connectionName);
367             }
368             
369             // Out of options.  There's not enough here to construct theConnInfo.
370             if (connInfo == null) {
371                 StringBuffer errorMsg = new StringBuffer();
372                 errorMsg.append("No state was found to construct an oauth connection.")
373                         .append(" Please provide an endpoint, key and secret or connection url.");
374             
375                 if (this.connectionName != null) {
376                     errorMsg
377                         .append(" Or create a classpath properties file, environment variable or java property for the name '")
378                         .append(this.connectionName).append("'");
379                 }
380             
381                 throw new IOException(errorMsg.toString());
382             }
383             
384             connInfo.validate();
385         }
386         
387         return connInfo;
388     }
